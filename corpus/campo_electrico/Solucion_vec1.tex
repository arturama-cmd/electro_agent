\section*{Enunciado del Problema}

Dados los vectores $\vec{A}$, $\vec{B}$ y $\vec{C}$, determine:
\begin{enumerate}
    \item $\left\Vert 2\vec{A}-\sqrt{3}\vec{B}+0.361\vec{C}\right\Vert$
    \item $\vec{D} = \dfrac{\vec{A}}{\sqrt{3}}-\left( \dfrac{\vec{B}}{2}+\dfrac{\vec{C}}{\sqrt{\sin(20^\circ)}}\right)$
\end{enumerate}

Donde:
\begin{eqnarray*}
\vec{A} &=& 3\hat{\imath}+8\hat{\jmath}+5\hat{k} \\
\vec{B} &=& -5\hat{\imath}+2\hat{\jmath}-3\hat{k} \\
\vec{C} &=& 4\hat{\imath}-7\hat{\jmath}-2\hat{k}
\end{eqnarray*}

\section*{Parte 1: C\'alculo de $\left\Vert 2\vec{A}-\sqrt{3}\vec{B}+0.361\vec{C}\right\Vert$}

\subsection*{Paso 1: Calcular $2\vec{A}$}

\begin{equation*}
2\vec{A} = 2(3\hat{\imath}+8\hat{\jmath}+5\hat{k}) = \boxed{6\hat{\imath}+16\hat{\jmath}+10\hat{k}}
\end{equation*}

\subsection*{Paso 2: Calcular $\sqrt{3}\vec{B}$}

\begin{equation*}
\sqrt{3}\vec{B} = \sqrt{3}(-5\hat{\imath}+2\hat{\jmath}-3\hat{k}) = -5\sqrt{3}\hat{\imath}+2\sqrt{3}\hat{\jmath}-3\sqrt{3}\hat{k}
\end{equation*}

Evaluando num\'ericamente ($\sqrt{3} \approx 1.7321$):
\begin{equation*}
\sqrt{3}\vec{B} \approx \boxed{-8.6603\hat{\imath}+3.4641\hat{\jmath}-5.1962\hat{k}}
\end{equation*}

\subsection*{Paso 3: Calcular $0.361\vec{C}$}

\begin{equation*}
0.361\vec{C} = 0.361(4\hat{\imath}-7\hat{\jmath}-2\hat{k}) = \boxed{1.444\hat{\imath}-2.527\hat{\jmath}-0.722\hat{k}}
\end{equation*}

\subsection*{Paso 4: Calcular el vector resultante $\vec{R} = 2\vec{A}-\sqrt{3}\vec{B}+0.361\vec{C}$}

Sumando componente a componente:

\textbf{Componente $\hat{\imath}$:}
\begin{equation*}
R_x = 6 - (-8.6603) + 1.444 = 6 + 8.6603 + 1.444 = \mathbf{16.104}
\end{equation*}

\textbf{Componente $\hat{\jmath}$:}
\begin{equation*}
R_y = 16 - 3.4641 + (-2.527) = 16 - 3.4641 - 2.527 = \mathbf{10.009}
\end{equation*}

\textbf{Componente $\hat{k}$:}
\begin{equation*}
R_z = 10 - (-5.1962) + (-0.722) = 10 + 5.1962 - 0.722 = \mathbf{14.474}
\end{equation*}

Por lo tanto:
\begin{equation*}
\vec{R} = 2\vec{A}-\sqrt{3}\vec{B}+0.361\vec{C} = 16.104\hat{\imath}+10.009\hat{\jmath}+14.474\hat{k}
\end{equation*}

\subsection*{Paso 5: Calcular la magnitud}

\begin{equation*}
\left\Vert \vec{R} \right\Vert = \sqrt{R_x^2 + R_y^2 + R_z^2}
\end{equation*}

\begin{equation*}
\left\Vert \vec{R} \right\Vert = \sqrt{(16.104)^2 + (10.009)^2 + (14.474)^2}
\end{equation*}

\begin{equation*}
\left\Vert \vec{R} \right\Vert = \sqrt{259.339 + 100.180 + 209.496} = \sqrt{569.015}
\end{equation*}

\begin{equation*}
\boxed{\left\Vert 2\vec{A}-\sqrt{3}\vec{B}+0.361\vec{C}\right\Vert \approx 23.854}
\end{equation*}

%========================================
% PARTE 2: VECTOR D (CORREGIDO CON SIGNO MENOS)
%========================================
\section*{Parte 2: C\'alculo de $\vec{D} = \dfrac{\vec{A}}{\sqrt{3}}-\left( \dfrac{\vec{B}}{2}+\dfrac{\vec{C}}{\sqrt{\sin(20^\circ)}}\right)$}

\textbf{Nota:} Observe el signo \textcolor{red}{\textbf{negativo}} antes del par\'entesis.

\subsection*{Paso 1: Calcular constantes}

\begin{itemize}
    \item $\sqrt{3} \approx 1.7321$
    \item $\sin(20^\circ) \approx 0.3420$
    \item $\sqrt{\sin(20^\circ)} \approx \sqrt{0.3420} \approx 0.5848$
\end{itemize}

\subsection*{Paso 2: Calcular $\dfrac{\vec{A}}{\sqrt{3}}$}

\begin{equation*}
\frac{\vec{A}}{\sqrt{3}} = \frac{1}{1.7321}(3\hat{\imath}+8\hat{\jmath}+5\hat{k}) = \boxed{1.732\hat{\imath}+4.619\hat{\jmath}+2.887\hat{k}}
\end{equation*}

\subsection*{Paso 3: Calcular $\dfrac{\vec{B}}{2}$}

\begin{equation*}
\frac{\vec{B}}{2} = \frac{1}{2}(-5\hat{\imath}+2\hat{\jmath}-3\hat{k}) = \boxed{-2.5\hat{\imath}+1.0\hat{\jmath}-1.5\hat{k}}
\end{equation*}

\subsection*{Paso 4: Calcular $\dfrac{\vec{C}}{\sqrt{\sin(20^\circ)}}$}

\begin{equation*}
\frac{\vec{C}}{\sqrt{\sin(20^\circ)}} = \frac{1}{0.5848}(4\hat{\imath}-7\hat{\jmath}-2\hat{k}) = \boxed{6.840\hat{\imath}-11.970\hat{\jmath}-3.420\hat{k}}
\end{equation*}

\subsection*{Paso 5: Calcular $\dfrac{\vec{B}}{2}+\dfrac{\vec{C}}{\sqrt{\sin(20^\circ)}}$}

\begin{equation*}
\frac{\vec{B}}{2}+\frac{\vec{C}}{\sqrt{\sin(20^\circ)}} = (-2.5+6.840)\hat{\imath}+(1.0-11.970)\hat{\jmath}+(-1.5-3.420)\hat{k}
\end{equation*}

\begin{equation*}
= \boxed{4.340\hat{\imath}-10.970\hat{\jmath}-4.920\hat{k}}
\end{equation*}

\subsection*{Paso 6: Calcular $\vec{D}$ (RESTANDO el par\'entesis)}

\begin{equation*}
\vec{D} = \frac{\vec{A}}{\sqrt{3}}\textcolor{red}{-}\left( \frac{\vec{B}}{2}+\frac{\vec{C}}{\sqrt{\sin(20^\circ)}}\right)
\end{equation*}

\begin{equation*}
\vec{D} = (1.732\textcolor{red}{-}4.340)\hat{\imath}+(4.619\textcolor{red}{-}(-10.970))\hat{\jmath}+(2.887\textcolor{red}{-}(-4.920))\hat{k}
\end{equation*}

\begin{equation*}
\vec{D} = (1.732-4.340)\hat{\imath}+(4.619+10.970)\hat{\jmath}+(2.887+4.920)\hat{k}
\end{equation*}

\begin{equation*}
\boxed{\vec{D} = -2.608\hat{\imath}+15.589\hat{\jmath}+7.807\hat{k}}
\end{equation*}

\subsection*{Verificaci\'on: Magnitud de $\vec{D}$}

\begin{equation*}
\left\Vert \vec{D} \right\Vert = \sqrt{(-2.608)^2 + (15.589)^2 + (7.807)^2}
\end{equation*}

\begin{equation*}
\left\Vert \vec{D} \right\Vert = \sqrt{6.802 + 243.018 + 60.949} = \sqrt{310.769} \approx \boxed{17.629}
\end{equation*}

\section*{Interpretaci\'on F\'isica}

Este tipo de operaciones vectoriales son fundamentales en electromagnetismo:

\begin{itemize}
    \item La \textbf{combinaci\'on lineal de vectores} aparece al calcular campos el\'ectricos o magn\'eticos resultantes de m\'ultiples fuentes.

    \item La \textbf{magnitud de un vector} representa cantidades f\'isicas como la intensidad del campo el\'ectrico $|\vec{E}|$ o la fuerza neta $|\vec{F}|$ sobre una carga.

    \item Las \textbf{operaciones con signos} (suma y resta) determinan si los efectos se suman constructivamente o se cancelan parcialmente, como ocurre con fuerzas de atracci\'on y repulsi\'on.

    \item Los \textbf{coeficientes escalares} pueden representar:
    \begin{itemize}
        \item Magnitudes de cargas el\'ectricas
        \item Factores geom\'etricos (distancias, \'angulos)
        \item Constantes f\'isicas
    \end{itemize}
\end{itemize}
