\section*{Enunciado del Problema}

Una l\'inea de carga con densidad lineal uniforme $\lambda = 4$ [nC/m] se encuentra sobre el eje $y$, entre $y = 0$ e $y = 30$ [cm]. Adem\'as hay una carga puntual negativa $q = -5$ [mC] sobre el eje $x$, a 50 cm del origen. Determinar el campo el\'ectrico total en el punto $P = (0{,}15;\;0)$ [m].

%========================================
% DATOS NUMERICOS
%========================================
\section*{Datos Num\'ericos}

\begin{itemize}
    \item $K = 9{,}00\times10^9\;\text{N}\cdot\text{m}^2/\text{C}^2$
    \item $\lambda = 4{,}00\;\text{nC/m} = 4{,}00\times10^{-9}\;\text{C/m}$
    \quad$\Rightarrow$\quad
    $K\lambda = 9{,}00\times10^9\times4{,}00\times10^{-9} = 36{,}0\;\text{N}\cdot\text{m}/\text{C}$
    \item L\'inea de carga: $0 \le y \le 0{,}30\;\text{m}$ sobre el eje $y$
    \item $\vec{r}_P = 0{,}15\,\hat{\imath}\;\text{[m]}$ \quad$\Rightarrow$\quad $a^2 = 0{,}15^2 = 0{,}0225\;\text{m}^2$
    \item $q = -5{,}00\;\text{mC} = -5{,}00\times10^{-3}\;\text{C}$ en $(0{,}50;\,0)$ m
\end{itemize}

%========================================
% ANALISIS CUALITATIVO
%========================================
\section*{An\'alisis Cualitativo}

El campo total en $P$ tiene dos contribuciones:

\begin{itemize}
    \item \textbf{L\'inea de carga} ($\vec{E}_1$): la l\'inea est\'a sobre el eje $y$, a la \textbf{izquierda} de $P$ ($x=0$ frente a $x_P = 0{,}15$ m). Cada elemento $dq$ contribuye con una componente $+\hat{\imath}$ (empuja hacia la derecha). La componente $\hat{\jmath}$ no se cancela por ser una l\'inea finita: los elementos en $y > 0$ empujan hacia $-\hat{\jmath}$, de modo que $E_{1y} < 0$.

    \item \textbf{Carga puntual} ($\vec{E}_2$): $q < 0$ est\'a a la \textbf{derecha} de $P$ ($x=0{,}50$ m). El campo de una carga negativa apunta \textbf{hacia} la carga, es decir en $+\hat{\imath}$, contribuyendo positivamente y dominando por su gran magnitud.
\end{itemize}

%========================================
% PASO 1: CAMPO DE LA LINEA DE CARGA
%========================================
\section*{Paso 1: Campo de la L\'inea de Carga $\vec{E}_1$}

\subsection*{Vectores de posici\'on}

El elemento en la l\'inea est\'a en $\vec{r}_{q} = y\,\hat{\jmath}$, el punto de campo en $\vec{r}_{P} = 0{,}15\,\hat{\imath}$:
\begin{equation*}
    \vec{r}_{P} - \vec{r}_{q} = 0{,}15\,\hat{\imath} - y\,\hat{\jmath}
    \qquad
    \|\vec{r}_{P}-\vec{r}_{q}\|^3 = \bigl(0{,}15^2 + y^2\bigr)^{3/2}
    = \bigl(0{,}0225 + y^2\bigr)^{3/2}
\end{equation*}

\subsection*{Planteamiento de la integral}

\begin{equation*}
    \vec{E}_1
    = K\lambda\int_0^{0{,}30}
    \frac{0{,}15\,\hat{\imath} - y\,\hat{\jmath}}
         {(0{,}0225 + y^2)^{3/2}}\,dy
\end{equation*}

Separando por componente ($a^2 = 0{,}0225\;\text{m}^2$):
\begin{align*}
    E_{1x} &= K\lambda\,(0{,}15)
    \int_0^{0{,}30}\frac{dy}{(0{,}0225+y^2)^{3/2}}
    \\[6pt]
    E_{1y} &= -K\lambda
    \int_0^{0{,}30}\frac{y\,dy}{(0{,}0225+y^2)^{3/2}}
\end{align*}

\subsection*{Antiderivadas est\'andar}

\begin{equation*}
    \int\frac{dy}{(a^2+y^2)^{3/2}} = \frac{y}{a^2\sqrt{a^2+y^2}}+C
    \qquad\qquad
    \int\frac{y\,dy}{(a^2+y^2)^{3/2}} = \frac{-1}{\sqrt{a^2+y^2}}+C
\end{equation*}

\subsection*{Componente $\hat{\imath}$ de la l\'inea}

\begin{equation*}
    E_{1x} = K\lambda\,(0{,}15)
    \left[\frac{y}{0{,}0225\,\sqrt{0{,}0225+y^2}}\right]_0^{0{,}30}
\end{equation*}

Evaluando:
\begin{align*}
    y = 0{,}30:\quad
    &\frac{0{,}30}{0{,}0225\times\sqrt{0{,}0225+0{,}09}}
    = \frac{0{,}30}{0{,}0225\times\sqrt{0{,}1125}}
    = \frac{0{,}30}{0{,}0225\times0{,}33541}
    = \frac{0{,}30}{7{,}547\times10^{-3}}
    = 39{,}752\;\text{m}^{-1}\\[4pt]
    y = 0:\quad &\frac{0}{(\ldots)} = 0
\end{align*}

\begin{equation*}
    E_{1x} = 36{,}0\times0{,}15\times39{,}752 = 5{,}40\times39{,}752
    = 214{,}66\;\text{N/C}
\end{equation*}

\subsection*{Componente $\hat{\jmath}$ de la l\'inea}

\begin{equation*}
    E_{1y} = -K\lambda
    \left[\frac{-1}{\sqrt{0{,}0225+y^2}}\right]_0^{0{,}30}
    = K\lambda
    \left[\frac{1}{\sqrt{0{,}0225+y^2}}\right]_0^{0{,}30}
\end{equation*}

Evaluando:
\begin{align*}
    y = 0{,}30:\quad &\frac{1}{\sqrt{0{,}1125}} = \frac{1}{0{,}33541} = 2{,}981\;\text{m}^{-1}\\[4pt]
    y = 0:\quad &\frac{1}{\sqrt{0{,}0225}} = \frac{1}{0{,}15} = 6{,}667\;\text{m}^{-1}
\end{align*}

\begin{equation*}
    E_{1y} = K\lambda\,(2{,}981 - 6{,}667) = 36{,}0\times(-3{,}686) = -132{,}70\;\text{N/C}
\end{equation*}

\subsection*{Campo de la l\'inea}

\begin{equation*}
    \vec{E}_1 = 214{,}66\,\hat{\imath} - 132{,}70\,\hat{\jmath}\;\;\text{[N/C]}
\end{equation*}

%========================================
% PASO 2: CAMPO DE LA CARGA PUNTUAL
%========================================
\section*{Paso 2: Campo de la Carga Puntual $\vec{E}_2$}

La carga $q = -5{,}00\times10^{-3}$ C est\'a en $\vec{r}_q = 0{,}50\,\hat{\imath}$. El vector de la fuente al punto de campo:
\begin{equation*}
    \vec{r}_P - \vec{r}_q = (0{,}15 - 0{,}50)\,\hat{\imath} = -0{,}35\,\hat{\imath}
    \qquad
    \|\vec{r}_P - \vec{r}_q\|^3 = (0{,}35)^3 = 0{,}042875\;\text{m}^3
\end{equation*}

\begin{equation*}
    \vec{E}_2
    = K\,\frac{q\,(\vec{r}_P - \vec{r}_q)}{\|\vec{r}_P - \vec{r}_q\|^3}
    = 9{,}00\times10^9\,\frac{(-5{,}00\times10^{-3})\times(-0{,}35\,\hat{\imath})}{0{,}042875}
\end{equation*}

\begin{equation*}
    \vec{E}_2
    = 9{,}00\times10^9\times\frac{5{,}00\times10^{-3}\times0{,}35}{0{,}042875}\,\hat{\imath}
    = 9{,}00\times10^9\times\frac{1{,}75\times10^{-3}}{0{,}042875}\,\hat{\imath}
\end{equation*}

\begin{equation*}
    \boxed{\vec{E}_2 = 3{,}673\times10^{8}\,\hat{\imath}\;\;\text{[N/C]}}
\end{equation*}

La carga negativa est\'a a la derecha de $P$; su campo en $P$ apunta hacia ella ($+\hat{\imath}$), consistente con el an\'alisis cualitativo.

%========================================
% PASO 3: CAMPO TOTAL
%========================================
\section*{Paso 3: Campo El\'ectrico Total en $P$}

\begin{equation*}
    \vec{E}_T = \vec{E}_1 + \vec{E}_2
    = \bigl(214{,}66 + 3{,}673\times10^{8}\bigr)\,\hat{\imath} - 132{,}70\,\hat{\jmath}
\end{equation*}

El t\'ermino 214,66 N/C es despreciable frente a $3{,}673\times10^{8}$ N/C:

\begin{equation*}
    \boxed{\vec{E}_T = 3{,}673\times10^{8}\,\hat{\imath} - 132{,}70\,\hat{\jmath}\;\;\text{[N/C]}}
\end{equation*}

%========================================
% INTERPRETACION FISICA
%========================================
\section*{Interpretaci\'on F\'isica}

\begin{itemize}
    \item El campo de la carga puntual $|\vec{E}_2| = 3{,}673\times10^8$ N/C domina completamente sobre el de la l\'inea $|\vec{E}_1| \approx 250$ N/C: la raz\'on es $\approx 1{,}5\times10^6$. Esto se debe a que $|q| = 5$ mC es enormemente mayor que $\lambda\cdot L = 4\;\text{nC/m}\times0{,}30\;\text{m} = 1{,}2\;\text{nC}$.

    \item La componente $\hat{\jmath}$ del campo total proviene \textbf{\'unicamente} de la l\'inea de carga, pues la carga puntual est\'a sobre el eje $x$ y su campo en $P$ (tambi\'en sobre el eje $x$) no tiene componente $\hat{\jmath}$.

    \item El signo negativo de $E_{1y}$ indica que la l\'inea, ubicada sobre el eje $+y$, empuja el campo en $P$ hacia $-\hat{\jmath}$ (hacia abajo), lo que es f\'isicamente correcto: la carga positiva de la l\'inea repele en direcci\'on contraria.
\end{itemize}
