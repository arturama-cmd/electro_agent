%========================================
% ENUNCIADO
%========================================
\section*{Enunciado del Problema}

Considere una carga puntual $Q$ ubicada en el centro de un cascar\'on esf\'erico conductor de
radio interior $a$ y radio exterior $b$, con carga total $-2Q$ distribuida uniformemente en su
volumen. Considerando la permitividad el\'ectrica en el vac\'io $\varepsilon_0$ y utilizando la
Ley de Gauss, encontrar:

\begin{enumerate}[(a)]
    \item \textbf{(12~pts)} La magnitud del campo el\'ectrico en las regiones:
          (i)~$r < a$,\quad (ii)~$a < r < b$,\quad (iii)~$r > b$.
    \item \textbf{(4~pts)} Si $a = 10$~[cm], $b = 15$~[cm] y $Q = 110$~[$\mu$C],
          calcular la magnitud del campo el\'ectrico a $r = 8$~[cm].
    \item \textbf{(3~pts)} Para los mismos valores de $a$ y $b$, encontrar la densidad de
          carga volum\'etrica del cascar\'on esf\'erico.
\end{enumerate}

\section*{Datos del Problema}

\[
    Q \quad (\text{carga puntual en el centro}),
    \qquad
    q_\text{cas} = -2Q \quad (\text{cascar\'on conductor, radios }a\text{ y }b)
\]
\[
    \varepsilon_0 = 8{,}85 \times 10^{-12}
    \ \left[\frac{\text{C}^2}{\text{N}\cdot\text{m}^2}\right]
\]

%========================================
% PARTE a)
%========================================
\section*{Parte a): Magnitud del Campo El\'ectrico}

Por simetr\'ia esf\'erica, $\vec{E} = E(r)\,\hat{r}$ es constante en m\'odulo sobre cualquier
superficie esf\'erica de radio $r$. La Ley de Gauss establece:

\[
    \oint \vec{E}\cdot d\vec{A} = \frac{q_\text{enc}}{\varepsilon_0}
\]

El lado izquierdo se desarrolla en coordenadas esf\'ericas
($dA = r^2\operatorname{sen}\phi\,d\phi\,d\theta$):

\begin{align*}
    \oint \vec{E}\cdot d\vec{A}
        &= \int_0^{2\pi}\!\int_0^{\pi}
           E\,r^2\,\operatorname{sen}\phi\;d\phi\;d\theta \\[4pt]
        &= E\,r^2
           \int_0^{2\pi}\!d\theta
           \cdot
           \int_0^{\pi}\operatorname{sen}\phi\;d\phi \\[4pt]
        &= E\,r^2
           \cdot \bigl(\theta\bigr)\big|_0^{2\pi}
           \cdot \bigl(-\cos\phi\bigr)\big|_0^{\pi} \\[4pt]
        &= E\,r^2 \cdot 2\pi \cdot 2
         = 4\pi r^2\,E
\end{align*}

De modo que en \textbf{todas las regiones}:
\[
    E \cdot 4\pi r^2 = \frac{q_\text{enc}}{\varepsilon_0}
\]
Solo cambia $q_\text{enc}$ de una regi\'on a otra.

La superficie gaussiana (radio $r < a$) encierra \'unicamente la carga

\]
Sustituyendo en la Ley de Gauss:
\begin{align*}
    E \cdot 4\pi r^2 &= \frac{Q}{\varepsilon_0} \\[6pt]

\end{align*}

\[
    \boxed{\vec{E} = \frac{Q}{4\pi\varepsilon_0\,r^2}\,\hat{r} \quad [\text{N/C}]}
\]

\begin{minipage}[c]{0.67\textwidth}
La superficie gaussiana encierra la totalidad de las cargas:
\[
    q_\text{enc} = Q + (-2Q) = -Q
\]
Sustituyendo:
\begin{align*}
    E \cdot 4\pi r^2 &= \frac{-Q}{\varepsilon_0} \\[6pt]
    \vec{E}          &= \frac{-Q}{4\pi\varepsilon_0\,r^2}\,\hat{r}
\end{align*}
\end{minipage}

\[
    \boxed{\vec{E} = \frac{-Q}{4\pi\varepsilon_0\,r^2}\,\hat{r} \quad [\text{N/C}]}
\]

%========================================
% RESUMEN
%========================================
\section*{Resumen: Campo el\'ectrico seg\'un la regi\'on}

\[
\vec{E}(r) =
\begin{cases}
    \dfrac{Q}{4\pi\varepsilon_0\,r^2}\,\hat{r}    & r < a        \\[10pt]
    \vec{0}                                          & a < r < b    \\[10pt]
    \dfrac{-Q}{4\pi\varepsilon_0\,r^2}\,\hat{r}   & r > b
\end{cases}
\]

%========================================
% PARTE b)
%========================================
\section*{Parte b): C\'alculo Num\'erico}

Con $a = 0{,}10$~[m], $b = 0{,}15$~[m], $Q = 110$~[$\mu$C] y $r = 0{,}08$~[m].

Como $r < a$ (pues $0{,}08 < 0{,}10$~[m]), corresponde la expresi\'on de la regi\'on~(i):

\begin{align*}
    \vec{E} &= \frac{Q}{4\pi\varepsilon_0\,r^2}\,\hat{r} \\[8pt]
            &= \frac{110\times10^{-6}\ [\text{C}]}
                   {4\pi \times (0{,}08)^2\ [\text{m}^2]
                    \times 8{,}85\times10^{-12}
                    \left[\dfrac{\text{C}^2}{\text{N}\cdot\text{m}^2}\right]}\,\hat{r} \\[8pt]
            &= \frac{1{,}10\times10^{-4}}{7{,}117\times10^{-13}}\,\hat{r}
\end{align*}

\[
    \boxed{\vec{E} = 1{,}545 \times 10^{8}\,\hat{r}\ [\text{N/C}]}
\]

El coeficiente positivo de $\hat{r}$ indica que el campo apunta radialmente hacia afuera
(la carga encerrada en esta regi\'on es $Q > 0$).

%========================================
% PARTE c)
%========================================
\section*{Parte c): Densidad de Carga Volum\'etrica del Cascar\'on}

La densidad de carga volum\'etrica se obtiene dividiendo la carga total del cascar\'on
entre su volumen:

\[
    \rho = \frac{q_\text{cas}}{V_\text{cas}}
         = \frac{-2Q}{\dfrac{4}{3}\pi\!\left(b^3 - a^3\right)}
\]

Reemplazando con $a = 0{,}10$~[m], $b = 0{,}15$~[m] y $Q = 110$~[$\mu$C]:

\begin{align*}
    \rho &= \frac{-2 \times 110\times10^{-6}\ [\text{C}]}
                 {\dfrac{4}{3}\pi
                  \!\left[(0{,}15)^3 - (0{,}10)^3\right][\text{m}^3]} \\[10pt]
         &= \frac{-2{,}20\times10^{-4}}
                 {\dfrac{4}{3}\pi \times 2{,}375\times10^{-3}} \\[10pt]
         &= \frac{-2{,}20\times10^{-4}}{9{,}948\times10^{-3}}
\end{align*}

\[
    \boxed{\rho = -2{,}21\times10^{-2}\ [\text{C/m}^3]}
\]

El signo negativo es consistente con la carga negativa del cascar\'on.
