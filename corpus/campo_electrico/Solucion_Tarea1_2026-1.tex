\section*{\textcolor{ubbblue}{Problema 1 \;--\; Tres Cargas Puntuales}}

\begin{enumerate}[(a)]

\subsection*{Datos Num\'ericos}

\begin{itemize}
    \item $K = 9{,}00\times10^9\;\text{N}\cdot\text{m}^2/\text{C}^2$

    \item $q_2 = -31{,}5\times10^{-9}$ C;\quad $\vec{r}_2 = (0{,}05;\;0{,}02)$ m
    \item $q_3 = -10{,}3\times10^{-9}$ C;\quad $\vec{r}_3 = (-0{,}03;\;-0{,}04)$ m
    \item Punto de campo: $P = (0{,}01;\;0)$ m
\end{itemize}

\subsection*{An\'alisis Cualitativo}

\begin{itemize}
    \item $q_1 > 0$, $q_2 < 0$: $\vec{F}_{12}$ es \textbf{atractiva}; $q_2$ es jalada hacia
          $q_1$ (arriba-izquierda, pues $q_1$ est\'a en el segundo cuadrante, a la izquierda
          y arriba de $q_2$).
    \item $q_3 < 0$, $q_2 < 0$: $\vec{F}_{32}$ es \textbf{repulsiva}; $q_2$ es empujada lejos de
          $q_3$ (arriba-derecha, pues $q_3$ est\'a abajo-izquierda de $q_2$).
    \item En $P$: el campo de $q_2 < 0$ apunta \textbf{hacia} $q_2$ (derecha-arriba, pues
          $q_2$ est\'a a la derecha y arriba de $P$);
          el de $q_1 > 0$ se aleja de $q_1$ (derecha-abajo desde $P$, pues $q_1$ est\'a
          arriba-izquierda de $P$);
          el de $q_3 < 0$ apunta hacia $q_3$ (izquierda-abajo).
\end{itemize}

%-------- Parte (a) --------
\section*{Parte (a): Fuerza neta sobre $q_2$}

Se usa la Ley de Coulomb vectorial: $\vec{F}_{ij} = K\dfrac{q_i\,q_j}{\|\vec{r}_{ij}\|^3}\,\vec{r}_{ij}$
con $\vec{r}_{ij} = \vec{r}_j - \vec{r}_i$.

\subsection*{Fuerza de $q_1$ sobre $q_2$ \;($\vec{F}_{12}$)}

\begin{align*}
    \vec{r}_{12} &= \vec{r}_2-\vec{r}_1
                   = (0{,}05-(-0{,}02736);\;0{,}02-0{,}07518)
                   = (0{,}07736\,\hat{\imath}-0{,}05518\,\hat{\jmath})\;\text{m}\\
    \|\vec{r}_{12}\|^2 &= (0{,}07736)^2+(0{,}05518)^2
                         = 5{,}985\times10^{-3}+3{,}045\times10^{-3} = 9{,}030\times10^{-3}\;\text{m}^2\\
    \|\vec{r}_{12}\|^3 &= (9{,}030\times10^{-3})^{3/2}
                         = 9{,}030\times10^{-3}\times9{,}503\times10^{-2} = 8{,}582\times10^{-4}\;\text{m}^3\\
    K\,q_1\,q_2 &= 9{,}00\times10^9\times(11{,}8\times10^{-9})\times(-31{,}5\times10^{-9})
                 = -3{,}345\times10^{-6}\;\text{N\,m}^2
\end{align*}

\begin{align*}
    \vec{F}_{12}
    &= \frac{-3{,}345\times10^{-6}}{8{,}582\times10^{-4}}
       \,(0{,}07736\,\hat{\imath}-0{,}05518\,\hat{\jmath})\\
    &= -3{,}898\times10^{-3}\,(0{,}07736\,\hat{\imath}-0{,}05518\,\hat{\jmath})
\end{align*}

\begin{equation*}
    \boxed{\vec{F}_{12} = (-3{,}016\,\hat{\imath}+2{,}151\,\hat{\jmath})\times10^{-4}\;\text{[N]}}
\end{equation*}

El factor escalar es negativo ($q_1 q_2 < 0$): fuerza atractiva, apunta en sentido contrario a
$\vec{r}_{12}$, jalando $q_2$ hacia $q_1$.

\subsection*{Fuerza de $q_3$ sobre $q_2$ \;($\vec{F}_{32}$)}

\begin{align*}
    \vec{r}_{32} &= \vec{r}_2-\vec{r}_3
                   = (0{,}05-(-0{,}03);\;0{,}02-(-0{,}04))
                   = (0{,}08\,\hat{\imath}+0{,}06\,\hat{\jmath})\;\text{m}\\
    \|\vec{r}_{32}\|^2 &= (0{,}08)^2+(0{,}06)^2
                         = 6{,}400\times10^{-3}+3{,}600\times10^{-3} = 1{,}000\times10^{-2}\;\text{m}^2\\
    \|\vec{r}_{32}\|^3 &= (1{,}000\times10^{-2})^{3/2}
                         = 1{,}000\times10^{-2}\times1{,}000\times10^{-1} = 1{,}000\times10^{-3}\;\text{m}^3\\
    K\,q_3\,q_2 &= 9{,}00\times10^9\times(-10{,}3\times10^{-9})\times(-31{,}5\times10^{-9})
                 = +2{,}920\times10^{-6}\;\text{N\,m}^2
\end{align*}

\begin{align*}
    \vec{F}_{32}
    &= \frac{2{,}920\times10^{-6}}{1{,}000\times10^{-3}}
       \,(0{,}08\,\hat{\imath}+0{,}06\,\hat{\jmath})\\
    &= 2{,}920\times10^{-3}\,(0{,}08\,\hat{\imath}+0{,}06\,\hat{\jmath})
\end{align*}

\begin{equation*}
    \boxed{\vec{F}_{32} = (2{,}336\,\hat{\imath}+1{,}752\,\hat{\jmath})\times10^{-4}\;\text{[N]}}
\end{equation*}

El factor positivo ($q_3 q_2 > 0$): fuerza repulsiva, misma direcci\'on que $\vec{r}_{32}$.

\subsection*{Fuerza neta sobre $q_2$}

\begin{align*}
    F_x &= -3{,}016\times10^{-4}+2{,}336\times10^{-4} = -0{,}680\times10^{-4}\;\text{N}\\
    F_y &= \phantom{-}2{,}151\times10^{-4}+1{,}752\times10^{-4} = +3{,}903\times10^{-4}\;\text{N}
\end{align*}

\begin{equation*}
    \boxed{\vec{F}_{\rm neta} = (-0{,}680\,\hat{\imath}+3{,}903\,\hat{\jmath})\times10^{-4}\;\text{[N]}}
\end{equation*}

\begin{equation*}
    |\vec{F}_{\rm neta}|
    = \sqrt{(0{,}680)^2+(3{,}903)^2}\times10^{-4}
    = \sqrt{0{,}4624+15{,}23}\times10^{-4}
    = 3{,}961\times10^{-4}\;\text{N}
\end{equation*}

\begin{equation*}
    \theta = 180^\circ - \arctan\!\Bigl(\tfrac{3{,}903}{0{,}680}\Bigr) = 180^\circ - 80{,}1^\circ = 99{,}9^\circ
    \quad\text{(segundo cuadrante: }F_x<0,\;F_y>0\text{)}
\end{equation*}

%-------- Parte (b) --------
\section*{Parte (b): Campo el\'ectrico en $P=(0{,}01;\,0)$ m}

Campo de cada carga: $\vec{E}_i = K\,\dfrac{q_i}{\|\vec{r}_{Pi}\|^3}\,\vec{r}_{Pi}$,\quad
$\vec{r}_{Pi} = \vec{r}_P-\vec{r}_i$.

\subsection*{Campo de $q_1$ en $P$}

\begin{align*}
    \vec{r}_{P1} &= \vec{r}_P-\vec{r}_1
                   = (0{,}01-(-0{,}02736);\;0-0{,}07518)
                   = (0{,}03736\,\hat{\imath}-0{,}07518\,\hat{\jmath})\;\text{m}\\
    \|\vec{r}_{P1}\|^2 &= (0{,}03736)^2+(0{,}07518)^2
                         = 1{,}396\times10^{-3}+5{,}652\times10^{-3} = 7{,}048\times10^{-3}\;\text{m}^2\\
    \|\vec{r}_{P1}\|^3 &= (7{,}048\times10^{-3})^{3/2}
                         = 7{,}048\times10^{-3}\times8{,}395\times10^{-2} = 5{,}917\times10^{-4}\;\text{m}^3
\end{align*}
\begin{align*}
    \vec{E}_1
    &= \frac{9{,}00\times10^9\times11{,}8\times10^{-9}}{5{,}917\times10^{-4}}
       \,(0{,}03736\,\hat{\imath}-0{,}07518\,\hat{\jmath})\\
    &= 1{,}795\times10^5\,(0{,}03736\,\hat{\imath}-0{,}07518\,\hat{\jmath})\\
    &= (6{,}706\times10^3\,\hat{\imath}-1{,}349\times10^4\,\hat{\jmath})\;\text{[N/C]}
\end{align*}

\subsection*{Campo de $q_2$ en $P$}

\begin{align*}
    \vec{r}_{P2} &= (-0{,}04\,\hat{\imath}-0{,}02\,\hat{\jmath})\;\text{m}\\
    \|\vec{r}_{P2}\|^3 &= (0{,}0016+0{,}0004)^{3/2} = (0{,}002)^{3/2}
                         = 0{,}002\times0{,}04472 = 8{,}944\times10^{-5}\;\text{m}^3
\end{align*}
\begin{align*}
    \vec{E}_2
    &= \frac{9{,}00\times10^9\times(-31{,}5\times10^{-9})}{8{,}944\times10^{-5}}
       \,(-0{,}04\,\hat{\imath}-0{,}02\,\hat{\jmath})\\
    &= (-3{,}169\times10^6)\,(-0{,}04\,\hat{\imath}-0{,}02\,\hat{\jmath})\\
    &= (1{,}268\times10^5\,\hat{\imath}+6{,}338\times10^4\,\hat{\jmath})\;\text{[N/C]}
\end{align*}

\subsection*{Campo de $q_3$ en $P$}

\begin{align*}
    \vec{r}_{P3} &= (0{,}04\,\hat{\imath}+0{,}04\,\hat{\jmath})\;\text{m}\\
    \|\vec{r}_{P3}\|^3 &= (0{,}0016+0{,}0016)^{3/2} = (0{,}0032)^{3/2}
                         = 0{,}0032\times0{,}05657 = 1{,}810\times10^{-4}\;\text{m}^3
\end{align*}
\begin{align*}
    \vec{E}_3
    &= \frac{9{,}00\times10^9\times(-10{,}3\times10^{-9})}{1{,}810\times10^{-4}}
       \,(0{,}04\,\hat{\imath}+0{,}04\,\hat{\jmath})\\
    &= (-5{,}121\times10^5)\,(0{,}04\,\hat{\imath}+0{,}04\,\hat{\jmath})\\
    &= (-2{,}048\times10^4\,\hat{\imath}-2{,}048\times10^4\,\hat{\jmath})\;\text{[N/C]}
\end{align*}

\subsection*{Campo neto en $P$}

\begin{align*}
    E_x &= 6{,}706\times10^3+1{,}268\times10^5-2{,}048\times10^4
         = (0{,}6706+12{,}68-2{,}048)\times10^4 = 1{,}130\times10^5\;\text{N/C}\\
    E_y &= -1{,}349\times10^4+6{,}338\times10^4-2{,}048\times10^4
         = (-1{,}349+6{,}338-2{,}048)\times10^4 = 2{,}941\times10^4\;\text{N/C}
\end{align*}

\begin{equation*}
    \boxed{\vec{E}_P = (1{,}130\times10^5\,\hat{\imath}+2{,}941\times10^4\,\hat{\jmath})\;\;[\text{N/C}]}
\end{equation*}

\begin{equation*}
    |\vec{E}_P|
    = \sqrt{(1{,}130)^2+(0{,}2941)^2}\times10^5
    = \sqrt{1{,}277+0{,}08650}\times10^5
    = 1{,}168\times10^5\;\text{N/C}
\end{equation*}

\begin{equation*}
    \theta = \arctan\!\Bigl(\tfrac{2{,}941}{11{,}30}\Bigr) = 14{,}6^\circ
    \quad\text{(primer cuadrante: }E_x>0,\;E_y>0\text{)}
\end{equation*}

\subsection*{Interpretaci\'on F\'isica}

\begin{itemize}
    \item El campo en $P$ est\'a dominado por $\vec{E}_2$: $q_2$ es la carga m\'as grande en
          valor absoluto y la m\'as cercana a $P$ ($d = \sqrt{(0{,}04)^2+(0{,}02)^2} = 4{,}47$ cm).
          Al ser negativa, su campo apunta \textbf{hacia} $q_2$ (en direcci\'on $+\hat{\imath}$,
          $+\hat{\jmath}$), consistente con el resultado.
    \item La fuerza neta sobre $q_2$ est\'a dominada por $\vec{F}_{12}$ (atracci\'on de $q_1$,
          ubicada en el segundo cuadrante a la izquierda y arriba de $q_2$). $F_y \gg |F_x|$
          refleja que el desplazamiento vertical entre $q_1$ y $q_2$ (5{,}52 cm) supera al
          horizontal (7{,}74 cm proyectado resulta casi cancelado por $\vec{F}_{32}$).
\end{itemize}

    \item $Q = 10\;\text{nC} = 1{,}00\times10^{-8}\;\text{C}$;\quad $R = 0{,}250\;\text{m}$

    \item $q_3 < 0$, $q_0 > 0$ $\;\to\;$ $\vec{F}_{30}$ \textbf{atractiva} (hacia $q_3$,

\end{itemize}

\subsection*{C\'alculo de $\vec{F}_{10}$}

\begin{align*}
    \vec{r}_{10} &= \vec{r}_0 - \vec{r}_1 = (0{,}250\,\hat{\imath}+0{,}250\,\hat{\jmath})\;\text{m}\\

    \quad\Rightarrow\quad
    \|\vec{r}_{10}\|^3 = (0{,}250\sqrt{2})^3 = 0{,}04419\;\text{m}^3\\
    \vec{F}_{10} &= \frac{9{,}00\times10^9\times2{,}00\times10^{-8}\times1{,}00\times10^{-8}}{0{,}04419}
                   \,(0{,}250\,\hat{\imath}+0{,}250\,\hat{\jmath})

\end{align*}

\end{equation*}

    \vec{F}_{20} &= \frac{9{,}00\times10^9\times2{,}00\times10^{-8}\times1{,}00\times10^{-8}}{0{,}1250}
                   \,(0{,}500\,\hat{\imath})
    = 1{,}440\times10^{-5}\times0{,}500\,\hat{\imath}
\end{align*}

\begin{equation*}
    \boxed{\vec{F}_{20} = 7{,}200\times10^{-6}\,\hat{\imath}\;\;[\text{N}]}
\end{equation*}

\subsection*{C\'alculo de $\vec{F}_{30}$}

\begin{align*}
    \vec{r}_{30} &= \vec{r}_0 - \vec{r}_3

    = (0{,}4268\,\hat{\imath}-0{,}1768\,\hat{\jmath})\;\text{m}\\

    = \sqrt{0{,}18216+0{,}03126} = 0{,}4622\;\text{m}
    \quad\Rightarrow\quad
    \|\vec{r}_{30}\|^3 = 0{,}09878\;\text{m}^3\\

    = -9{,}109\times10^{-6}\,(0{,}4268\,\hat{\imath}-0{,}1768\,\hat{\jmath})
\end{align*}

\begin{equation*}
    \boxed{\vec{F}_{30} = (-3{,}888\times10^{-6}\,\hat{\imath}+1{,}610\times10^{-6}\,\hat{\jmath})\;\;[\text{N}]}
\end{equation*}

\end{equation*}

\subsection*{Interpretaci\'on F\'isica}

          e incrementa la $+\hat{\jmath}$, dando un \'angulo resultante de $41{,}2^\circ$.
    \item El par $(q_1, q_2)$ aporta la mayor magnitud por ser cargas dobles $2Q$.
\end{itemize}

\section*{\textcolor{ubbblue}{Problema 3 \;--\; Barra Finita y Carga Puntual}}

entre $x=5$ [cm] e $x=18$ [cm]. Adem\'as, en el punto $(-2;\,6)$ [cm] se ubica una carga puntual

\subsection*{Datos Num\'ericos}

    \item $\lambda = -3{,}6\times10^{-9}\;\text{C/m}$\quad$\Rightarrow$\quad

\end{itemize}

\subsection*{An\'alisis Cualitativo}

La barra tiene carga \textbf{negativa} ($\lambda < 0$). El punto $P$ est\'a a la izquierda
($x_P=-2$ cm) y arriba ($y_P=6$ cm) respecto a la barra (eje $x$, $x\in[5,18]$ cm).
El campo de una distribuci\'on negativa apunta \textbf{hacia} ella:
\begin{itemize}
    \item Componente $+\hat{\imath}$: la barra est\'a a la derecha de $P$.
    \item Componente $-\hat{\jmath}$: la barra est\'a debajo de $P$.
\end{itemize}
La fuerza sobre $q < 0$ ser\'a opuesta al campo: $-\hat{\imath}$, $+\hat{\jmath}$
(repulsi\'on entre $q$ y $\lambda$, ambas negativas).

\subsection*{Planteamiento de la Integral}

Elemento $dq = \lambda\,dx$ en $\vec{r}_q = x\,\hat{\imath}$. Vector del elemento al campo:
\begin{equation*}
    \vec{r}_P - \vec{r}_q = (-0{,}02-x)\,\hat{\imath} + 0{,}06\,\hat{\jmath}
\end{equation*}

\textbf{Sustituci\'on:} $u = x + 0{,}02$ (distancia en $x$ de $P$ al elemento):
\begin{equation*}
    x = 0{,}05\;\text{m}\;\to\; u = 0{,}07\;\text{m}
    \qquad
    x = 0{,}18\;\text{m}\;\to\; u = 0{,}20\;\text{m}
\end{equation*}

Con $a = 0{,}06$ m:
\begin{equation*}
    \vec{E}_{\rm barra} = K\lambda\int_{0{,}07}^{0{,}20}
    \frac{-u\,\hat{\imath}+a\,\hat{\jmath}}{(u^2+a^2)^{3/2}}\,du
\end{equation*}

\subsection*{Antiderivadas Est\'andar}

\begin{equation*}
    \int\frac{u\,du}{(u^2+a^2)^{3/2}} = \frac{-1}{\sqrt{u^2+a^2}}+C
    \qquad\qquad
    \int\frac{du}{(u^2+a^2)^{3/2}} = \frac{u}{a^2\sqrt{u^2+a^2}}+C
\end{equation*}

\subsection*{Componente $\hat{\imath}$}

\begin{equation*}
    E_x = K\lambda\times(-1)\left[\frac{-1}{\sqrt{u^2+a^2}}\right]_{0{,}07}^{0{,}20}
    = K\lambda\times(-1)\left(-\frac{1}{\sqrt{0{,}0436}}+\frac{1}{\sqrt{0{,}0085}}\right)
\end{equation*}

\begin{equation*}
    \frac{1}{\sqrt{0{,}0436}} = \frac{1}{0{,}2088} = 4{,}789\;\text{m}^{-1}
    \qquad
    \frac{1}{\sqrt{0{,}0085}} = \frac{1}{0{,}09220} = 10{,}847\;\text{m}^{-1}
\end{equation*}

\begin{equation*}
    E_x = (-32{,}4)\times(-1)\times(-4{,}789+10{,}847)
    = (-32{,}4)\times(-6{,}058) = \mathbf{196{,}3}\;\text{N/C}
\end{equation*}

\subsection*{Componente $\hat{\jmath}$}

\begin{equation*}
    E_y = K\lambda\times a\times\frac{1}{a^2}\left[\frac{u}{\sqrt{u^2+a^2}}\right]_{0{,}07}^{0{,}20}
    = K\lambda\times\frac{1}{a}\left[\frac{u}{\sqrt{u^2+a^2}}\right]_{0{,}07}^{0{,}20}
\end{equation*}

\begin{equation*}
    \frac{u}{\sqrt{u^2+a^2}}\Bigg|_{u=0{,}20} = \frac{0{,}20}{\sqrt{0{,}0436}} = \frac{0{,}20}{0{,}2088} = 0{,}9578
    \qquad
    \frac{u}{\sqrt{u^2+a^2}}\Bigg|_{u=0{,}07} = \frac{0{,}07}{\sqrt{0{,}0085}} = \frac{0{,}07}{0{,}09220} = 0{,}7593
\end{equation*}

\begin{equation*}
    E_y = (-32{,}4)\times\frac{1}{0{,}06}\times(0{,}9578-0{,}7593)
    = (-32{,}4)\times16{,}667\times0{,}1985 = -107{,}2\;\text{N/C}
\end{equation*}

\subsection*{Campo de la Barra en $P$}

\begin{equation*}
    \boxed{\vec{E}_{\rm barra} = (196{,}3\,\hat{\imath}-107{,}2\,\hat{\jmath})\;\;[\text{N/C}]}
\end{equation*}

Consistente con el an\'alisis cualitativo: $E_x > 0$ (hacia la derecha, donde est\'a la barra)
y $E_y < 0$ (hacia abajo, donde est\'a la barra).

\subsection*{Fuerza El\'ectrica sobre $q$}

\begin{equation*}
    \vec{F} = q\,\vec{E}_{\rm barra}
    = (-16\times10^{-9})\,(196{,}3\,\hat{\imath}-107{,}2\,\hat{\jmath})
\end{equation*}

\begin{equation*}
    \boxed{\vec{F} = (-3{,}141\,\hat{\imath}+1{,}715\,\hat{\jmath})\;\;\mu\text{N}}
\end{equation*}

\begin{equation*}
    |\vec{F}| = \sqrt{(3{,}141)^2+(1{,}715)^2}\;\mu\text{N}
    = \sqrt{9{,}866+2{,}941}\;\mu\text{N} = 3{,}579\;\mu\text{N}
    \qquad
    \theta = 180^\circ - \arctan\!\Bigl(\tfrac{1{,}715}{3{,}141}\Bigr) = 151{,}4^\circ
\end{equation*}

\subsection*{Interpretaci\'on F\'isica}

\begin{itemize}
    \item El campo $\vec{E}_{\rm barra}$ apunta \textbf{hacia} la barra ($+\hat{\imath}$, $-\hat{\jmath}$),
          comportamiento t\'ipico de una distribuci\'on negativa que atrae las l\'ineas de campo.
    \item La fuerza sobre $q < 0$ es contraria a $\vec{E}$: apunta \textbf{alejĂˇndose} de la barra
          ($-\hat{\imath}$, $+\hat{\jmath}$). Coherente con la repulsi\'on entre $q$ y $\lambda$,
          ambas negativas.
    \item La componente $E_x$ supera a $|E_y|$ porque la barra se extiende a lo largo de $x$,
          acumulando contribuciones en esa direcci\'on.
\end{itemize}

    \item $z = 0{,}210\;\text{m}$;\quad $z^2 = 0{,}0441\;\text{m}^2$

    \item $(R^2+z^2)^{3/2} = 0{,}1341\times\sqrt{0{,}1341}
          = 0{,}1341\times0{,}3662 = 4{,}911\times10^{-2}\;\text{m}^3$
    \item $q = 22{,}0\times10^{-6}\;\text{C}$
\end{itemize}

\subsection*{Parametrizaci\'on del Semianillo}

Se toma la \textbf{mitad superior} del c\'irculo de radio $R$ en el plano $xy$
(cuadrantes I y II, $y\geq0$):
\begin{equation*}
    \vec{r}_q = R\cos\theta\,\hat{\imath}+R\,\text{sen}\,\theta\,\hat{\jmath}
    \qquad \theta\in[0,\,\pi]
\end{equation*}

El punto de campo: $\vec{r}_P = z\,\hat{k}$. Vector del elemento al campo:
\begin{equation*}
    \vec{r}_P-\vec{r}_q = -R\cos\theta\,\hat{\imath}-R\,\text{sen}\,\theta\,\hat{\jmath}+z\,\hat{k}
    \quad (\text{sen}\,\theta \geq 0\;\text{para}\;\theta\in[0,\pi])
    \qquad
    \|\vec{r}_P-\vec{r}_q\|^3 = (R^2+z^2)^{3/2} = \text{cte.}
\end{equation*}

Campo diferencial ($dq = \lambda R\,d\theta$):
\begin{equation*}
    d\vec{E}
    = \frac{K\lambda R\,d\theta}{(R^2+z^2)^{3/2}}
    \Bigl[-R\cos\theta\,\hat{\imath}-R\,\text{sen}\,\theta\,\hat{\jmath}+z\,\hat{k}\Bigr]
\end{equation*}

\subsection*{An\'alisis de Simetr\'ia -- Componente $\hat{\imath}$}

\begin{equation*}
    E_x = \frac{K\lambda R(-R)}{(R^2+z^2)^{3/2}}\int_0^{\pi}\cos\theta\,d\theta
    = \frac{K\lambda R(-R)}{(R^2+z^2)^{3/2}}\,[\text{sen}\,\theta]_0^{\pi}
    = \frac{K\lambda R(-R)}{(R^2+z^2)^{3/2}}\,(0-0) = \mathbf{0}
\end{equation*}

La mitad izquierda del semianillo ($\theta\in[\pi/2,\pi]$) y la mitad derecha
($\theta\in[0,\pi/2]$) generan contribuciones en $\hat{\imath}$ iguales y opuestas.

\subsection*{Integraci\'on -- Componente $\hat{\jmath}$}

\begin{equation*}
    \int_0^{\pi}\text{sen}\,\theta\,d\theta
    = [-\cos\theta]_0^{\pi} = (-\cos\pi+\cos 0) = 1+1 = 2
\end{equation*}

\begin{equation*}
    E_y = \frac{K\lambda R(-R)}{(R^2+z^2)^{3/2}}\,(+2)
    = \frac{-2K\lambda R^2}{(R^2+z^2)^{3/2}}
\end{equation*}

\textbf{Nota clave:} en el anillo completo, $\int_0^{2\pi}\text{sen}\,\theta\,d\theta = 0$ y esta
componente desaparece. El semianillo \textbf{rompe esa simetr\'ia} y genera $E_y\neq0$.
El signo negativo indica que el campo apunta en $-\hat{\jmath}$: el arco est\'a en $y>0$
y el campo se aleja del arco (hacia $-\hat{\jmath}$).

\begin{equation*}
    E_y = \frac{-2\times1{,}620\times10^5\times(0{,}300)^2}{4{,}911\times10^{-2}}
    = \frac{-2\times1{,}620\times10^5\times0{,}0900}{4{,}911\times10^{-2}}
    = \frac{-2{,}916\times10^4}{4{,}911\times10^{-2}}
    = -5{,}938\times10^5\;\text{N/C}
\end{equation*}

\subsection*{Integraci\'on -- Componente $\hat{k}$}

\begin{equation*}
    \int_0^{\pi}d\theta = \pi
    \qquad\Rightarrow\qquad
    E_z = \frac{K\lambda R\,z}{(R^2+z^2)^{3/2}}\,\pi
    = \frac{K\,(\lambda\pi R)\,z}{(R^2+z^2)^{3/2}}
\end{equation*}

La cantidad $\lambda\pi R = Q_{\rm mitad}$ es la carga total del semianillo. La expresi\'on es
id\'entica a la del anillo completo con $Q\to Q_{\rm mitad}$.

\begin{equation*}
    E_z = \frac{1{,}620\times10^5\times\pi\times0{,}300\times0{,}210}{4{,}911\times10^{-2}}
    = \frac{1{,}620\times10^5\times0{,}19792}{4{,}911\times10^{-2}}
    = \frac{3{,}206\times10^4}{4{,}911\times10^{-2}}
    = 6{,}529\times10^5\;\text{N/C}
\end{equation*}

\subsection*{Campo El\'ectrico en $P$}

\begin{equation*}
    \boxed{\vec{E}_P = -5{,}938\times10^5\,\hat{\jmath}+6{,}529\times10^5\,\hat{k}\;\;[\text{N/C}]}
\end{equation*}

\begin{align*}
    |\vec{E}_P| = \sqrt{(5{,}938)^2+(6{,}529)^2}\times10^5
    = \sqrt{35{,}26+42{,}63}\times10^5
    = \sqrt{77{,}89}\times10^5
    = 8{,}826\times10^5\;\text{N/C}
    \quad
\end{align*}

\begin{align*}
\alpha = \arctan\!\Bigl(\tfrac{|E_y|}{E_z}\Bigr) 
= \arctan\!\Bigl(\tfrac{5{,}938}{6{,}529}\Bigr) 
= 42{,}3^\circ\;\text{desde }\hat{k}\text{ hacia }{-\hat{\jmath}}
\end{align*}

\subsection*{Fuerza El\'ectrica sobre $q$}

\begin{equation*}
    \vec{F} = q\,\vec{E}_P
    = 22{,}0\times10^{-6}\times(-5{,}938\times10^5\,\hat{\jmath}+6{,}529\times10^5\,\hat{k})
\end{equation*}

\begin{equation*}
    \boxed{\vec{F} = -13{,}06\,\hat{\jmath}+14{,}37\,\hat{k}\;\;[\text{N}]}
\end{equation*}

\begin{equation*}
    |\vec{F}| = \sqrt{(13{,}06)^2+(14{,}37)^2} = \sqrt{170{,}6+206{,}5} = 19{,}42\;\text{N}
\end{equation*}

\subsection*{Interpretaci\'on F\'isica}

\begin{itemize}
    \item \textbf{Componente $\hat{k}$}: presente en todo arco cargado. Apunta alej\'andose del
          semianillo positivo (en $+\hat{k}$). Su expresi\'on es equivalente a la del anillo
          completo con $Q_{\rm mitad} = \lambda\pi R = 1{,}696\times10^{-5}$ C.
    \item \textbf{Componente $-\hat{\jmath}$}: exclusiva del semianillo (en el anillo completo se
          cancela por simetr\'ia). Como el arco est\'a en $y\geq0$ (cuadrantes I y II), el campo
          apunta \textbf{hacia} $-\hat{\jmath}$, alej\'andose del arco positivo en la
          direcci\'on transversal.
    \item Las dos componentes son de magnitud comparable ($|E_y| \approx E_z$), resultando en un
          \'angulo de $42{,}3^\circ$ desde $\hat{k}$ hacia $-\hat{\jmath}$.
    \item La fuerza sobre $q > 0$ apunta en la misma direcci\'on que $\vec{E}_P$:
          $-\hat{\jmath}$ (alejĂˇndose del arco lateralmente) y $+\hat{k}$ (alejĂˇndose axialmente).
\end{itemize}
