\section*{Enunciado del Problema}

Dados los vectores $\vec{A}$ y $\vec{B}$, determine:
\begin{enumerate}
    \item El producto escalar $\vec{A} \cdot \vec{B}$
    \item El producto vectorial $\vec{A} \times \vec{B}$
    \item El \'angulo $\theta$ que forman entre s\'i, usando el producto escalar
\end{enumerate}

Donde:
\begin{eqnarray*}
\vec{A} &=& -4\hat{\imath}+7\hat{\jmath}-12\hat{k} \\
\vec{B} &=& 7\hat{\imath}-2\hat{\jmath}+8\hat{k}
\end{eqnarray*}

%========================================
% PARTE 1: PRODUCTO ESCALAR
%========================================
\section*{Parte 1: Producto Escalar $\vec{A} \cdot \vec{B}$}

\subsection*{Definici\'on}

El producto escalar (o producto punto) de dos vectores se define como:
\begin{equation*}
\vec{A} \cdot \vec{B} = A_x B_x + A_y B_y + A_z B_z
\end{equation*}

\subsection*{C\'alculo}

Identificando las componentes:
\begin{itemize}
    \item $\vec{A} = (-4, 7, -12)$ \quad $\Rightarrow$ \quad $A_x = -4$, $A_y = 7$, $A_z = -12$
    \item $\vec{B} = (7, -2, 8)$ \quad $\Rightarrow$ \quad $B_x = 7$, $B_y = -2$, $B_z = 8$
\end{itemize}

Sustituyendo:
\begin{equation*}
\vec{A} \cdot \vec{B} = (-4)(7) + (7)(-2) + (-12)(8)
\end{equation*}

\begin{equation*}
\vec{A} \cdot \vec{B} = -28 - 14 - 96
\end{equation*}

\begin{equation*}
\boxed{\vec{A} \cdot \vec{B} = -138}
\end{equation*}

\textbf{Interpretaci\'on:} El resultado negativo indica que el \'angulo entre los vectores es \textbf{obtuso} (mayor que $90^\circ$).

%========================================
% PARTE 2: PRODUCTO VECTORIAL
%========================================
\section*{Parte 2: Producto Vectorial $\vec{A} \times \vec{B}$}

\subsection*{Definici\'on}

El producto vectorial se calcula mediante el determinante:
\begin{equation*}
\vec{A} \times \vec{B} = \begin{vmatrix}
\hat{\imath} & \hat{\jmath} & \hat{k} \\
A_x & A_y & A_z \\
B_x & B_y & B_z
\end{vmatrix}
\end{equation*}

\subsection*{C\'alculo del Determinante}

\begin{equation*}
\vec{A} \times \vec{B} = \begin{vmatrix}
\hat{\imath} & \hat{\jmath} & \hat{k} \\
-4 & 7 & -12 \\
7 & -2 & 8
\end{vmatrix}
\end{equation*}

Expandiendo por cofactores de la primera fila:

\textbf{Componente $\hat{\imath}$:}
\begin{equation*}
(\vec{A} \times \vec{B})_x = \begin{vmatrix} 7 & -12 \\ -2 & 8 \end{vmatrix} = (7)(8) - (-12)(-2) = 56 - 24 = \mathbf{32}
\end{equation*}

\textbf{Componente $\hat{\jmath}$:}
\begin{equation*}
(\vec{A} \times \vec{B})_y = -\begin{vmatrix} -4 & -12 \\ 7 & 8 \end{vmatrix} = -[(-4)(8) - (-12)(7)] = -[-32 + 84] = -52 = \mathbf{-52}
\end{equation*}

\textbf{Componente $\hat{k}$:}
\begin{equation*}
(\vec{A} \times \vec{B})_z = \begin{vmatrix} -4 & 7 \\ 7 & -2 \end{vmatrix} = (-4)(-2) - (7)(7) = 8 - 49 = \mathbf{-41}
\end{equation*}

Por lo tanto:
\begin{equation*}
\boxed{\vec{A} \times \vec{B} = 32\hat{\imath} - 52\hat{\jmath} - 41\hat{k}}
\end{equation*}

\subsection*{Verificaci\'on: Magnitud del Producto Vectorial}

\begin{equation*}
\left\Vert \vec{A} \times \vec{B} \right\Vert = \sqrt{32^2 + (-52)^2 + (-41)^2} = \sqrt{1024 + 2704 + 1681} = \sqrt{5409} \approx 73.55
\end{equation*}

\textbf{Interpretaci\'on:} El vector $\vec{A} \times \vec{B}$ es \textbf{perpendicular} tanto a $\vec{A}$ como a $\vec{B}$, y su magnitud representa el \'area del paralelogramo formado por ambos vectores.

%========================================
% PARTE 3: ANGULO ENTRE VECTORES
%========================================
\section*{Parte 3: \'Angulo $\theta$ entre los Vectores}

\subsection*{F\'ormula}

El \'angulo entre dos vectores se obtiene de la relaci\'on:
\begin{equation*}
\vec{A} \cdot \vec{B} = \left\Vert \vec{A} \right\Vert \left\Vert \vec{B} \right\Vert \cos\theta
\end{equation*}

Despejando $\theta$:
\begin{equation*}
\theta = \arccos\left( \frac{\vec{A} \cdot \vec{B}}{\left\Vert \vec{A} \right\Vert \left\Vert \vec{B} \right\Vert} \right)
\end{equation*}

\subsection*{Paso 1: Calcular $\left\Vert \vec{A} \right\Vert$}

\begin{equation*}
\left\Vert \vec{A} \right\Vert = \sqrt{(-4)^2 + 7^2 + (-12)^2} = \sqrt{16 + 49 + 144} = \sqrt{209} \approx \boxed{14.457}
\end{equation*}

\subsection*{Paso 2: Calcular $\left\Vert \vec{B} \right\Vert$}

\begin{equation*}
\left\Vert \vec{B} \right\Vert = \sqrt{7^2 + (-2)^2 + 8^2} = \sqrt{49 + 4 + 64} = \sqrt{117} \approx \boxed{10.817}
\end{equation*}

\subsection*{Paso 3: Calcular $\cos\theta$}

\begin{equation*}
\cos\theta = \frac{\vec{A} \cdot \vec{B}}{\left\Vert \vec{A} \right\Vert \left\Vert \vec{B} \right\Vert} = \frac{-138}{(14.457)(10.817)} = \frac{-138}{156.39}
\end{equation*}

\begin{equation*}
\cos\theta \approx -0.8824
\end{equation*}

\subsection*{Paso 4: Calcular $\theta$}

\begin{equation*}
\theta = \arccos(-0.8824)
\end{equation*}

\begin{equation*}
\boxed{\theta \approx 151.94^\circ \approx 152^\circ}
\end{equation*}

En radianes: $\theta \approx 2.652$ rad

\subsection*{Verificaci\'on usando el Producto Vectorial}

Tambi\'en se puede verificar usando:
\begin{equation*}
\left\Vert \vec{A} \times \vec{B} \right\Vert = \left\Vert \vec{A} \right\Vert \left\Vert \vec{B} \right\Vert \sin\theta
\end{equation*}

\begin{equation*}
\sin(151.94^\circ) \approx 0.4706
\end{equation*}

\begin{equation*}
\left\Vert \vec{A} \right\Vert \left\Vert \vec{B} \right\Vert \sin\theta = (14.457)(10.817)(0.4706) \approx 73.57 \approx \left\Vert \vec{A} \times \vec{B} \right\Vert \quad \checkmark
\end{equation*}

\section*{Interpretaci\'on F\'isica}

En electromagnetismo, estos productos vectoriales tienen aplicaciones fundamentales:

\begin{itemize}
    \item \textbf{Producto escalar ($\vec{A} \cdot \vec{B}$):}
    \begin{itemize}
        \item Trabajo realizado: $W = \vec{F} \cdot \vec{d}$
        \item Flujo el\'ectrico: $\Phi_E = \vec{E} \cdot \vec{A}$
        \item Potencia el\'ectrica: $P = \vec{F} \cdot \vec{v}$
    \end{itemize}

    \item \textbf{Producto vectorial ($\vec{A} \times \vec{B}$):}
    \begin{itemize}
        \item Fuerza magn\'etica: $\vec{F} = q\vec{v} \times \vec{B}$
        \item Torque: $\vec{\tau} = \vec{r} \times \vec{F}$
        \item Campo magn\'etico (Biot-Savart): $d\vec{B} = \frac{\mu_0 I}{4\pi} \frac{d\vec{l} \times \hat{r}}{r^2}$
    \end{itemize}

    \item \textbf{\'Angulo entre vectores:}
    \begin{itemize}
        \item Determina la componente de un vector en la direcci\'on de otro
        \item Un \'angulo obtuso ($\theta > 90^\circ$) indica que las proyecciones son opuestas
    \end{itemize}
\end{itemize}
