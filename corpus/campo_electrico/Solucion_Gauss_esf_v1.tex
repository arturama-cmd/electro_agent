%========================================
% ENUNCIADO
%========================================
\section*{Enunciado del Problema}

Considere una esfera maciza aislante de radio $a$, con carga $q_1 = -3Q$ distribuida de forma homog\'enea. Rodeando esa esfera hay un cascar\'on esf\'erico conductor de radio interior $b$ y radio exterior $c$, con carga total $q_2 = 4Q$ distribuida de forma homog\'enea. Para este sistema, obtener utilizando la Ley de Gauss:

\begin{enumerate}[(a)]
    \item \textbf{(16~pts)} La magnitud del campo el\'ectrico en las regiones:
          (i)~$r < a$,\quad (ii)~$a < r < b$,\quad (iii)~$b < r < c$,\quad (iv)~$r > c$.
    \item \textbf{(3~pts)} Si $a = 10$~[cm], $b = 12$~[cm], $c = 15$~[cm] y $Q = 120$~[$\mu$C],
          calcular la magnitud del campo el\'ectrico a $r = 11$~[cm].
\end{enumerate}

\section*{Datos del Problema}

\[
    q_1 = -3Q \quad (\text{esfera maciza aislante, radio }a),
    \qquad
    q_2 = 4Q \quad (\text{cascar\'on conductor, radios }b\text{ y }c)
\]
\[
    \varepsilon_0 = 8{,}85 \times 10^{-12}
    \ \left[\frac{\text{C}^2}{\text{N}\cdot\text{m}^2}\right]
\]

%========================================
% PARTE a)
%========================================
\section*{Parte a): Magnitud del Campo El\'ectrico}

Por simetr\'ia esf\'erica, $\vec{E} = E(r)\,\hat{r}$ es constante en m\'odulo sobre cualquier
superficie esf\'erica de radio $r$. La Ley de Gauss establece:

\[
    \oint \vec{E}\cdot d\vec{A} = \frac{q_\text{enc}}{\varepsilon_0}
\]

El lado izquierdo se desarrolla en coordenadas esf\'ericas
($dA = r^2\operatorname{sen}\phi\,d\phi\,d\theta$):

\begin{align*}
    \oint \vec{E}\cdot d\vec{A}
        &= \int_0^{2\pi}\!\int_0^{\pi}
           E\,r^2\,\operatorname{sen}\phi\;d\phi\;d\theta \\[4pt]
        &= E\,r^2
           \int_0^{2\pi}\!d\theta
           \cdot
           \int_0^{\pi}\operatorname{sen}\phi\;d\phi \\[4pt]
        &= E\,r^2
           \cdot \bigl(\theta\bigr)\big|_0^{2\pi}
           \cdot \bigl(-\cos\phi\bigr)\big|_0^{\pi} \\[4pt]
        &= E\,r^2 \cdot 2\pi \cdot 2
         = 4\pi r^2\,E
\end{align*}

De modo que en \textbf{todas las regiones}:
\[
    E \cdot 4\pi r^2 = \frac{q_\text{enc}}{\varepsilon_0}
\]
Solo cambia $q_\text{enc}$ de una regi\'on a otra.

    \draw[gaussred, dashed, thick] (0,0) circle (0.7);

        node[right, font=\scriptsize]{$r$};
    % Radio a
    \draw[->] (0,0) -- ({1.2*cos(130)},{1.2*sin(130)})
        node[above left, font=\scriptsize]{$a$};
    \node[font=\scriptsize] at (0,0.25) {$-3Q$};
\end{tikzpicture}
\end{minipage}%
\hfill
\begin{minipage}[c]{0.67\textwidth}
La superficie gaussiana (radio $r < a$) encierra solo la fracci\'on
de carga proporcional al volumen encerrado:
\[
    \frac{q_\text{enc}}{-3Q}
    = \frac{\tfrac{4}{3}\pi r^3}{\tfrac{4}{3}\pi a^3}
    \qquad\Rightarrow\qquad
    q_\text{enc} = \frac{-3Q\,r^3}{a^3}
\]
Sustituyendo en la Ley de Gauss:
\begin{align*}
    E \cdot 4\pi r^2 &= \frac{-3Q\,r^3}{a^3\,\varepsilon_0} \\[6pt]
    \vec{E}         &= \frac{-3Q\,r}{4\pi\varepsilon_0\,a^3}\,\hat{r}
\end{align*}
\end{minipage}

\[
    \boxed{\vec{E} = \frac{-3Q\,r}{4\pi\varepsilon_0\,a^3}\,\hat{r} \quad [\text{N/C}]}
\]

La superficie gaussiana est\'a \textbf{dentro del material conductor}.

\[
    \boxed{\vec{E} = \vec{0} \quad [\text{N/C}]}

conductor quedan sobre la superficie exterior ($r = c$).

%========================================
% RESUMEN
%========================================
\section*{Resumen: Campo el\'ectrico seg\'un la regi\'on}

\[
\vec{E}(r) =
\begin{cases}
    \dfrac{-3Q\,r}{4\pi\varepsilon_0\,a^3}\,\hat{r}  & r < a        \\[10pt]
    \dfrac{-3Q}{4\pi\varepsilon_0\,r^2}\,\hat{r}     & a < r < b    \\[10pt]
    \vec{0}                                            & b < r < c    \\[10pt]
    \dfrac{Q}{4\pi\varepsilon_0\,r^2}\,\hat{r}        & r > c
\end{cases}
\]

%========================================
% PARTE b)
%========================================
\section*{Parte b): C\'alculo Num\'erico}

Con $a = 0{,}10$~[m], $b = 0{,}12$~[m], $c = 0{,}15$~[m], $Q = 120$~[$\mu$C] y $r = 0{,}11$~[m].

Como $a < r < b$ (pues $0{,}10 < 0{,}11 < 0{,}12$~[m]), corresponde la expresi\'on de la regi\'on~(ii):

\begin{align*}
    \vec{E} &= \frac{-3Q}{4\pi\varepsilon_0\,r^2}\,\hat{r} \\[8pt]
            &= \frac{-3 \times 120\times10^{-6}\ [\text{C}]}
                   {4\pi \times (0{,}11)^2\ [\text{m}^2]
                    \times 8{,}85\times10^{-12}
                    \left[\dfrac{\text{C}^2}{\text{N}\cdot\text{m}^2}\right]}\,\hat{r} \\[8pt]
            &= \frac{-3{,}6\times10^{-4}}{1{,}346\times10^{-12}}\,\hat{r}
\end{align*}

\[
    \boxed{\vec{E} = -2{,}675 \times 10^{8}\,\hat{r}\ [\text{N/C}]}
\]

El coeficiente negativo de $\hat{r}$ indica que el campo apunta radialmente hacia el centro
(la carga neta encerrada en esta regi\'on es $-3Q < 0$).
