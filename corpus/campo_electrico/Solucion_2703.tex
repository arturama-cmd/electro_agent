\section*{Enunciado del Problema}

Determinar el campo el\'ectrico neto $\vec{E}_{\text{neto}}$ en el punto $P$, producido por tres cargas puntuales. Datos:

\begin{center}
\begin{tabular}{|c|c|c|}
\hline
\textbf{Carga} & \textbf{Valor} & \textbf{Posici\'on} \\
\hline
$q_1$ & $+1\;\mu\text{C}$ & $(7,\;14)$ cm \\
\hline
$q_2$ & $-2\;\mu\text{C}$ & $(-2,\;8)$ cm \\
\hline
$q_3$ & $-3\;\mu\text{C}$ & $(-13,\;-17)$ cm \\
\hline
$P$ & (punto de inter\'es) & $(15,\;-3)$ cm \\
\hline
\end{tabular}
\end{center}

\section*{Datos Num\'ericos}

\begin{itemize}
    \item $K = 8{,}99 \times 10^9\;\text{N}\cdot\text{m}^2/\text{C}^2$
    \item $q_1 = +1\;\mu\text{C} = +1\times10^{-6}\;\text{C}$;\quad $q_2 = -2\;\mu\text{C} = -2\times10^{-6}\;\text{C}$;\quad $q_3 = -3\;\mu\text{C} = -3\times10^{-6}\;\text{C}$
    \item Coordenadas en metros: $\vec{r}_1=(0{,}07;\,0{,}14)$ m;\; $\vec{r}_2=(-0{,}02;\,0{,}08)$ m;\; $\vec{r}_3=(-0{,}13;\,-0{,}17)$ m;\; $P=(0{,}15;\,-0{,}03)$ m
\end{itemize}

%========================================
% ANALISIS CUALITATIVO
%========================================
\section*{An\'alisis Cualitativo}

El campo el\'ectrico neto se obtiene por \textbf{superposici\'on}:
$\vec{E}_{\text{neto}} = \vec{E}_1 + \vec{E}_2 + \vec{E}_3$.

Usando la forma vectorial:
\begin{equation*}
    \vec{E}_i = K \frac{q_i\,\vec{r}_{iP}}{|\vec{r}_{iP}|^3}
    \qquad \text{donde } \vec{r}_{iP} = P - \vec{r}_i
\end{equation*}

\begin{itemize}
    \item $q_1 > 0$: el campo $\vec{E}_1$ en $P$ apunta \textbf{alejandose} de $q_1$ (hacia $P$).
    \item $q_2 < 0$: el campo $\vec{E}_2$ en $P$ apunta \textbf{hacia} $q_2$ (opuesto a $\vec{r}_{2P}$).
    \item $q_3 < 0$: el campo $\vec{E}_3$ en $P$ apunta \textbf{hacia} $q_3$ (opuesto a $\vec{r}_{3P}$).
\end{itemize}

%========================================
% PASO 1: VECTORES DE POSICION
%========================================
\section*{Paso 1: Vectores de Posici\'on $\vec{r}_{iP} = P - \vec{r}_i$}

\subsection*{De $q_1$ a $P$}
\begin{equation*}
\vec{r}_{1P} = (0{,}15 - 0{,}07)\,\hat{\imath} + (-0{,}03 - 0{,}14)\,\hat{\jmath}
= 0{,}08\,\hat{\imath} - 0{,}17\,\hat{\jmath} \;\text{[m]}
\end{equation*}
\begin{equation*}
|\vec{r}_{1P}| = \sqrt{(0{,}08)^2 + (0{,}17)^2}
= \sqrt{0{,}0064 + 0{,}0289}
= \sqrt{0{,}0353}
= 0{,}18788\;\text{m}
\end{equation*}
\begin{equation*}
|\vec{r}_{1P}|^3 = (0{,}18788)^3 = 0{,}006630\;\text{m}^3
\end{equation*}

\subsection*{De $q_2$ a $P$}
\begin{equation*}
\vec{r}_{2P} = (0{,}15 - (-0{,}02))\,\hat{\imath} + (-0{,}03 - 0{,}08)\,\hat{\jmath}
= 0{,}17\,\hat{\imath} - 0{,}11\,\hat{\jmath} \;\text{[m]}
\end{equation*}
\begin{equation*}
|\vec{r}_{2P}| = \sqrt{(0{,}17)^2 + (0{,}11)^2}
= \sqrt{0{,}0289 + 0{,}0121}
= \sqrt{0{,}0410}
= 0{,}20248\;\text{m}
\end{equation*}
\begin{equation*}
|\vec{r}_{2P}|^3 = (0{,}20248)^3 = 0{,}008302\;\text{m}^3
\end{equation*}

\subsection*{De $q_3$ a $P$}
\begin{equation*}
\vec{r}_{3P} = (0{,}15 - (-0{,}13))\,\hat{\imath} + (-0{,}03 - (-0{,}17))\,\hat{\jmath}
= 0{,}28\,\hat{\imath} + 0{,}14\,\hat{\jmath} \;\text{[m]}
\end{equation*}
\begin{equation*}
|\vec{r}_{3P}| = \sqrt{(0{,}28)^2 + (0{,}14)^2}
= \sqrt{0{,}0784 + 0{,}0196}
= \sqrt{0{,}0980}
= 0{,}31305\;\text{m}
\end{equation*}
\begin{equation*}
|\vec{r}_{3P}|^3 = (0{,}31305)^3 = 0{,}030679\;\text{m}^3
\end{equation*}

%========================================
% PASO 2: CAMPO DE CADA CARGA EN P
%========================================
\section*{Paso 2: Campo El\'ectrico de Cada Carga en $P$}

\subsection*{Campo de $q_1$ en $P$}

\begin{equation*}
\vec{E}_1 = K\,\frac{q_1\,\vec{r}_{1P}}{|\vec{r}_{1P}|^3}
= (8{,}99\times10^9)\,\frac{(+1\times10^{-6})(0{,}08\,\hat{\imath} - 0{,}17\,\hat{\jmath})}{0{,}006630}
\end{equation*}

\begin{equation*}
\vec{E}_1 = \frac{8990\,(0{,}08\,\hat{\imath} - 0{,}17\,\hat{\jmath})}{0{,}006630}
= \frac{719{,}2\,\hat{\imath} - 1528{,}3\,\hat{\jmath}}{0{,}006630}
\end{equation*}

\begin{equation*}
\boxed{\vec{E}_1 = 108{,}5\,\hat{\imath} - 230{,}5\,\hat{\jmath} \;\;\text{[kN/C]}}
\end{equation*}

\subsection*{Campo de $q_2$ en $P$}

\begin{equation*}
\vec{E}_2 = K\,\frac{q_2\,\vec{r}_{2P}}{|\vec{r}_{2P}|^3}
= (8{,}99\times10^9)\,\frac{(-2\times10^{-6})(0{,}17\,\hat{\imath} - 0{,}11\,\hat{\jmath})}{0{,}008302}
\end{equation*}

\begin{equation*}
\vec{E}_2 = \frac{-17980\,(0{,}17\,\hat{\imath} - 0{,}11\,\hat{\jmath})}{0{,}008302}
= \frac{-3056{,}6\,\hat{\imath} + 1977{,}8\,\hat{\jmath}}{0{,}008302}
\end{equation*}

\begin{equation*}
\boxed{\vec{E}_2 = -368{,}1\,\hat{\imath} + 238{,}2\,\hat{\jmath} \;\;\text{[kN/C]}}
\end{equation*}

\subsection*{Campo de $q_3$ en $P$}

\begin{equation*}
\vec{E}_3 = K\,\frac{q_3\,\vec{r}_{3P}}{|\vec{r}_{3P}|^3}
= (8{,}99\times10^9)\,\frac{(-3\times10^{-6})(0{,}28\,\hat{\imath} + 0{,}14\,\hat{\jmath})}{0{,}030679}
\end{equation*}

\begin{equation*}
\vec{E}_3 = \frac{-26970\,(0{,}28\,\hat{\imath} + 0{,}14\,\hat{\jmath})}{0{,}030679}
= \frac{-7551{,}6\,\hat{\imath} - 3775{,}8\,\hat{\jmath}}{0{,}030679}
\end{equation*}

\begin{equation*}
\boxed{\vec{E}_3 = -246{,}1\,\hat{\imath} - 123{,}1\,\hat{\jmath} \;\;\text{[kN/C]}}
\end{equation*}

%========================================
% PASO 3: SUPERPOSICION
%========================================
\section*{Paso 3: Campo El\'ectrico Neto por Superposici\'on}

\begin{equation*}
\vec{E}_{\text{neto}} = \vec{E}_1 + \vec{E}_2 + \vec{E}_3
\end{equation*}

\textbf{Componente $\hat{\imath}$:}
\begin{equation*}
E_x = 108{,}5 + (-368{,}1) + (-246{,}1) = 108{,}5 - 368{,}1 - 246{,}1 = -505{,}7\;\text{kN/C}
\end{equation*}

\textbf{Componente $\hat{\jmath}$:}
\begin{equation*}
E_y = -230{,}5 + 238{,}2 + (-123{,}1) = -230{,}5 + 238{,}2 - 123{,}1 = -115{,}4\;\text{kN/C}
\end{equation*}

\begin{equation*}
\boxed{\vec{E}_{\text{neto}} = -505{,}7\,\hat{\imath} - 115{,}4\,\hat{\jmath} \;\;\text{[kN/C]}}
\end{equation*}

%========================================
% PASO 4: MAGNITUD Y DIRECCION
%========================================
\section*{Paso 4: Magnitud y Direcci\'on del Campo Neto}

\subsection*{Magnitud}

\begin{equation*}
|\vec{E}_{\text{neto}}| = \sqrt{E_x^2 + E_y^2}
= \sqrt{(-505{,}7)^2 + (-115{,}4)^2}
= \sqrt{255{,}733 + 13{,}317}
\end{equation*}

\begin{equation*}
|\vec{E}_{\text{neto}}| = \sqrt{269{,}050}\;\text{[kN/C]}
\end{equation*}

\begin{equation*}
\boxed{|\vec{E}_{\text{neto}}| \approx 518{,}7\;\text{kN/C}}
\end{equation*}

\subsection*{Direcci\'on}

Como $E_x < 0$ y $E_y < 0$, el vector $\vec{E}_{\text{neto}}$ apunta al \textbf{tercer cuadrante}.

El \'angulo de referencia respecto al eje $-x$ es:
\begin{equation*}
\alpha = \arctan\!\left(\frac{|E_y|}{|E_x|}\right)
= \arctan\!\left(\frac{115{,}4}{505{,}7}\right)
= \arctan(0{,}2282)
\approx 12{,}84^\circ
\end{equation*}

El \'angulo medido desde el eje $+x$ (sentido antihorario) es:
\begin{equation*}
\boxed{\theta = 180^\circ + 12{,}84^\circ = 192{,}84^\circ}
\end{equation*}

%========================================
% INTERPRETACION FISICA
%========================================
\section*{Interpretaci\'on F\'isica}

\begin{itemize}
    \item El campo neto apunta predominantemente en la direcci\'on $-\hat{\imath}$ (tercer cuadrante), lo que refleja la influencia dominante de las cargas negativas $q_2$ y $q_3$, cuyo campo en $P$ se orienta \textbf{hacia} ellas.

    \item $q_3$ est\'a m\'as lejos de $P$ que $q_2$ ($r_{3P} \approx 31{,}3$ cm vs.\ $r_{2P} \approx 20{,}2$ cm), pero su mayor valor $|q_3|=3\,\mu$C la mantiene como contribuyente significativa.

    \item La carga positiva $q_1$ genera el \'unico campo con componente $+\hat{\imath}$, pero no logra compensar la suma de las dos negativas.

    \item Una carga de prueba positiva colocada en $P$ experimentar\'ia una fuerza de magnitud $F = qE_{\text{neto}}$, orient\'andose hacia el tercer cuadrante (hacia las cargas negativas).
\end{itemize}
