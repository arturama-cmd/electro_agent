%========================================
% ENUNCIADO
%========================================
\section*{Enunciado del Problema}

La mol\'ecula de \'acido clorh\'idrico (HCl) se modela como un dipolo el\'ectrico
formado por el ion Cl$^-$ (carga $-q$) y el ion H$^+$ (carga $+q$), separados por
$d = 0{,}127$~nm. Determinar el \textbf{campo el\'ectrico resultante} en el punto~$P$,
ubicado a $a = 0{,}5$~nm del ion Cl$^-$, siendo la recta $P$--Cl$^-$ inclinada
$20^\circ$ respecto a la vertical (v\'ease diagrama).

%========================================
% DATOS NUMERICOS
%========================================
\section*{Datos Num\'ericos}

\[
q = 1{,}6\times10^{-19}\ [\text{C}],\qquad
d = 0{,}127\times10^{-9}\ [\text{m}],\qquad
a = 0{,}5\times10^{-9}\ [\text{m}],\qquad
K = 9\times10^{9}\ \left[\frac{\text{N\,m}^2}{\text{C}^2}\right]
\]

%========================================
% DESARROLLO
%========================================
\section*{Desarrollo}

\subsection*{1.\; Sistema de coordenadas y posici\'on de $P$}

Se ubica Cl$^-$ en el origen y H$^+$ sobre el eje $+x$:
\[
\text{Cl}^-:\;(0,\,0)\quad[\text{nm}],\qquad
\text{H}^+:\;(d,\,0) = (0{,}127,\,0)\quad[\text{nm}]
\]

El punto $P$ se encuentra a distancia $a = 0{,}5$~nm de Cl$^-$, con la recta
$P$--Cl$^-$ inclinada $20^\circ$ a la izquierda de la vertical:
\[
P = \bigl(-a\sin20^\circ,\;\;a\cos20^\circ\bigr)
= \bigl(-0{,}5\times0{,}342,\;\;0{,}5\times0{,}940\bigr)\ \text{nm}
= (-0{,}171,\;\;0{,}470)\ \text{nm}
\]

\subsection*{2.\; Campo el\'ectrico de Cl$^-$ en $P$}

El vector de desplazamiento de Cl$^-$ a $P$ es (con $|\vec{r}_{-\to P}| = a$):
\[
\vec{r}_{-\to P} = P - \text{Cl}^-
= (-a\sin20^\circ,\;a\cos20^\circ)
= (-0{,}171,\;0{,}470)\times10^{-9}\ [\text{m}]
\]

Aplicando la Ley de Coulomb vectorial con $q_- = -1{,}6\times10^{-19}$~C:
\[
\vec{E}_{\text{Cl}^-}
= K\frac{q_-}{|\vec{r}_{-\to P}|^3}\,\vec{r}_{-\to P}
= \frac{9\times10^{9}\cdot(-1{,}6\times10^{-19})}{(0{,}5\times10^{-9})^3}
  \cdot(-0{,}171,\;0{,}470)\times10^{-9}
\]
\[
= \frac{-14{,}4\times10^{-10}}{1{,}25\times10^{-28}}
  \cdot(-0{,}171,\;0{,}470)\times10^{-9}
= -11{,}52\times10^{18}\cdot(-0{,}171,\;0{,}470)\times10^{-9}
\]
\[
\vec{E}_{\text{Cl}^-}
= \bigl(+1{,}97,\;-5{,}41\bigr)\times10^{9}\ [\text{N/C}]
\]

\subsection*{3.\; Distancia de H$^+$ a $P$}

\[
\vec{r}_{H^+\to P}
= P - H^+
= (-0{,}171 - 0{,}127,\;\;0{,}470 - 0)\times10^{-9}
= (-0{,}298,\;\;0{,}470)\times10^{-9}\ [\text{m}]
\]

\[
r_+
= \sqrt{(0{,}298)^2 + (0{,}470)^2}\times10^{-9}
= \sqrt{0{,}0888 + 0{,}2209}\times10^{-9}
= \sqrt{0{,}3097}\times10^{-9}
= 0{,}5565\times10^{-9}\ [\text{m}]
\]

Vector unitario de H$^+$ hacia $P$:
\[
\hat{r}_{H^+\to P}
= \frac{(-0{,}298,\;\;0{,}470)}{0{,}5565}
= (-0{,}536,\;\;0{,}845)
\]

\subsection*{4.\; Campo el\'ectrico de H$^+$ en $P$}

Aplicando la Ley de Coulomb vectorial con $q_+ = +1{,}6\times10^{-19}$~C:
\[
\vec{E}_{H^+}
= K\frac{q_+}{|\vec{r}_{H^+\to P}|^3}\,\vec{r}_{H^+\to P}
= \frac{9\times10^{9}\cdot1{,}6\times10^{-19}}{(0{,}5565\times10^{-9})^3}
  \cdot(-0{,}298,\;0{,}470)\times10^{-9}
\]

Con $(0{,}5565)^3 = 0{,}1724$, es decir $(0{,}5565\times10^{-9})^3 = 1{,}724\times10^{-28}$~m$^3$:
\[
= \frac{14{,}4\times10^{-10}}{1{,}724\times10^{-28}}
  \cdot(-0{,}298,\;0{,}470)\times10^{-9}
= 8{,}354\times10^{18}\cdot(-0{,}298,\;0{,}470)\times10^{-9}
\]
\[
\vec{E}_{H^+}
= \bigl(-2{,}49,\;+3{,}93\bigr)\times10^{9}\ [\text{N/C}]
\]

\subsection*{5.\; Campo el\'ectrico total en $P$}

Por el principio de superposici\'on:

\begin{align*}
\vec{E}_P
&= \vec{E}_{\text{Cl}^-} + \vec{E}_{H^+} \\[6pt]
&= \bigl(+1{,}97 + (-2{,}49)\bigr)\,\hat{\imath}\times10^{9}
   \;+\; \bigl(-5{,}41 + 3{,}93\bigr)\,\hat{\jmath}\times10^{9}\ [\text{N/C}]\\[6pt]
&= \bigl(-0{,}52\,\hat{\imath} - 1{,}48\,\hat{\jmath}\bigr)\times10^{9}\ [\text{N/C}]
\end{align*}

%========================================
% RESULTADO FINAL
%========================================
\section*{Resultado Final}

\[
\boxed{
\vec{E}_P
= \bigl(-0{,}52\,\hat{\imath} - 1{,}48\,\hat{\jmath}\bigr)\times10^{9}\ [\text{N/C}]
}
\]

\vspace{0.4cm}

\textbf{M\'odulo:}
\[
|\vec{E}_P|
= \sqrt{(0{,}52)^2+(1{,}48)^2}\times10^{9}
= \sqrt{0{,}270 + 2{,}190}\times10^{9}
= \sqrt{2{,}460}\times10^{9}
\approx \boxed{1{,}57\times10^{9}\ [\text{N/C}]}
\]

\textbf{Direcci\'on} (\'angulo respecto a la vertical descendente $-\hat{\jmath}$,
hacia el lado de Cl$^-$):
\[
\alpha
= \arctan\!\left(\frac{|E_x|}{|E_y|}\right)
= \arctan\!\left(\frac{0{,}52}{1{,}48}\right)
= \arctan(0{,}351)
\approx \boxed{19{,}4^\circ}
\]

%========================================
% INTERPRETACION FISICA
%========================================
\section*{Interpretaci\'on F\'isica}

\begin{itemize}
    \item $\vec{E}_P$ apunta principalmente en la direcci\'on $-\hat{\jmath}$
    (hacia abajo) porque la componente vertical de $\vec{E}_{\text{Cl}^-}$
    ($-5{,}41\times10^{9}$~N/C) supera en m\'odulo a la de $\vec{E}_{H^+}$
    ($+3{,}93\times10^{9}$~N/C), al ser Cl$^-$ el ion m\'as cercano a $P$.

    \item La componente horizontal neta es peque\~na y negativa
    ($-0{,}52\times10^{9}$~N/C): H$^+$ est\'a m\'as alejado y con mayor
    inclinaci\'on respecto a $P$, lo que hace que su campo ``jale'' hacia la
    izquierda m\'as de lo que el de Cl$^-$ empuja hacia la derecha.

    \item Aunque $|\vec{E}_{\text{Cl}^-}|$ es solo un $24\%$ mayor que
    $|\vec{E}_{H^+}|$, el m\'odulo resultante ($1{,}57\times10^{9}$~N/C) es
    m\'as de \textit{tres veces} menor que cada campo individual: esto ilustra
    la fuerte \textbf{cancelaci\'on vectorial} caracter\'istica de los dipolos.

    \item El \'angulo del campo resultante ($\approx 19{,}4^\circ$ de la vertical)
    pr\'acticamente coincide con el \'angulo geom\'etrico de la configuraci\'on
    ($20^\circ$), revelando la fuerte influencia de la carga m\'as cercana
    (Cl$^-$) en la direcci\'on del campo neto.

    \item \textbf{Nota metodol\'ogica}: no se usa la aproximaci\'on de dipolo
    ($r \gg d$) porque $a/d \approx 4$ no es suficientemente grande. Se aplica
    la \textbf{superposici\'on exacta} de los campos de las dos cargas puntuales.
\end{itemize}
