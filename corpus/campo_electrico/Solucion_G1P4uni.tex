\section*{Enunciado del Problema}

Dos cargas puntuales est\'an ubicadas sobre el eje $X$:
\[
q_1 = +2{,}5\ [\mu\text{C}] \quad \text{en } x = 0\ \text{[m]},
\qquad
q_2 = -3{,}5\ [\mu\text{C}] \quad \text{en } x = 0{,}6\ \text{[m]}
\]
Encuentre el punto sobre el eje $X$ donde, al colocar una carga puntual de
magnitud $q = 3\ [\mu\text{C}]$, la \textbf{fuerza neta} sobre ella sea nula.

%========================================
% ANALISIS CUALITATIVO
%========================================
\section*{An\'alisis Cualitativo}

La carga de prueba $q = +3\ \mu\text{C}$ es positiva. Se analizan las tres regiones del eje:

\begin{itemize}
    \item \textbf{Regi\'on A} (izquierda de $q_1$): $q_1$ (positiva) \textit{repele} a $q$ hacia la izquierda
    $(-\hat{\imath})$; $q_2$ (negativa) \textit{atrae} a $q$ hacia la derecha $(+\hat{\imath})$.
    Las fuerzas son opuestas $\Rightarrow$ pueden cancelarse. \checkmark

    \item \textbf{Regi\'on B} (entre $q_1$ y $q_2$): $q_1$ repele hacia la derecha $(+\hat{\imath})$ y
    $q_2$ atrae tambi\'en hacia la derecha $(+\hat{\imath})$. Las fuerzas son \textit{paralelas}
    $\Rightarrow$ no puede haber equilibrio. $\times$

    \item \textbf{Regi\'on C} (derecha de $q_2$): $q_1$ repele hacia la derecha y $q_2$ atrae
    hacia la izquierda. Las fuerzas son opuestas $\Rightarrow$ candidata, pero se verificar\'a
    si existe soluci\'on f\'isica. \checkmark (por verificar)
\end{itemize}

Solo deben calcularse las regiones \textbf{A} y \textbf{C}.

%========================================
% SITUACION A
%========================================
\section*{Situaci\'on A: $q$ a la izquierda de $q_1$}

Sea $x > 0$ la distancia entre la carga de prueba $q$ y $q_1$.
Entonces la distancia entre $q$ y $q_2$ es $x + 0{,}6$ [m].

\subsection*{Fuerzas sobre $q$}

Usando la Ley de Coulomb vectorial con $K = 9\times10^9\ \text{[N\,m}^2/\text{C}^2\text{]}$:

\begin{align*}
\vec{F}_{q_1} &= K\frac{|q_1|\,q}{x^2}\,(-\hat{\imath})
= \frac{9\times10^9\cdot(2{,}5\times10^{-6})(3\times10^{-6})}{x^2}\,(-\hat{\imath})
= \frac{0{,}0675}{x^2}\,(-\hat{\imath})\ \text{[N]} \\[8pt]
\vec{F}_{q_2} &= K\frac{|q_2|\,q}{(x+0{,}6)^2}\,(+\hat{\imath})
= \frac{9\times10^9\cdot(3{,}5\times10^{-6})(3\times10^{-6})}{(x+0{,}6)^2}\,(+\hat{\imath})
= \frac{0{,}0945}{(x+0{,}6)^2}\,(+\hat{\imath})\ \text{[N]}
\end{align*}

\subsection*{Condici\'on de equilibrio}

\[
\vec{F}_{q_1} + \vec{F}_{q_2} = \vec{0}
\quad\Rightarrow\quad
\frac{0{,}0945}{(x+0{,}6)^2} = \frac{0{,}0675}{x^2}
\]

Productos cruzados:
\[
0{,}0945\,x^2 = 0{,}0675\,(x^2 + 1{,}2x + 0{,}36)
\]
\[
0{,}0945\,x^2 = 0{,}0675\,x^2 + 0{,}081\,x + 0{,}0243
\]

\[
\boxed{0{,}027\,x^2 - 0{,}081\,x - 0{,}0243 = 0}
\]

\subsection*{Resoluci\'on de la ecuaci\'on de segundo grado}

Aplicando la f\'ormula cuadr\'atica con $a=0{,}027$, $b=-0{,}081$, $c=-0{,}0243$:

\[
x = \frac{0{,}081 \pm \sqrt{(-0{,}081)^2 - 4(0{,}027)(-0{,}0243)}}{2(0{,}027)}
= \frac{0{,}081 \pm \sqrt{0{,}006561 + 0{,}002624}}{0{,}054}
= \frac{0{,}081 \pm \sqrt{0{,}009185}}{0{,}054}
\]

\[
x = \frac{0{,}081 \pm 0{,}09584}{0{,}054}
\]

\[
x_1 = \frac{0{,}081 + 0{,}09584}{0{,}054} = \frac{0{,}17684}{0{,}054} \approx \mathbf{3{,}275}\ \text{[m]}
\quad\checkmark
\]
\[
x_2 = \frac{0{,}081 - 0{,}09584}{0{,}054} = \frac{-0{,}01484}{0{,}054} \approx -0{,}275\ \text{[m]}
\quad\times\ \text{(descartada: }x \text{ debe ser } >0\text{)}
\]

$\Rightarrow$ \textbf{En la Situaci\'on A}: $x = 3{,}275$ [m], es decir, la carga de prueba debe colocarse
$3{,}275$~m a la izquierda de $q_1$.

%========================================
% SITUACION C
%========================================
\section*{Situaci\'on C: $q$ a la derecha de $q_2$}

Sea $x > 0$ la distancia entre $q$ y $q_2$.
La distancia entre $q$ y $q_1$ es $x + 0{,}6$ [m].

\subsection*{Fuerzas sobre $q$}

\begin{align*}
\vec{F}_{q_2} &= K\frac{|q_2|\,q}{x^2}\,(-\hat{\imath})
= \frac{0{,}0945}{x^2}\,(-\hat{\imath})\ \text{[N]}
\quad\text{($q_2$ negativa atrae hacia la izquierda)}\\[8pt]
\vec{F}_{q_1} &= K\frac{|q_1|\,q}{(x+0{,}6)^2}\,(+\hat{\imath})
= \frac{0{,}0675}{(x+0{,}6)^2}\,(+\hat{\imath})\ \text{[N]}
\quad\text{($q_1$ positiva repele hacia la derecha)}
\end{align*}

\subsection*{Condici\'on de equilibrio}

\[
\frac{0{,}0945}{x^2} = \frac{0{,}0675}{(x+0{,}6)^2}
\]
\[
0{,}0945\,(x^2 + 1{,}2x + 0{,}36) = 0{,}0675\,x^2
\]
\[
0{,}0945\,x^2 + 0{,}1134\,x + 0{,}0340 = 0{,}0675\,x^2
\]

\[
\boxed{0{,}027\,x^2 + 0{,}1134\,x + 0{,}0340 = 0}
\]

Discriminante:
\[
\Delta = (0{,}1134)^2 - 4(0{,}027)(0{,}0340) = 0{,}012860 - 0{,}003672 = 0{,}009188
\]
\[
x = \frac{-0{,}1134 \pm \sqrt{0{,}009188}}{0{,}054}
= \frac{-0{,}1134 \pm 0{,}09585}{0{,}054}
\]
\[
x_1 = \frac{-0{,}1134 + 0{,}09585}{0{,}054} = \frac{-0{,}01755}{0{,}054} \approx -0{,}325\ \text{[m]}
\quad\times
\]
\[
x_2 = \frac{-0{,}1134 - 0{,}09585}{0{,}054} = \frac{-0{,}20925}{0{,}054} \approx -3{,}875\ \text{[m]}
\quad\times
\]

Ambas ra\'ices son negativas ($x$ debe ser $\geq 0$).
$\Rightarrow$ \textbf{No existe soluci\'on f\'isicamente v\'alida en la Regi\'on C.} $\times$

%========================================
% RESULTADO FINAL
%========================================
\section*{Resultado Final}

La \'unica posici\'on de equilibrio se encuentra en la \textbf{Regi\'on A}:

\[
\boxed{
\vec{F}_{\text{neta}} = 0 \quad \text{a } 3{,}275\ \text{[m] a la izquierda de } q_1\ (= {+}2{,}5\ \mu\text{C})
}
\]

%========================================
% INTERPRETACION FISICA
%========================================
\section*{Interpretaci\'on F\'isica}

\begin{itemize}
    \item La soluci\'on existe \textit{solo} a la izquierda del sistema porque $q_1$ es positiva
    (repele a $q$) y $q_2$ es negativa (atrae a $q$): desde la izquierda, estas dos acciones
    apuntan en direcciones opuestas y pueden equilibrarse.

    \item La regi\'on C no ofrece soluci\'on porque $|q_2| > |q_1|$: la atracci\'on de $q_2$
    siempre supera la repulsi\'on de $q_1$ en esa zona, independientemente de la distancia.

    \item La distancia de equilibrio es relativamente grande ($3{,}275$~m comparado con la
    separaci\'on de $0{,}6$~m entre las cargas) porque la fuerza de $q_2$ cae con el cuadrado
    de la distancia; para que $|\vec{F}_{q_2}|$ (con $r_{q_2} = 3{,}875$~m) iguale a
    $|\vec{F}_{q_1}|$ (con $r_{q_1} = 3{,}275$~m) siendo $|q_2|$ solo un 40\% mayor,
    se requiere una separaci\'on sustancial.

    \item Este tipo de problema, en que se busca el punto de campo el\'ectrico nulo de un sistema
    de cargas, es equivalente a hallar los \textbf{puntos de equilibrio del campo} $\vec{E} = \vec{0}$,
    y tiene aplicaci\'on directa en el dise\~no de trampas electrost\'aticas y en el an\'alisis
    de la distribuci\'on de carga en conductores.
\end{itemize}
