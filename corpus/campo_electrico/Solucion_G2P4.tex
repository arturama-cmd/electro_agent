%========================================
% ENUNCIADO
%========================================
\section*{Enunciado del Problema}

En la configuraci\'on de la Figura~5, una carga $+q$ se ubica en la esquina superior
izquierda y una carga $-2q$ directamente debajo de ella, separadas verticalmente
por $a = 10$~[cm]. El punto P se encuentra a distancia horizontal $b = 15$~[cm]
de la carga $+q$, en el mismo nivel vertical que ella. Con $q = 8{,}0$~[nC]:

\begin{enumerate}[(a)]
    \item Determinar el campo el\'ectrico resultante en el punto P.
    \item Si una carga puntual $q_2 = -30$~[nC] se ubica en P, determinar la fuerza
          el\'ectrica neta ejercida sobre ella.
\end{enumerate}

%========================================
% DATOS
%========================================
\section*{Datos del problema}

\[
    q = 8{,}0\ [\text{nC}] = 8{,}0 \times 10^{-9}\ [\text{C}], \qquad
    a = 0{,}10\ [\text{m}], \qquad
    b = 0{,}15\ [\text{m}]
\]
\[
    K = 9 \times 10^{9}\ \left[\frac{\text{N\,m}^2}{\text{C}^2}\right]
\]

% SISTEMA DE COORDENADAS
%========================================

    Punto P &  ---                           & $(b,\ a) = (0{,}15;\ 0{,}10)\ [\text{m}]$ \\

%========================================
% ANALISIS CUALITATIVO
%========================================
\section*{An\'alisis Cualitativo}

En el punto P act\'uan dos campos el\'ectricos:

\begin{itemize}
    \item $\vec{E}_1$ producido por $+q$ en $(0,a)$: P est\'a directamente a la derecha
    de $+q$ (mismo nivel $y$), por lo que $\vec{E}_1$ apunta en direcci\'on $+\hat{\imath}$.

    \item $\vec{E}_2$ producido por $-2q$ en el origen: el campo apunta \textit{hacia}
    la carga negativa, es decir, desde P hacia el origen. Como el origen est\'a
    abajo-izquierda respecto a P, $\vec{E}_2$ tiene componentes $-\hat{\imath}$
    y $-\hat{\jmath}$.
\end{itemize}

Se espera un campo resultante con componente $x$ negativa (pues $|-2q| > |{+q}|$)
y componente $y$ negativa (solo aportada por $-2q$).

%========================================
% PARTE (a): CAMPO ELECTRICO EN P
%========================================
\section*{Parte (a): Campo el\'ectrico en P}

\subsection*{Vectores de posici\'on hacia P}

Desde la carga $+q$ en $(0,\ a)$ hasta P en $(b,\ a)$:
\[
    \vec{r}_1 = (b-0)\,\hat{\imath} + (a-a)\,\hat{\jmath} = b\,\hat{\imath}
    = 0{,}15\,\hat{\imath}\ [\text{m}]
    \qquad \Rightarrow \qquad r_1 = 0{,}15\ [\text{m}]
\]

Desde la carga $-2q$ en $(0,\ 0)$ hasta P en $(b,\ a)$:
\[
    \vec{r}_2 = b\,\hat{\imath} + a\,\hat{\jmath}
    = 0{,}15\,\hat{\imath} + 0{,}10\,\hat{\jmath}\ [\text{m}]
\]
\[
    r_2 = \sqrt{b^2 + a^2} = \sqrt{(0{,}15)^2 + (0{,}10)^2}
    = \sqrt{0{,}0325} \approx 0{,}18\ [\text{m}]
    \qquad \Rightarrow \qquad r_2^3 = (0{,}18)^3 = 5{,}832\times10^{-3}\ [\text{m}^3]
\]

\subsection*{Campo $\vec{E}_1$ (aporte de $+q$)}

\[
    \vec{E}_1 = K\,\frac{q}{r_1^3}\,\vec{r}_1
    = K\,\frac{q}{r_1^2}\,\hat{\imath}
    = \frac{9\times10^9 \times 8{,}0\times10^{-9}}{(0{,}15)^2}\,\hat{\imath}
    = \frac{72}{0{,}0225}\,\hat{\imath}
\]

\[
    \boxed{\vec{E}_1 = 3\,200\,\hat{\imath}\ [\text{N/C}]}
\]

\subsection*{Campo $\vec{E}_2$ (aporte de $-2q$)}

\[
    \vec{E}_2 = K\,\frac{(-2q)}{r_2^3}\,\vec{r}_2
    = \frac{9\times10^9 \times (-16{,}0\times10^{-9})}{5{,}832\times10^{-3}}
      \left(0{,}15\,\hat{\imath} + 0{,}10\,\hat{\jmath}\right)
\]

\[
    \vec{E}_2 = \frac{-144}{5{,}832\times10^{-3}}
      \left(0{,}15\,\hat{\imath} + 0{,}10\,\hat{\jmath}\right)
    = -24\,691{,}4\left(0{,}15\,\hat{\imath} + 0{,}10\,\hat{\jmath}\right)
\]

\begin{align*}
    E_{2x} &= -24\,691{,}4 \times 0{,}15 = -3\,703{,}7\ [\text{N/C}]\\[4pt]
    E_{2y} &= -24\,691{,}4 \times 0{,}10 = -2\,469{,}1\ [\text{N/C}]
\end{align*}

\[
    \boxed{\vec{E}_2 = -3\,703{,}7\,\hat{\imath} - 2\,469{,}1\,\hat{\jmath}\ [\text{N/C}]}
\]

\subsection*{Campo resultante en P}

\[
    \vec{E}_P = \vec{E}_1 + \vec{E}_2
    = \bigl(3\,200 - 3\,703{,}7\bigr)\,\hat{\imath}
    + \bigl(0 - 2\,469{,}1\bigr)\,\hat{\jmath}
\]

\[
    \boxed{\vec{E}_P = -503{,}7\,\hat{\imath} - 2\,469{,}1\,\hat{\jmath}\ [\text{N/C}]}
\]

%========================================
% PARTE (b): FUERZA SOBRE q2 EN P
%========================================
\section*{Parte (b): Fuerza el\'ectrica sobre $q_2 = -30$~[nC] en P}

\[
    \vec{F} = q_2\,\vec{E}_P
    = (-30\times10^{-9})\times\left(-503{,}7\,\hat{\imath} - 2\,469{,}1\,\hat{\jmath}\right)
\]

\begin{align*}
    F_x &= (-30\times10^{-9})\times(-503{,}7) = +15{,}1\times10^{-6}\ [\text{N}] = +15{,}1\ [\mu\text{N}]\\[4pt]
    F_y &= (-30\times10^{-9})\times(-2\,469{,}1) = +74{,}1\times10^{-6}\ [\text{N}] = +74{,}1\ [\mu\text{N}]
\end{align*}

\[
    \boxed{\vec{F} = 15{,}1\,\hat{\imath} + 74{,}1\,\hat{\jmath}\ [\mu\text{N}]}
\]

%========================================
% INTERPRETACION FISICA
%========================================
\section*{Interpretaci\'on F\'isica}

\begin{itemize}
    \item Con el origen en $-2q$, el vector $\vec{r}_1 = b\,\hat{\imath}$ es puramente
    horizontal (no tiene componente $y$) porque $+q$ y P est\'an a la misma altura $y = a$.
    Esto simplifica el c\'alculo de $\vec{E}_1$, que resulta solo en direcci\'on
    $+\hat{\imath}$.

    \item El vector $\vec{r}_2 = b\,\hat{\imath} + a\,\hat{\jmath}$ apunta desde el
    origen hacia P, con ambas componentes positivas. Sin embargo, como $-2q$ es negativa,
    $\vec{E}_2$ resulta en direcci\'on $-\hat{\imath}$ y $-\hat{\jmath}$: el campo
    apunta \textit{hacia} la carga fuente negativa.

    \item La componente $E_{Px} = -503{,}7$~[N/C] es negativa porque el aporte de
    $-2q$ en $x$ supera al de $+q$: aunque $-2q$ est\'a m\'as lejos ($r_2 \approx 0{,}18$~m
    vs.\ $r_1 = 0{,}15$~m), su magnitud es el doble.

    \item La fuerza sobre $q_2 = -30$~nC apunta en sentido contrario al campo
    (signos positivos en $\hat{\imath}$ y $\hat{\jmath}$): las cargas negativas son
    aceleradas en direcci\'on opuesta al campo el\'ectrico.
\end{itemize}
