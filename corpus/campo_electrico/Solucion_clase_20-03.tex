\section*{Enunciado del Problema}

Cuatro cargas puntuales se ubican en el plano $xy$ con coordenadas en cent\'imetros:

\begin{center}
\begin{tabular}{|c|c|c|}
\hline
\textbf{Carga} & \textbf{Valor} & \textbf{Posici\'on} [cm] \\
\hline
$q_1$ & $+5$ [nC] & $(12,\ 7)$ \\
$q_2$ & $-4$ [nC] & $(-2,\ 17)$ \\
$q_3$ & $+12$ [nC] & $(-10,\ -4)$ \\
$q_4$ & $-18$ [nC] & $(8,\ -15)$ \\
\hline
\end{tabular}
\end{center}

Donde $1\,\text{nC} = 10^{-9}\,\text{C}$. \quad \textbf{Determinar la fuerza resultante sobre $q_1$.}

%========================================
% FORMULA
%========================================
\section*{Ley de Coulomb -- Forma Vectorial}

Usando las cargas en valor absoluto y el vector de separaci\'on \textbf{sin normalizar}:

\begin{equation*}
\vec{F}_{ij} = k\,\frac{|q_i|\,|q_j|}{|\vec{r}_{ij}|^3}\,\vec{r}_{ij}
\qquad \text{con} \quad k = 9\times10^9\;\left[\frac{\text{N\,m}^2}{\text{C}^2}\right]
\end{equation*}

donde $\vec{r}_{ij}$ es el vector de separaci\'on apuntando en la \textbf{direcci\'on f\'isica de la fuerza}:
\begin{itemize}
    \item Cargas de \textbf{igual signo} (repulsi\'on): $\vec{r}_{ij}$ apunta \emph{desde la carga fuente hacia $q_1$}.
    \item Cargas de \textbf{signo opuesto} (atracci\'on): $\vec{r}_{ij}$ apunta \emph{desde $q_1$ hacia la carga fuente}.
\end{itemize}

%========================================
% PASO 1: ANALISIS CUALITATIVO Y VECTORES
%========================================
\section*{Paso 1: An\'alisis Cualitativo y Vectores de Direcci\'on}

Convirtiendo coordenadas a metros y determinando el tipo de interacci\'on:

\subsection*{Fuerza $\vec{F}_{21}$: $q_2 = -4\,\text{nC}$ sobre $q_1 = +5\,\text{nC}$}

Signos \textbf{opuestos} $\Rightarrow$ \textbf{atracci\'on}: la fuerza sobre $q_1$ apunta \emph{hacia} $q_2$.

\begin{equation*}
\vec{r}_{21} = \vec{r}_2 - \vec{r}_1
= (-0{,}02 - 0{,}12)\,\hat{\imath} + (0{,}17 - 0{,}07)\,\hat{\jmath}
= -0{,}14\,\hat{\imath} + 0{,}10\,\hat{\jmath}\quad[\text{m}]
\end{equation*}
\begin{equation*}
|\vec{r}_{21}| = \sqrt{(-0{,}14)^2 + (0{,}10)^2} = \sqrt{0{,}0296} \approx 0{,}1720\;\text{m}
\end{equation*}

\subsection*{Fuerza $\vec{F}_{31}$: $q_3 = +12\,\text{nC}$ sobre $q_1 = +5\,\text{nC}$}

Signos \textbf{iguales} $\Rightarrow$ \textbf{repulsi\'on}: la fuerza sobre $q_1$ apunta \emph{alejándose} de $q_3$.

\begin{equation*}
\vec{r}_{31} = \vec{r}_1 - \vec{r}_3
= (0{,}12 - (-0{,}10))\,\hat{\imath} + (0{,}07 - (-0{,}04))\,\hat{\jmath}
= 0{,}22\,\hat{\imath} + 0{,}11\,\hat{\jmath}\quad[\text{m}]
\end{equation*}
\begin{equation*}
|\vec{r}_{31}| = \sqrt{(0{,}22)^2 + (0{,}11)^2} = \sqrt{0{,}0605} \approx 0{,}2460\;\text{m}
\end{equation*}

\subsection*{Fuerza $\vec{F}_{41}$: $q_4 = -18\,\text{nC}$ sobre $q_1 = +5\,\text{nC}$}

Signos \textbf{opuestos} $\Rightarrow$ \textbf{atracci\'on}: la fuerza sobre $q_1$ apunta \emph{hacia} $q_4$.

\begin{equation*}
\vec{r}_{41} = \vec{r}_4 - \vec{r}_1
= (0{,}08 - 0{,}12)\,\hat{\imath} + (-0{,}15 - 0{,}07)\,\hat{\jmath}
= -0{,}04\,\hat{\imath} - 0{,}22\,\hat{\jmath}\quad[\text{m}]
\end{equation*}
\begin{equation*}
|\vec{r}_{41}| = \sqrt{(-0{,}04)^2 + (-0{,}22)^2} = \sqrt{0{,}0500} \approx 0{,}2236\;\text{m}
\end{equation*}

%========================================
% PASO 2: FUERZAS INDIVIDUALES
%========================================
\section*{Paso 2: C\'alculo de Cada Fuerza}

\subsection*{Fuerza $\vec{F}_{21}$}

\begin{equation*}
\vec{F}_{21} = \left[9\times10^9\;\frac{(5\times10^{-9})(4\times10^{-9})}{(0{,}1720)^3}\right]
\left(-0{,}14\,\hat{\imath} + 0{,}10\,\hat{\jmath}\right)
\end{equation*}

\begin{equation*}
\vec{F}_{21} = \left[\frac{1{,}80\times10^{-7}}{5{,}09\times10^{-3}}\right]
\left(-0{,}14\,\hat{\imath} + 0{,}10\,\hat{\jmath}\right)
= 3{,}54\times10^{-5}\,
\left(-0{,}14\,\hat{\imath} + 0{,}10\,\hat{\jmath}\right)
\end{equation*}

\begin{equation*}
\boxed{\vec{F}_{21} = \left(-4{,}949\,\hat{\imath} + 3{,}535\,\hat{\jmath}\right)\;\mu\text{N}}
\end{equation*}

\subsection*{Fuerza $\vec{F}_{31}$}

\begin{equation*}
\vec{F}_{31} = \left[9\times10^9\;\frac{(12\times10^{-9})(5\times10^{-9})}{(0{,}2460)^3}\right]
\left(0{,}22\,\hat{\imath} + 0{,}11\,\hat{\jmath}\right)
\end{equation*}

\begin{equation*}
\vec{F}_{31} = \left[\frac{5{,}40\times10^{-7}}{1{,}489\times10^{-2}}\right]
\left(0{,}22\,\hat{\imath} + 0{,}11\,\hat{\jmath}\right)
= 3{,}63\times10^{-5}\,
\left(0{,}22\,\hat{\imath} + 0{,}11\,\hat{\jmath}\right)
\end{equation*}

\begin{equation*}
\boxed{\vec{F}_{31} = \left(+7{,}984\,\hat{\imath} + 3{,}992\,\hat{\jmath}\right)\;\mu\text{N}}
\end{equation*}

\subsection*{Fuerza $\vec{F}_{41}$}

\begin{equation*}
\vec{F}_{41} = \left[9\times10^9\;\frac{(18\times10^{-9})(5\times10^{-9})}{(0{,}2236)^3}\right]
\left(-0{,}04\,\hat{\imath} - 0{,}22\,\hat{\jmath}\right)
\end{equation*}

\begin{equation*}
\vec{F}_{41} = \left[\frac{8{,}10\times10^{-7}}{1{,}118\times10^{-2}}\right]
\left(-0{,}04\,\hat{\imath} - 0{,}22\,\hat{\jmath}\right)
= 7{,}25\times10^{-5}\,
\left(-0{,}04\,\hat{\imath} - 0{,}22\,\hat{\jmath}\right)
\end{equation*}

\begin{equation*}
\boxed{\vec{F}_{41} = \left(-2{,}898\,\hat{\imath} - 15{,}939\,\hat{\jmath}\right)\;\mu\text{N}}
\end{equation*}

%========================================
% PASO 3: SUMA VECTORIAL
%========================================
\section*{Paso 3: Fuerza Resultante $\vec{F}_R = \vec{F}_{21} + \vec{F}_{31} + \vec{F}_{41}$}

\subsection*{Componente $x$}

\begin{equation*}
F_{Rx} = (-4{,}949) + (+7{,}984) + (-2{,}898) = +0{,}137\;\mu\text{N}
\end{equation*}

\subsection*{Componente $y$}

\begin{equation*}
F_{Ry} = (+3{,}535) + (+3{,}992) + (-15{,}939) = -8{,}412\;\mu\text{N}
\end{equation*}

\subsection*{Magnitud}

\begin{equation*}
|\vec{F}_R| = \sqrt{(0{,}137)^2 + (-8{,}412)^2}
= \sqrt{0{,}019 + 70{,}762}
= \sqrt{70{,}781}
\approx 8{,}413\;\mu\text{N}
\end{equation*}

\subsection*{Direcci\'on}

\begin{equation*}
\theta = \arctan\!\left(\frac{-8{,}412}{0{,}137}\right) \approx -89{,}1^\circ
\end{equation*}

Como $F_{Rx}>0$ y $F_{Ry}<0$: cuarto cuadrante, fuerza apuntando pr\'acticamente en $-\hat{\jmath}$.

%========================================
% INTERPRETACION FISICA
%========================================
\section*{Interpretaci\'on F\'isica}

\begin{itemize}
    \item La fuerza dominante es $\vec{F}_{41}$, con $|\vec{F}_{41}| \approx 16{,}1\,\mu\text{N}$.
          $q_4$ es la carga de mayor magnitud ($18\,\text{nC}$) y la m\'as cercana en la
          direcci\'on $-\hat{\jmath}$, por lo que su atracci\'on sobre $q_1$ es intensa y
          casi vertical hacia abajo.

    \item Las componentes $x$ se cancelan casi completamente
          ($F_{Rx} = +0{,}137\,\mu\text{N} \approx 0$),
          de modo que la fuerza resultante apunta pr\'acticamente en $-\hat{\jmath}$
          ($\theta \approx -89{,}1^\circ$).

    \item Las fuerzas $\vec{F}_{21}$ y $\vec{F}_{31}$ contribuyen ambas positivamente
          en $y$ ($+3{,}54$ y $+3{,}99\,\mu\text{N}$) pero no compensan la gran
          componente $-15{,}94\,\mu\text{N}$ de $\vec{F}_{41}$.

    \item Este problema ilustra el \textbf{Principio de Superposici\'on}: la fuerza
          total sobre $q_1$ es la suma vectorial de las contribuciones individuales,
          calculadas de forma independiente con la Ley de Coulomb.
\end{itemize}
