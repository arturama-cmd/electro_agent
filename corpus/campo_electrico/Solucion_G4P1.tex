%========================================
% ENUNCIADO
%========================================
\section*{Enunciado del Problema}

Un conductor en forma de anillo de radio $R = 7$ [cm] tiene una carga total $Q = 4{,}5\times10^{-4}$ [C] distribuida uniformemente en toda su circunferencia. Determinar la fuerza el\'ectrica ejercida sobre una carga puntual $q = -5\;[\mu\text{C}]$ situada sobre el eje del anillo en el punto $P$, a una distancia $z = 15$ [cm] del centro del anillo.

%========================================
% DATOS NUMERICOS
%========================================
\section*{Datos Num\'ericos}

\begin{itemize}
    \item $K = 9{,}00\times10^9\;\text{N}\cdot\text{m}^2/\text{C}^2$
    \item $R = 7{,}00\;\text{cm} = 0{,}0700\;\text{m}$ \quad $\Rightarrow$ \quad $R^2 = 4{,}90\times10^{-3}\;\text{m}^2$
    \item $Q = 4{,}50\times10^{-4}\;\text{C}$
    \item $z = 15{,}0\;\text{cm} = 0{,}150\;\text{m}$ \quad $\Rightarrow$ \quad $z^2 = 2{,}25\times10^{-2}\;\text{m}^2$
    \item $q = -5{,}00\;\mu\text{C} = -5{,}00\times10^{-6}\;\text{C}$
    \item $R^2 + z^2 = 0{,}0049 + 0{,}0225 = 0{,}0274\;\text{m}^2$
\end{itemize}

%========================================
% ANALISIS CUALITATIVO
%========================================
\section*{An\'alisis Cualitativo}

El anillo tiene carga positiva distribuida uniformemente. El punto $P$ est\'a sobre el \textbf{eje de simetr\'ia} del anillo (eje $z$). Por simetr\'ia, las componentes $\hat{\imath}$ y $\hat{\jmath}$ del campo se cancelan al integrar en $\theta \in [0, 2\pi]$: cada elemento $dq$ contribuye con una componente radial que se anula con el elemento opuesto. Solo sobrevive la componente axial $\hat{k}$.

Dado que $Q > 0$ y $z > 0$, el campo en $P$ apunta en $+\hat{k}$. La fuerza sobre $q < 0$ apuntará en $-\hat{k}$ (la carga negativa es atra\'ida hacia el anillo).

%========================================
% PASO 1: DENSIDAD LINEAL
%========================================
\section*{Paso 1: Densidad Lineal de Carga $\lambda$}

La carga est\'a distribuida uniformemente en la circunferencia. El elemento de carga es $dq = \lambda\,R\,d\theta$, y la carga total:
\begin{equation*}
    Q = \int_0^{2\pi} \lambda\,R\,d\theta = \lambda\,R\,2\pi
    \quad\Rightarrow\quad
    \lambda = \frac{Q}{2\pi R}
    = \frac{4{,}50\times10^{-4}}{2\pi\times0{,}0700}
    = 1{,}023\times10^{-3}\;\text{C/m}
\end{equation*}

%========================================
% PASO 2: VECTORES DE POSICION
%========================================
\section*{Paso 2: Vectores de Posici\'on}

El elemento $dq$ est\'a ubicado en el anillo a \'angulo $\theta$:
\begin{equation*}
    \vec{r}_{q} = R\cos\theta\,\hat{\imath} + R\,\text{sen}\,\theta\,\hat{\jmath}
\end{equation*}

El punto de campo $P$ est\'a sobre el eje $z$:
\begin{equation*}
    \vec{r}_{P} = z\,\hat{k}
\end{equation*}

El vector desplazamiento de la fuente al campo:
\begin{equation*}
    \vec{r}_{P} - \vec{r}_{q}
    = -R\cos\theta\,\hat{\imath} - R\,\text{sen}\,\theta\,\hat{\jmath} + z\,\hat{k}
\end{equation*}

Su norma al cubo (constante para todo $\theta$, por simetr\'ia del anillo):
\begin{equation*}
    \|\vec{r}_{P} - \vec{r}_{q}\|^3
    = \bigl(R^2\cos^2\theta + R^2\,\text{sen}^2\theta + z^2\bigr)^{3/2}
    = \bigl(R^2 + z^2\bigr)^{3/2}
\end{equation*}

%========================================
% PASO 3: CAMPO DIFERENCIAL
%========================================
\section*{Paso 3: Contribuci\'on Diferencial $d\vec{E}$}

Sustituyendo $dq = \lambda R\,d\theta$:
\begin{equation*}
    d\vec{E} = K\,\frac{\lambda R\,d\theta}{(R^2+z^2)^{3/2}}
    \Bigl[-R\cos\theta\,\hat{\imath} - R\,\text{sen}\,\theta\,\hat{\jmath} + z\,\hat{k}\Bigr]
\end{equation*}

%========================================
% PASO 4: INTEGRACION
%========================================
\section*{Paso 4: Integraci\'on en $\theta \in [0,\,2\pi]$}

\begin{equation*}
    \vec{E}
    = \frac{K\lambda R}{(R^2+z^2)^{3/2}}
    \left[
        -R\underbrace{\int_0^{2\pi}\cos\theta\,d\theta}_{=\,0}\,\hat{\imath}
        -R\underbrace{\int_0^{2\pi}\text{sen}\,\theta\,d\theta}_{=\,0}\,\hat{\jmath}
        +z\underbrace{\int_0^{2\pi}d\theta}_{=\,2\pi}\,\hat{k}
    \right]
\end{equation*}

Las integrales de $\cos\theta$ y $\text{sen}\,\theta$ en un per\'iodo completo son nulas. El resultado es:
\begin{equation*}
    \vec{E} = \frac{K\lambda R\,z\,2\pi}{(R^2+z^2)^{3/2}}\,\hat{k}
    = \frac{K\,Q\,z}{(R^2+z^2)^{3/2}}\,\hat{k}
\end{equation*}

donde se us\'o $\lambda\cdot R\cdot 2\pi = Q$.

%========================================
% PASO 5: RESULTADO NUMERICO
%========================================
\section*{Paso 5: Campo El\'ectrico y Fuerza en $P$}

\subsection*{Campo el\'ectrico}

\begin{equation*}
    \vec{E}
    = \frac{9{,}00\times10^9 \times 4{,}50\times10^{-4} \times 0{,}150}
           {(0{,}0274)^{3/2}}\,\hat{k}
\end{equation*}

\begin{align*}
    (0{,}0274)^{3/2} &= 0{,}0274\times\sqrt{0{,}0274} = 0{,}0274\times0{,}16553 = 4{,}5355\times10^{-3}\;\text{m}^3\\
    K\,Q\,z &= 9{,}00\times10^9 \times 4{,}50\times10^{-4} \times 0{,}150 = 6{,}075\times10^5\;\text{N\,m}^2/\text{C}
\end{align*}

\begin{equation*}
    \boxed{\vec{E}_P = \frac{6{,}075\times10^5}{4{,}5355\times10^{-3}}\,\hat{k}
    = 1{,}3392\times10^{8}\,\hat{k}\;\;\text{[N/C]}}
\end{equation*}

\subsection*{Fuerza el\'ectrica sobre $q$}

\begin{equation*}
    \vec{F} = q\,\vec{E}_P
    = \bigl(-5{,}00\times10^{-6}\bigr)\times1{,}3392\times10^{8}\,\hat{k}
\end{equation*}

\begin{equation*}
    \boxed{\vec{F} = -669{,}6\,\hat{k}\;\;\text{[N]}}
\end{equation*}

%========================================
% INTERPRETACION FISICA
%========================================
\section*{Interpretaci\'on F\'isica}

\begin{itemize}
    \item Por simetr\'ia axial del anillo, las componentes radiales $(\hat{\imath},\,\hat{\jmath})$ del campo se cancelan exactamente: el campo en cualquier punto del eje es estrictamente axial ($\hat{k}$).

    \item $\vec{E}_P$ apunta en $+\hat{k}$ (alejándose del anillo positivo). La fuerza $\vec{F}$ sobre $q < 0$ apunta en $-\hat{k}$: la carga negativa es \textbf{atra\'ida} hacia el anillo.

    \item La magnitud $|\vec{F}| = 669{,}6$ N es elevada porque $Q = 0{,}45$ mC es una carga muy grande y la separaci\'on $z = 15$ cm es peque\~na.

    \item La expresi\'on $\vec{E} = KQz/(R^2+z^2)^{3/2}\,\hat{k}$ tiene un m\'aximo en $z = R/\sqrt{2}$; en este problema $z = 0{,}15$ m $> R/\sqrt{2} \approx 0{,}049$ m, es decir, $P$ est\'a m\'as all\'a del m\'aximo del campo.
\end{itemize}
