\section*{Problema 2: Fem inducida en una bobina sobre un solenoide}

\noindent Una bobina de $N = 15$ vueltas y radio $R = 10\ [\text{cm}]$ rodea un solenoide muy
extenso de radio $r = 2\ [\text{cm}]$ y $n = 1000\ [\text{vueltas/m}]$. Por el solenoide circula
una corriente $I(t) = 5\sin(120t)\ [\text{A}]$. Determine la fem inducida en la bobina.

\medskip

\noindent\textbf{Campo del solenoide} (uniforme dentro, nulo fuera):
\[
    B = \mu_0\,n\,I(t)
\]
\textbf{Flujo a trav\'es de la bobina} (solo abarca el \'area del solenoide, $A = \pi r^2$):
\[
    \Phi = \int \vec{B}\cdot d\vec{S} = B\,(\pi r^2) = \mu_0\,n\,I(t)\,\pi r^2
\]

\noindent\textbf{Fem inducida} (ley de Faraday, $N$ vueltas):
\begin{align*}
    \varepsilon = -N\frac{d\Phi}{dt}
       &= -N\,\mu_0\,n\,\pi r^2\,\frac{dI}{dt} \\
       &= -15\cdot(4\pi\times10^{-7})\cdot 1000\cdot \pi (0{,}02)^2\cdot 5\cdot 120\cos(120t)
\end{align*}
\[
    \boxed{\varepsilon = -14\cos(120t)\ [\text{mV}]}
\]

\section*{Problema 6: Dos solenoides y corrientes inducidas (Faraday + Kirchhoff)}

\noindent Dos solenoides muy largos (radios $r_1 = 0{,}10\ [\text{m}]$ y $r_2 = 0{,}15\ [\text{m}]$)
tienen campos que aumentan a una tasa $\dfrac{dB}{dt} = 2\ [\text{T/s}]$. El circuito exterior
tiene resistencias de $6$, $3$ y $5\ [\Omega]$. Determine la corriente inducida en cada
resistencia.

\medskip

\noindent\textbf{Fem inducida por cada solenoide} ($\varepsilon = -\dfrac{d\Phi}{dt} = -\pi r^2\dfrac{dB}{dt}$):
\[
    \varepsilon_1 = -\pi r_1^2\frac{dB}{dt} = -\pi(0{,}10)^2(2) = -0{,}063\ [\text{V}]
\]
\[
    \varepsilon_2 = -\pi r_2^2\frac{dB}{dt} = -\pi(0{,}15)^2(2) = -0{,}141\ [\text{V}]
\]

\noindent\textbf{Ecuaciones de Kirchhoff} (con $I_1$ en $6\,\Omega$, $I_2$ en $5\,\Omega$,
$I_3$ en $3\,\Omega$):
\[
    |\varepsilon_1| - 6\,I_1 - 3\,I_3 = 0,\qquad
    |\varepsilon_2| - 5\,I_2 - 3\,I_3 = 0,\qquad
    I_3 = I_1 + I_2
\]

\noindent Resolviendo el sistema:
\[
    \boxed{I_{6\Omega} = 1{,}286\ [\text{mA}],\quad
           I_{5\Omega} = 17{,}14\ [\text{mA}],\quad
           I_{3\Omega} = 18{,}43\ [\text{mA}]}
\]

\section*{Problema 3: Corriente inducida en la secci\'on $PQ$}

\noindent El circuito de la figura (dos lazos contiguos de lado $a = 65\ [\text{cm}]$) est\'a en
una regi\'on de campo magn\'etico $B(t) = 0{,}001\,t$ (entrando a la p\'agina). La resistencia por
unidad de longitud del cable es $0{,}1\ [\Omega/\text{m}]$. Determine la corriente a trav\'es de
la secci\'on $PQ$.

\medskip
\begin{center}
\begin{tikzpicture}[scale=1.0, line width=0.9pt]
    % marco exterior (dos lazos)
    \draw (0,0) rectangle (4.5,1.6);
    \draw (3,0)--(3,1.6); % divisor PQ
    % campo entrante (cruces)
    \foreach \x in {0.4,0.9,1.4,1.9,2.4} \foreach \y in {0.4,0.9,1.3}
        \draw[blue,line width=0.5pt] (\x-0.07,\y-0.07)--(\x+0.07,\y+0.07) (\x-0.07,\y+0.07)--(\x+0.07,\y-0.07);
    \foreach \x in {3.4,3.9} \foreach \y in {0.4,0.9,1.3}
        \draw[blue,line width=0.5pt] (\x-0.07,\y-0.07)--(\x+0.07,\y+0.07) (\x-0.07,\y+0.07)--(\x+0.07,\y-0.07);
    % etiquetas
    \node[above] at (3,1.6){$P$}; \node[below] at (3,0){$Q$};
    \node[left,font=\small] at (0,0.8){$a$};
    \node[below,font=\small] at (1.5,0){$2a$}; \node[below,font=\small] at (3.75,0){$a$};
    \node at (1.5,0.85){\textcolor{blue!70!black}{$\varepsilon_1$}};
    \node at (3.75,0.85){\textcolor{blue!70!black}{$\varepsilon_2$}};
\end{tikzpicture}
\end{center}
\medskip

\noindent\textbf{\'Areas y fem inducidas} ($\varepsilon = -A\,\dfrac{dB}{dt}$, con $\dfrac{dB}{dt}=0{,}001$):
\[
    A_1 = 2a^2 \;\Rightarrow\; |\varepsilon_1| = 2a^2(0{,}001) = 0{,}845\ [\text{mV}]
\]
\[
    A_2 = a^2 \;\Rightarrow\; |\varepsilon_2| = a^2(0{,}001) = 0{,}423\ [\text{mV}]
\]

\noindent\textbf{Resistencias} (longitudes de cable $5a$, $3a$ y $a$; $\,R = 0{,}1\cdot\ell$):
\[
    R_1 = 0{,}1(5a) = 0{,}325\ \Omega,\quad
    R_2 = 0{,}1(3a) = 0{,}195\ \Omega,\quad
    R_3 = 0{,}1(a) = 0{,}065\ \Omega
\]

\noindent\textbf{Ecuaciones de Kirchhoff:}
\[
    |\varepsilon_1| - I_1 R_1 - I_3 R_2 = 0,\qquad
    |\varepsilon_2| + I_3 R_2 - I_2 R_3 = 0,\qquad
    I_3 + I_2 = I_1
\]

\noindent Resolviendo:
\[
    I_1 = 3{,}11\ [\text{mA}],\qquad
    I_2 = 3{,}96\ [\text{mA}],\qquad
    I_3 = -0{,}85\ [\text{mA}]
\]
\[
    \boxed{I_{PQ} = |I_3| = 0{,}85\ [\text{mA}]}
\]
(el signo negativo indica que la corriente en $PQ$ circula en sentido opuesto al supuesto).
