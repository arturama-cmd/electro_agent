\begin{formbox}{Ley de Coulomb (forma vectorial)}
\[
    \vec{F}_{12} = k_e\,\frac{q_1 q_2}{|\vec{r}_1 - \vec{r}_2|^{3}}\,(\vec{r}_1 - \vec{r}_2)
\]
\textbf{Descripci\'on:} La fuerza que ejerce la carga $q_2$ sobre $q_1$ es dirigida a lo largo de la
l\'inea que une ambas cargas, con direcci\'on dependiente del signo de las cargas.
\end{formbox}

\begin{formbox}{Campo el\'ectrico debido a una distribuci\'on discreta}
\[
    \vec{E} = k_e\sum_i \frac{q_i}{|\vec{r} - \vec{r}_i|^{3}}\,(\vec{r} - \vec{r}_i)
\]
\textbf{Descripci\'on:} Superposici\'on de los campos el\'ectricos producidos por un conjunto discreto
de cargas puntuales $q_i$.
\end{formbox}

\begin{formbox}{Potencial el\'ectrico de una carga puntual}
\[
    V(\vec{r}) = k_e\,\frac{q}{|\vec{r} - \vec{r}_q|}
\]
\textbf{Descripci\'on:} El potencial en un punto $\vec{r}$ debido a una carga puntual $q$ ubicada en
$\vec{r}_q$.
\end{formbox}

\begin{formbox}{Formulario}
\begin{align*}
    R_{/\!/} &= \left(\frac{1}{R_1} + \frac{1}{R_2} + \dots \frac{1}{R_n}\right)^{-1}
        && \rightarrow \textbf{Resistencia paralelo}\\
    R_{serie} &= R_1 + R_2 + \dots R_n && \rightarrow \textbf{Resistencia Serie}\\
    \textstyle\sum V &= I\cdot R && \rightarrow \textbf{Ley de Ohm}\\
    \textstyle\sum I_{in} &= \textstyle\sum I_{out} && \rightarrow \textbf{Ley de nodos}\\
    \textstyle\sum V_{in} &= 0 && \rightarrow \textbf{Ley de Mallas}\\
    \vec{F} &= q(\vec{v}_q \times \vec{B}) && \rightarrow \textbf{Fuerza}\\
    d\vec{F} &= I_0\,d\vec{l} \times \vec{B} && \rightarrow \textbf{Fuerza en un cable}\\
    \vec{m} &= I A\,\hat{n} && \rightarrow \textbf{Momento Magn\'etico}\\
    \vec{\tau} &= \vec{m} \times \vec{B} && \rightarrow \textbf{Torque}
\end{align*}
\end{formbox}

    \item $\displaystyle\int \frac{dx}{\sqrt{a^2 + x^2}} = \ln\left|x + \sqrt{x^2 + a^2}\right| + C$
    \item $\displaystyle\int \frac{1}{r}\,dr = \ln(r) + C$
    \item $\displaystyle\int \frac{1}{r^2}\,dr = -\frac{1}{r} + C$
    \item $\displaystyle\int \frac{dx}{(x^2 + a^2)^{3/2}} = \frac{x}{a^2\sqrt{x^2 + a^2}}$
\end{itemize}
