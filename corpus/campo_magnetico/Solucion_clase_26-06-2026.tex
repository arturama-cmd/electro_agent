\section*{Fem inducida en un lazo dentro de un solenoide}

\noindent Un solenoide de $n = 400\ [\text{vueltas/m}]$ conduce una corriente
$I(t) = 30\,(1 - e^{-1{,}6t})\ [\text{A}]$. Coaxial y \emph{dentro} de \'el se encuentra un lazo de
radio $R = 6\ [\text{cm}]$ con $N = 250$ vueltas de alambre delgado. Determine la fem inducida en el
lazo y la direcci\'on de la corriente inducida.

\medskip

\noindent
\begin{minipage}[t]{0.44\textwidth}
\centering\vspace{0pt}
\begin{tikzpicture}[scale=1.0, >=Stealth, line width=0.9pt]
    % solenoide (bobinas)
    \foreach \x in {-2.0,-1.6,...,2.0}{
        \draw[cyan!55!black,line width=1pt] (\x,-0.9) .. controls (\x+0.2,1.0) and (\x+0.2,1.0) .. (\x,0.9);
        \draw[cyan!55!black,line width=1pt] (\x,0.9) .. controls (\x-0.2,-1.0) and (\x-0.2,-1.0) .. (\x,-0.9);}
    \draw[dashed,gray] (-2.4,0.9)--(2.4,0.9);
    \draw[dashed,gray] (-2.4,-0.9)--(2.4,-0.9);
    % lazo interior (elipse inclinada)
    \draw[orange!85!black,line width=1.6pt] (0,0) ellipse (0.42 and 0.75);
    \draw[<->,gray] (0,0)--(0,0.75); \node[right,font=\footnotesize,gray] at (0.05,0.42){$R$};
    \node[font=\footnotesize] at (0,-1.25){$N$ vueltas};
    % corriente del solenoide
    \draw[->,cyan!55!black,line width=1pt] (1.6,1.05)--(2.05,1.05); \node[above,font=\footnotesize] at (1.85,1.05){$I$};
    \node[above,font=\footnotesize] at (-1.3,1.0){$n$ vueltas/m};
\end{tikzpicture}\\[2pt]
{\footnotesize Lazo coaxial dentro del solenoide}
\end{minipage}%
\hfill
\begin{minipage}[t]{0.53\textwidth}
\vspace{0pt}
\textbf{Datos:}
\[
    n = 400\ \tfrac{\text{vueltas}}{\text{m}},\quad N = 250,\quad R = 0{,}06\ [\text{m}]
\]
\[
    I(t) = 30\,(1 - e^{-1{,}6t})\ [\text{A}]
\]

\smallskip
\textbf{Campo del solenoide} (uniforme en su interior):
\[
    B = \mu_0\,n\,I(t)
\]
\textbf{Flujo} a trav\'es del lazo. Por definici\'on,
$\Phi = \int_S \vec{B}\cdot d\vec{S}$; como $\vec{B}$ es uniforme y perpendicular al lazo
(\'area $\pi R^2$):
\[
    \Phi = \int_S \vec{B}\cdot d\vec{S} = B\,\pi R^2 = \mu_0\,n\,\pi R^2\,I(t)
\]
\end{minipage}

\medskip

\noindent\textbf{Derivada de la corriente:}
\[
    \frac{dI}{dt} = 30\,(1{,}6)\,e^{-1{,}6t} = 48\,e^{-1{,}6t}\ [\text{A/s}]
\]

\noindent\textbf{Fem inducida} (ley de Faraday, $N$ vueltas):
\begin{align*}
    \eps = -N\frac{d\Phi}{dt}
       &= -N\,\mu_0\,n\,\pi R^2\,\frac{dI}{dt} \\
       &= -(250)(4\pi\times10^{-7})(400)\,\pi(0{,}06)^2\,(48)\,e^{-1{,}6t}
\end{align*}
\[
    \boxed{\eps(t) = -68{,}2\,e^{-1{,}6t}\ [\text{mV}]}
\]

\medskip
\noindent\textbf{Direcci\'on de la corriente inducida:} la corriente del solenoide
$I(t)=30(1-e^{-1{,}6t})$ \emph{crece} en el tiempo ($dI/dt>0$), por lo que el campo $B$ y el flujo
$\Phi$ a trav\'es del lazo \textbf{aumentan}. Por la ley de Lenz, la corriente inducida se
\emph{opone} a ese aumento generando un campo opuesto a $B$; por lo tanto \textbf{circula en sentido
contrario a la corriente del solenoide}.

\medskip
\noindent\small\textit{Observaci\'on:} la fem es m\'axima en $t=0$ ($\eps=-68{,}2$ mV) y decae
exponencialmente; cuando $t\to\infty$ la corriente se estabiliza en $30$ A, $dI/dt\to 0$ y la fem
inducida se anula.\normalsize
