\section*{Pregunta 2: Circuito con resistencias y bater\'ias}

\[
    \text{Derecha: } 2\parallel3\parallel6 = \left(\tfrac12+\tfrac13+\tfrac16\right)^{-1} = 1\ \Omega,
    \qquad
    \text{Izquierda: } 18\parallel9\parallel3 = \left(\tfrac1{18}+\tfrac19+\tfrac13\right)^{-1} = 2\ \Omega.
\]
En la rama izquierda, $4\,\Omega$ y $2\,\Omega$ quedan en serie ($6\,\Omega$). El circuito reducido tiene
tres ramas entre los nodos $A$ (superior) y $B$ (inferior):

\noindent\textbf{(b) Leyes de Kirchhoff.} Con $I_1,I_2$ entrando a $A$ e $I_3$ saliendo:
\begin{align*}
    \text{Nodo } A:&\quad I_1 + I_2 = I_3 \\[2pt]
    \text{Malla } M_1:&\quad 12 + 6 - 4I_1 - 2I_1 + 6I_2 = 0 \;\Longrightarrow\; 6I_1 - 6I_2 = 18
        \;\Longrightarrow\; I_1 - I_2 = 3 \\[2pt]
    \text{Malla } M_2:&\quad 3 - 14 + I_3 + 6I_2 = 0 \;\Longrightarrow\; 6I_2 + I_3 = 11
\end{align*}
(el t\'ermino $6I_2$ aparece porque la rama central $6\,\Omega$ es compartida por ambas mallas).

\medskip
\noindent\textbf{(c) Soluci\'on del sistema.} De $M_1$: $I_1 = I_2+3$. En el nodo:
$I_3 = I_1+I_2 = 2I_2+3$. Sustituyendo en $M_2$:
\[
    6I_2 + (2I_2+3) = 11 \;\Longrightarrow\; 8I_2 = 8 \;\Longrightarrow\; I_2 = 1.
\]
\[
    \boxed{I_1 = 4\ [\text{A}],\qquad I_2 = 1\ [\text{A}],\qquad I_3 = 5\ [\text{A}]}
\]
Verificaci\'on de nodo: $I_1+I_2 = 4+1 = 5 = I_3$. \checkmark

\section*{Pregunta 3: Fuerza magn\'etica}

\noindent\textbf{(a) Inducci\'on $\vec B$ sobre el prot\'on.} Con $\vec v = 2\times10^{6}\,\hat\jmath$
[m/s], $\vec F = 3{,}2\times10^{-13}\,\hat\imath$ [N] y $q=1{,}6\times10^{-19}$ [C]. Sea
$\vec B = B_x\hat\imath + B_y\hat\jmath + B_z\hat k$:
\[
    \vec v\times\vec B = 2\times10^{6}\,\hat\jmath\times(B_x\hat\imath+B_y\hat\jmath+B_z\hat k)
    = 2\times10^{6}\big(B_z\,\hat\imath - B_x\,\hat k\big)
\]
(usando $\hat\jmath\times\hat\imath=-\hat k$, $\hat\jmath\times\hat k=\hat\imath$). Entonces
\[
    \vec F = q\,\vec v\times\vec B = (1{,}6\times10^{-19})(2\times10^{6})\big(B_z\hat\imath - B_x\hat k\big)
    = 3{,}2\times10^{-13}\big(B_z\hat\imath - B_x\hat k\big).
\]
Igualando con $\vec F = 3{,}2\times10^{-13}\,\hat\imath$: $B_z = 1$, $B_x = 0$. La componente $B_y$
(paralela a $\vec v$) no produce fuerza, se toma nula:
\[
    \boxed{\vec B = 1{,}0\,\hat k\ [\text{T}]}
\]

\vspace{0.3cm}
\noindent\textbf{(b) Espira cuadrada} ($a=1$ m, $I_0=0{,}5$ A) en
$\vec B = B_0\left(\hat\jmath+\tfrac12\hat k\right)$, $B_0=0{,}05$ T, es decir
$\vec B = 0{,}05\,\hat\jmath + 0{,}025\,\hat k$ [T].

\textbf{(b.1) Fuerza en el tramo 1.} El tramo 1 est\'a sobre el eje $Y$ con corriente en $+\hat\jmath$,
$\vec L_1 = a\,\hat\jmath = 1\,\hat\jmath$:
\[
    \vec F_1 = I_0\,\vec L_1\times\vec B = 0{,}5\,(1\,\hat\jmath)\times(0{,}05\,\hat\jmath+0{,}025\,\hat k)
\]
\[
    = 0{,}5\big(0{,}025\,(\hat\jmath\times\hat k)\big) = 0{,}5\,(0{,}025)\,\hat\imath
\]
\[
    \boxed{\vec F_1 = 1{,}25\times10^{-2}\,\hat\imath\ [\text{N}]}
\]
\end{minipage}

\medskip
\noindent\textbf{(b.2) Momento magn\'etico.} $\vec m = I_0\,A\,\hat n$, con $A=a^2=1$ m$^2$. Por la regla
de la mano derecha, el sentido de circulaci\'on (horario visto desde $+z$) da $\hat n = -\hat k$:
\[
    \boxed{\vec m = I_0\,a^2\,(-\hat k) = -0{,}5\,\hat k\ [\text{A m}^2]}
\]

\medskip
\noindent\textbf{(b.3) Torque sobre la espira.} $\vec\tau = \vec m\times\vec B$:
\[
    \vec\tau = (-0{,}5\,\hat k)\times(0{,}05\,\hat\jmath + 0{,}025\,\hat k)
    = -0{,}5\,(0{,}05)\,(\hat k\times\hat\jmath)
    = -0{,}025\,(-\hat\imath)
\]
\[
    \boxed{\vec\tau = 2{,}5\times10^{-2}\,\hat\imath\ [\text{N m}]}
\]
