% REPASO: PRODUCTO CRUZ
%========================================================
\section*{Repaso: el producto cruz}

\noindent El \textbf{producto cruz} (o producto vectorial) de dos vectores $\vec{A}$ y $\vec{B}$
es un \textbf{vector} $\vec{A}\times\vec{B}$ con las siguientes propiedades:

\begin{itemize}\setlength\itemsep{3pt}
    \item \textbf{M\'odulo:} su magnitud es el producto de las magnitudes (m\'odulos) de ambos
          vectores por el seno del \'angulo $\theta$ que forman:
          \[ |\vec{A}\times\vec{B}| = |\vec{A}|\,|\vec{B}|\operatorname{sen}\theta \]
          donde $|\vec{A}|$ y $|\vec{B}|$ son las magnitudes de cada vector y
          $0 \le \theta \le \pi$ es el \'angulo entre ellos.
    \item \textbf{Direcci\'on:} es \textbf{perpendicular} al plano que contiene a $\vec{A}$ y
          $\vec{B}$, y se determina con la \textbf{regla de la mano derecha}.
    \item \textbf{Casos l\'imite:}
          \begin{itemize}
              \item Si $\vec{A}\parallel\vec{B}$ ($\theta = 0$ o $\pi$): $\operatorname{sen}\theta = 0$,
                    el producto cruz es \textbf{nulo}.
              \item Si $\vec{A}\perp\vec{B}$ ($\theta = \pi/2$): $\operatorname{sen}\theta = 1$,
                    el m\'odulo es \textbf{m\'aximo}: $|\vec{A}\times\vec{B}| = |\vec{A}||\vec{B}|$.
          \end{itemize}
\end{itemize}

\noindent En la ley de Biot--Savart estas dos ideas son clave: el \textbf{m\'odulo}
$|\vec{A}\times\vec{B}| = |\vec{A}||\vec{B}|\operatorname{sen}\theta$ entrega la \emph{magnitud} del
campo, y la \textbf{regla de la mano derecha} entrega su \emph{direcci\'on}.

\section*{Enunciado}

\noindent El cable de la figura conduce una corriente $I = 5\ [\text{A}]$ y forma un arco
circular de radio $R = 4\ [\text{cm}]$. Determine el campo magn\'etico en el origen.

\vspace{0pt}
\textbf{Vectores de posici\'on:}
\begin{itemize}\setlength\itemsep{2pt}
    \item $\vec{r}_{P}$: punto donde se \textbf{calcula} el campo magn\'etico (el origen $O$):
          \[ \vec{r}_{P} = \vec{0} \]
    \item $\vec{r}_{I}$: punto de la \textbf{fuente} (elemento de corriente sobre el arco):
          \[ \vec{r}_{I} = R\cos\theta\,\hat{\imath} + R\operatorname{sen}\theta\,\hat{\jmath} \]
    \item $\vec{r}_{P} - \vec{r}_{I} = -R\cos\theta\,\hat{\imath} - R\operatorname{sen}\theta\,\hat{\jmath}$
\end{itemize}
\end{minipage}

\medskip

\noindent\textbf{M\'odulo de $\vec{r}_{P}-\vec{r}_{I}$.} Usando la identidad trigonom\'etrica
$\cos^2\theta + \operatorname{sen}^2\theta = 1$, los t\'erminos de seno y coseno se cancelan:
\begin{align*}
    |\vec{r}_{P}-\vec{r}_{I}|
       &= \sqrt{(-R\cos\theta)^2 + (-R\operatorname{sen}\theta)^2}
        = \sqrt{R^2\cos^2\theta + R^2\operatorname{sen}^2\theta} \\
       &= \sqrt{R^2\big(\cos^2\theta + \operatorname{sen}^2\theta\big)}
        = \sqrt{R^2\,(1)} = R
\end{align*}
La distancia de cada elemento del arco al origen es \textbf{constante} e igual a $R$.

\medskip

\noindent\textbf{Ley de Biot--Savart:}
\[
    d\vec{B} = \frac{\mu_0 I}{4\pi}\,\frac{d\vec{\ell}\times(\vec{r}_{P}-\vec{r}_{I})}{|\vec{r}_{P}-\vec{r}_{I}|^3}
\]

%--------------------------------------------------------
\subsection*{Contribuci\'on de los tramos rectos: nula}
En cada tramo recto, el elemento $d\vec{\ell}$ apunta \textbf{a lo largo del mismo eje} que une la
fuente con el origen, es decir $d\vec{\ell} \parallel (\vec{r}_{P}-\vec{r}_{I})$. Por lo tanto:
\[
    d\vec{\ell}\times(\vec{r}_{P}-\vec{r}_{I}) = \vec{0}
    \quad\Longrightarrow\quad
    \vec{B}_{\text{rectos}} = \vec{0}
\]
Solo el \textbf{arco} contribuye al campo en el origen.

%--------------------------------------------------------
\subsection*{Contribuci\'on del arco (m\'odulo del producto cruz)}
Sobre el arco, $d\vec{\ell}$ (tangente) es \textbf{perpendicular} a $(\vec{r}_{P}-\vec{r}_{I})$
(radial), de modo que $\operatorname{sen}90^\circ = 1$. Tomando el m\'odulo, con
$|\vec{r}_{P}-\vec{r}_{I}| = R$ y $d\ell = R\,d\theta$:
\[
    dB = \frac{\mu_0 I}{4\pi}\,\frac{d\ell\,|\vec{r}_{P}-\vec{r}_{I}|\,\operatorname{sen}90^\circ}{|\vec{r}_{P}-\vec{r}_{I}|^3}
       = \frac{\mu_0 I}{4\pi}\,\frac{(R\,d\theta)(R)(1)}{R^3}
       = \frac{\mu_0 I}{4\pi R}\,d\theta
\]
Integrando sobre el cuarto de circunferencia ($\theta$ de $0$ a $\pi/2$):
\[
    B = \frac{\mu_0 I}{4\pi R}\int_{0}^{\pi/2} d\theta
      = \frac{\mu_0 I}{4\pi R}\cdot\frac{\pi}{2}
      = \frac{\mu_0 I}{8R}
\]
\[
    B = \frac{(4\pi\times10^{-7})(5)}{8\,(0{,}04)} = 1{,}96\times10^{-5}\ [\text{T}]
\]

\noindent\textbf{Direcci\'on (regla de la mano derecha):} la corriente recorre el arco en sentido
horario (de $+\hat{\jmath}$ hacia $+\hat{\imath}$); el pulgar apunta \textbf{hacia adentro de la
p\'agina} ($-\hat{k}$). Por lo tanto:
\[
    \boxed{\vec{B} = 1{,}96\times10^{-5}\,(-\hat{k})\ [\text{T}]\quad\text{(entrando a la p\'agina)}}
\]
