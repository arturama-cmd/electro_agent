\section*{Problema 2: Fuerza entre tres alambres paralelos}

\noindent Tres alambres rectos, paralelos y muy largos transportan cada uno una corriente
$I = 4\ [\text{A}]$ y est\'an separados una distancia $d = 15\ [\text{cm}]$ entre alambres vecinos,
como muestra la Fig.\ 2. Los alambres extremos (1) y (3) conducen la corriente en el mismo sentido,
y el alambre central (2) en sentido opuesto. Calcule la \textbf{fuerza magn\'etica por unidad de
longitud} sobre cada alambre.

\medskip

\textbf{Datos:}
\[
    I = 4\ [\text{A}],\qquad d = 0{,}15\ [\text{m}]
\]

\smallskip
\textbf{Fuerza por unidad de longitud} entre dos alambres paralelos separados $r$:
\[
    \frac{F}{l} = \frac{\mu_0\,I_a\,I_b}{2\pi\,r}
\]
\noindent\emph{Atractiva} si las corrientes son paralelas (mismo sentido); \emph{repulsiva} si son
antiparalelas (sentidos opuestos).
\end{minipage}

\medskip

\noindent\textbf{Las dos magnitudes que intervienen} (con $I_a=I_b=I$):

\noindent Entre alambres \emph{vecinos} (separaci\'on $d$):
\[
    \frac{F_d}{l} = \frac{\mu_0 I^2}{2\pi d}
    = \frac{(4\pi\times10^{-7})(4)^2}{2\pi(0{,}15)}
    = 2{,}13\times10^{-5}\ [\text{N/m}]
\]
Entre los alambres \emph{extremos} (1) y (3) (separaci\'on $2d$):
\[
    \frac{F_{2d}}{l} = \frac{\mu_0 I^2}{2\pi (2d)} = \frac{1}{2}\,\frac{F_d}{l}
    = 1{,}07\times10^{-5}\ [\text{N/m}]
\]

%--------------------------------------------------
\subsection*{Alambre central (2)}

Recibe una fuerza de cada vecino, ambos a distancia $d$ y \emph{antiparalelos} a \'el, por lo que
lo \textbf{repelen}: el alambre (1) lo empuja hacia abajo y el alambre (3) lo empuja hacia arriba,
con \emph{igual magnitud}. Se cancelan:
\[
    \boxed{\frac{F_2}{l} = 0}
\]

%--------------------------------------------------
\subsection*{Alambre superior (1)}

Sobre \'el act\'uan:
\begin{itemize}\setlength\itemsep{1pt}
    \item el alambre (2), antiparalelo y a distancia $d$ $\Rightarrow$ \textbf{repulsi\'on hacia
          arriba}, $\dfrac{F_d}{l}=2{,}13\times10^{-5}$ [N/m];
    \item el alambre (3), paralelo y a distancia $2d$ $\Rightarrow$ \textbf{atracci\'on hacia abajo},
          $\dfrac{F_{2d}}{l}=1{,}07\times10^{-5}$ [N/m].
\end{itemize}
La fuerza neta (hacia arriba) es la diferencia:
\[
    \frac{F_1}{l} = \frac{F_d}{l} - \frac{F_{2d}}{l}
    = (2{,}13 - 1{,}07)\times10^{-5}
\]
\[
    \boxed{\frac{F_1}{l} = 1{,}07\times10^{-5}\ [\text{N/m}]\quad\text{(hacia arriba)}}
\]

%--------------------------------------------------
\subsection*{Alambre inferior (3)}

Por simetr\'ia con el alambre (1): el alambre (2) lo \textbf{repele hacia abajo}
($2{,}13\times10^{-5}$ [N/m]) y el alambre (1) lo \textbf{atrae hacia arriba}
($1{,}07\times10^{-5}$ [N/m]). La fuerza neta (hacia abajo) es:
\[
    \boxed{\frac{F_3}{l} = 1{,}07\times10^{-5}\ [\text{N/m}]\quad\text{(hacia abajo)}}
\]

%--------------------------------------------------
\subsection*{Resumen}

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|c|c|l|}
\hline
\textbf{Alambre} & \textbf{$F/l$ [N/m]} & \textbf{Direcci\'on} \\
\hline
(1) superior & $1{,}07\times10^{-5}$ & hacia arriba (se aleja del conjunto) \\
(2) central  & $0$                   & se cancelan las repulsiones \\
(3) inferior & $1{,}07\times10^{-5}$ & hacia abajo (se aleja del conjunto) \\
\hline
\end{tabular}
\end{center}

\medskip
\noindent\textbf{Interpretaci\'on:} el alambre central, repelido sim\'etricamente por sus dos
vecinos, queda en equilibrio; los alambres extremos son expulsados del arreglo (la repulsi\'on del
vecino cercano supera la atracci\'on del extremo lejano).
