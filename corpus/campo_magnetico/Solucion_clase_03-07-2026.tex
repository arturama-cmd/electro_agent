\section*{Problema 1: Campo magn\'etico de dos arcos (Biot--Savart)}

\noindent Dos filamentos circulares centrados en el origen transportan una corriente constante $I$.
El filamento~1 es un arco de radio $R_1$ en el plano $xy$, desde $\theta=\pi/8$ hasta $\theta=3\pi/8$,
con corriente \textbf{antihoraria}. El filamento~2 es un arco de radio $R_2$ en el plano $yz$, que
abarca un \'angulo total de $\pi/2$, con corriente \textbf{horaria}. Determine el campo magn\'etico
en el origen debido a cada filamento y el total (desprecie los segmentos rectos).

\medskip

\subsection*{Deducci\'on del m\'odulo del diferencial de campo $dB$}

Partimos de la ley de Biot--Savart en forma vectorial:
\[
    d\vec{B} = \frac{\mu_0 I}{4\pi}\,
    \frac{d\vec{\ell}\times(\vec{r}-\vec{r}\,')}{\|\vec{r}-\vec{r}\,'\|^{3}}
\]

\noindent\textbf{Paso 1 (m\'odulo).} La magnitud del producto cruz es
$\|d\vec{\ell}\times(\vec{r}-\vec{r}\,')\| = d\ell\,\|\vec{r}-\vec{r}\,'\|\sin\phi$, donde $\phi$ es el
\'angulo entre $d\vec{\ell}$ y $(\vec{r}-\vec{r}\,')$. Luego:
\[
    dB = \frac{\mu_0 I}{4\pi}\,
    \frac{d\ell\,\|\vec{r}-\vec{r}\,'\|\,\sin\phi}{\|\vec{r}-\vec{r}\,'\|^{3}}
    = \frac{\mu_0 I}{4\pi}\,\frac{d\ell\,\sin\phi}{\|\vec{r}-\vec{r}\,'\|^{2}}
\]

\noindent\textbf{Paso 2 (geometr\'ia).} El campo se eval\'ua en el origen, $\vec{r}=\vec{0}$, y la
fuente est\'a sobre el arco, $\vec{r}\,' = R\cos\theta\,\hat{\imath}+R\sin\theta\,\hat{\jmath}$.
Entonces $\vec{r}-\vec{r}\,'=-\vec{r}\,'$ y su norma es el radio:
\[
    \|\vec{r}-\vec{r}\,'\| = \sqrt{(R\cos\theta)^2+(R\sin\theta)^2} = R
\]

\noindent\textbf{Paso 3 (\'angulo $\phi$).} El elemento $d\vec{\ell}$ es \emph{tangente} al arco y
$(\vec{r}-\vec{r}\,')$ es \emph{radial} (apunta al centro). Tangente y radio son perpendiculares:
\[
    \phi = 90^\circ \quad\Rightarrow\quad \sin\phi = 1
\]

\noindent\textbf{Paso 4 (elemento de arco).} La longitud de arco subtendida por $d\theta$ es
$d\ell = R\,d\theta$.

\noindent\textbf{Paso 5 (sustituci\'on).} Reemplazando los pasos 2--4 en el m\'odulo:
\[
    dB = \frac{\mu_0 I}{4\pi}\,\frac{(R\,d\theta)(1)}{R^{2}}
    \quad\Rightarrow\quad
    \boxed{dB = \frac{\mu_0 I}{4\pi R}\,d\theta}
\]
Este resultado vale para \emph{cualquier} arco centrado en el punto de observaci\'on; s\'olo cambian
el radio $R$ y los l\'imites de integraci\'on.

\subsection*{(a) Filamento de radio $R_1$}

El arco va de $\theta=\pi/8$ a $3\pi/8$ (abarca $\Delta\theta=\pi/4$):
\[
    B_1 = \frac{\mu_0 I}{4\pi R_1}\int_{\pi/8}^{3\pi/8}\! d\theta
        = \frac{\mu_0 I}{4\pi R_1}\cdot\frac{\pi}{4} = \frac{\mu_0 I}{16 R_1}
\]
Por la regla de la mano derecha (corriente antihoraria en el plano $xy$), el campo \textbf{sale del
plano $xy$}, en $+\hat{k}$:
\[
    \boxed{\vec{B}_1 = \frac{\mu_0 I}{16 R_1}\,\hat{k}}
\]

\subsection*{(b) Filamento de radio $R_2$}

El arco en el plano $yz$ abarca $\Delta\theta=\pi/2$:
\[
    B_2 = \frac{\mu_0 I}{4\pi R_2}\int_{0}^{\pi/2}\! d\theta
        = \frac{\mu_0 I}{4\pi R_2}\cdot\frac{\pi}{2} = \frac{\mu_0 I}{8 R_2}
\]
Por la regla de la mano derecha (corriente horaria en el plano $yz$), el campo apunta en
$-\hat{\imath}$:
\[
    \boxed{\vec{B}_2 = \frac{\mu_0 I}{8 R_2}\,(-\hat{\imath})}
\]

\subsection*{(c) Campo total en el origen}
\[
    \boxed{\vec{B} = \vec{B}_1 + \vec{B}_2
        = \mu_0 I\left(\frac{1}{16 R_1}\,\hat{k} - \frac{1}{8 R_2}\,\hat{\imath}\right)}
\]

\section*{Problema 2: Campo y fuerza entre tres alambres (Amp\`ere y $\vec{F}=I\vec{L}\times\vec{B}$)}

\noindent Tres alambres rectos, muy largos y paralelos conducen corrientes uniformes. Las corrientes
(1) y (2) \textbf{salen} del plano; la (3) \textbf{entra}. Las corrientes (2) y (3) est\'an a una
distancia $d$ de la (1), con $\theta=60^\circ$ y $\alpha=30^\circ$. Determine: (a) el campo de (2) y
(3) por Amp\`ere; (b) la fuerza sobre una secci\'on $L$ de la corriente (1); (c) la fuerza neta si
$I_1=1$ [A], $I_2=I_3=2$ [A], $d=2L=20$ [cm].

\textbf{(a) Campo por la ley de Amp\`ere.} Para un alambre recto infinito, con un lazo amperiano
circular de radio $r$ centrado en el alambre, $\vec{B}$ es tangente y de m\'odulo constante:
\[
    \oint \vec{B}\cdot d\vec{\ell} = B\,(2\pi r) = \mu_0 I
    \;\Rightarrow\; B = \frac{\mu_0 I}{2\pi r}
\]
\end{minipage}

\medskip

\noindent\textbf{Signo del vector unitario angular $\hat{\varphi}$ (regla de la mano derecha).}
El sentido de $\vec{B}$ lo fija la mano derecha: se apunta el \textbf{pulgar} en el sentido de la
corriente y los \textbf{dedos} se curvan indicando el sentido de circulaci\'on de $\vec{B}$.
\begin{itemize}\setlength\itemsep{1pt}
    \item Corriente (2), que \textbf{sale} del plano ($+\hat{k}$): $\vec{B}$ circula
          \textbf{antihorario} $\Rightarrow +\hat{\varphi}$.
    \item Corriente (3), que \textbf{entra} al plano ($-\hat{k}$): $\vec{B}$ circula
          \textbf{horario} $\Rightarrow -\hat{\varphi}$.
\end{itemize}
\[
    \vec{B}_2 = \frac{\mu_0 I_2}{2\pi r}\,\hat{\varphi},
    \qquad
    \vec{B}_3 = \frac{\mu_0 I_3}{2\pi r}\,(-\hat{\varphi})
\]

\subsection*{(b) Fuerza sobre una secci\'on $L$ de la corriente (1)}

\noindent\textbf{Geometr\'ia de los vectores unitarios.} En la posici\'on de la corriente (1), el
campo $\vec{B}_2$ es \emph{perpendicular} a la recta $I_1$--$I_2$ (direcci\'on radial). Como esa
recta forma un \'angulo $\theta=60^\circ$ con la horizontal, y $\vec{B}_2\perp$ a ella, el campo
$\vec{B}_2$ forma con el eje $x$ el \'angulo complementario $\alpha=90^\circ-\theta=30^\circ$:

Con $\alpha=30^\circ$ ($\cos\alpha=\tfrac{\sqrt3}{2}$, $\sin\alpha=\tfrac12$), el campo de (2) en la
posici\'on de (1) es:
\[
    \vec{B}_2 = \frac{\mu_0 I_2}{2\pi d}\big(\cos\alpha\,\hat{\imath}+\sin\alpha\,\hat{\jmath}\big)
              = \frac{\mu_0 I_2}{2\pi d}\Big(\tfrac{\sqrt{3}}{2}\,\hat{\imath}+\tfrac{1}{2}\,\hat{\jmath}\Big)
\]
La corriente (3) est\'a sobre el eje $x$ y entra al plano; su campo en (1) apunta en $+\hat{\jmath}$:
\[
    \vec{B}_3 = \frac{\mu_0 I_3}{2\pi d}\,\hat{\jmath}
\]

\medskip

\noindent Con $\vec{L}=L\,\hat{k}$ (la corriente (1) sale del plano) y $\vec{F}=I_1\vec{L}\times\vec{B}$,
usando $\hat{k}\times\hat{\imath}=\hat{\jmath}$ y $\hat{k}\times\hat{\jmath}=-\hat{\imath}$:
\begin{align*}
    \vec{F}_{12} &= I_1 L\,\hat{k}\times\frac{\mu_0 I_2}{2\pi d}
        \Big(\tfrac{\sqrt{3}}{2}\hat{\imath}+\tfrac12\hat{\jmath}\Big)
        = \frac{\mu_0 I_1 I_2 L}{2\pi d}\Big(\tfrac{\sqrt{3}}{2}\,\hat{\jmath}-\tfrac12\,\hat{\imath}\Big)\\
    \vec{F}_{13} &= I_1 L\,\hat{k}\times\frac{\mu_0 I_3}{2\pi d}\,\hat{\jmath}
        = \frac{\mu_0 I_1 I_3 L}{2\pi d}\,(-\hat{\imath})
\end{align*}

\subsection*{(c) Fuerza neta sobre la corriente (1)}

Con $I_1=1$, $I_2=I_3=2$ [A], $L=0{,}10$ [m], $d=0{,}20$ [m], el coeficiente
$\dfrac{\mu_0 I_1 I_2 L}{2\pi d}=\dfrac{\mu_0 I_1 I_3 L}{2\pi d}=\dfrac{\mu_0}{2\pi}$. Por
superposici\'on:
\begin{align*}
    \vec{F}_1 = \vec{F}_{12}+\vec{F}_{13}
      &= \frac{\mu_0}{2\pi}\Big(\tfrac{\sqrt{3}}{2}\,\hat{\jmath}-\tfrac12\,\hat{\imath}\Big)
         + \frac{\mu_0}{2\pi}\,(-\hat{\imath})
       = \frac{\mu_0}{2\pi}\Big(-\tfrac{3}{2}\,\hat{\imath}+\tfrac{\sqrt{3}}{2}\,\hat{\jmath}\Big)
\end{align*}
\[
    \boxed{\vec{F}_1 = \frac{\mu_0}{4\pi}\left(-3\,\hat{\imath}+\sqrt{3}\,\hat{\jmath}\right)\ [\text{N}]}
\]
