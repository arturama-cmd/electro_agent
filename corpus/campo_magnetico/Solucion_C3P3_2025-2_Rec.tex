%========================================
% PROBLEMA 1: ESPIRA CIRCULAR (momento dipolar y torque)
%========================================
\section*{Problema 1: Momento dipolar magn\'etico y torque sobre una espira circular}

\noindent Una espira \textbf{circular} de radio $r = 0{,}1\ [\text{m}]$, por la que circula una
corriente $I = 5\ [\text{A}]$, est\'a orientada de modo que su vector unitario normal es
$\vec{u} = \hat{\imath} + \hat{\jmath}$ (perpendicular al plano de la espira). La espira se
encuentra inmersa en un campo magn\'etico uniforme $\vec{B} = 5\,(\hat{\jmath} - \hat{k})\ [\text{T}]$.
Determine:

\medskip
\noindent\textbf{Datos:}
\begin{itemize}
    \item $I = 5\ [\text{A}]$,\quad $r = 0{,}1\ [\text{m}]$
    \item $\vec{u} = \hat{\imath} + \hat{\jmath}$ \quad (direcci\'on normal, \emph{no unitaria})
    \item $\vec{B} = 5\,\hat{\jmath} - 5\,\hat{k}\ [\text{T}]$
\end{itemize}

\bigskip

%--- (a) momento dipolar ---
\subsection*{(a) Momento dipolar magn\'etico $\vec{\mu}$}

El \'area de la espira circular es:
\[
    A = \pi r^2 = \pi (0{,}1)^2 = 0{,}01\pi\ [\text{m}^2]
\]

Como el vector dado $\vec{u}$ \textbf{no es unitario}, primero se normaliza
($|\vec{u}| = \sqrt{1^2+1^2} = \sqrt{2}$):
\[
    \hat{n} = \frac{\vec{u}}{|\vec{u}|} = \frac{\hat{\imath}+\hat{\jmath}}{\sqrt{2}}
\]

El vector de \'area es $\vec{A} = A\,\hat{n}$, y el momento dipolar magn\'etico:
\begin{align*}
    \vec{\mu} = I\,\vec{A} = I\,A\,\hat{n}
             &= (5)\,(0{,}01\pi)\,\frac{\hat{\imath}+\hat{\jmath}}{\sqrt{2}}
\end{align*}

\[
    \boxed{\vec{\mu} = \frac{0{,}05\pi}{\sqrt{2}}\,(\hat{\imath}+\hat{\jmath})\ [\text{A}\cdot\text{m}^2]}
\]

\bigskip

%--- (b) torque ---
\subsection*{(b) Momento de torsi\'on (torque) sobre la espira $\vec{\tau}$}

El torque sobre el dipolo magn\'etico en el campo uniforme es:
\[
    \vec{\tau} = \vec{\mu} \times \vec{B}
    = \begin{vmatrix}
        \hat{\imath} & \hat{\jmath} & \hat{k} \\[4pt]
        \dfrac{0{,}05\pi}{\sqrt{2}} & \dfrac{0{,}05\pi}{\sqrt{2}} & 0 \\[6pt]
        0 & 5 & -5
      \end{vmatrix}
\]

Desarrollando el determinante por la primera fila:
\begin{align*}
    \vec{\tau} &= \hat{\imath}\left[\frac{0{,}05\pi}{\sqrt{2}}(-5) - (0)(5)\right]
                - \hat{\jmath}\left[\frac{0{,}05\pi}{\sqrt{2}}(-5) - (0)(0)\right]
                + \hat{k}\left[\frac{0{,}05\pi}{\sqrt{2}}(5) - \frac{0{,}05\pi}{\sqrt{2}}(0)\right] \\[6pt]
               &= -\frac{0{,}25\pi}{\sqrt{2}}\,\hat{\imath}
                  + \frac{0{,}25\pi}{\sqrt{2}}\,\hat{\jmath}
                  + \frac{0{,}25\pi}{\sqrt{2}}\,\hat{k}
\end{align*}

\[
    \boxed{\vec{\tau} = \frac{0{,}25\pi}{\sqrt{2}}\,(-\hat{\imath}+\hat{\jmath}+\hat{k})\ [\text{N}\cdot\text{m}]}
\]

\section*{Problema 2: Fuerza magn\'etica sobre un conductor de geometr\'ia mixta}

\textbf{segmento curvo} de radio $R$ transporta una corriente $I = I_0$ en una regi\'on de
campos magn\'eticos uniformes de diferentes direcciones, seg\'un la figura. Determine la
fuerza magn\'etica sobre cada tramo y la fuerza total.

\hfill
\begin{minipage}[t]{0.50\textwidth}
    \vspace{0pt}
    \textbf{Ley de fuerza magn\'etica sobre una corriente:}
    \[
        \vec{F} = I\!\int \vec{d\ell}\times\vec{B}
    \]

    \smallskip
    \textbf{Productos vectoriales \'utiles:}
    \[
        \hat{\imath}\times\hat{k} = -\hat{\jmath},\qquad
        \hat{\jmath}\times\hat{k} = \hat{\imath}
    \]

    \smallskip
    \textbf{Tramo recto:} diagonal desde $(-L,R)$ hasta $(0,-R)$, inmerso
    en $\vec{B}=B_0\hat{k}$, con $\vec{d\ell}=dx\,\hat{\imath}+dy\,\hat{\jmath}$,
    donde $x:\,-L\to 0$ e $y:\,R\to -R$.

    \smallskip
    \textbf{Tramo curvo:} arco de radio $R$ recorrido en sentido
    antihorario, inmerso en $\vec{B}=-B_0\hat{k}$, parametrizado por
    \[
        \vec{d\ell}=R\sin\theta\,d\theta(-\hat{\imath})+R\cos\theta\,d\theta\,\hat{\jmath},
    \]
    con $\theta:\,-\tfrac{\pi}{2}\to \tfrac{\pi}{3}$.
\end{minipage}

\bigskip

%--- (a) segmento recto ---
\subsection*{(a) Fuerza sobre el segmento recto $\vec{F}_1$}

Con $\vec{d\ell}=dx\,\hat{\imath}+dy\,\hat{\jmath}$ y $\vec{B}=B_0\hat{k}$:
\begin{align*}
    \vec{d\ell}\times\vec{B}
      &= (dx\,\hat{\imath}+dy\,\hat{\jmath})\times(B_0\hat{k})
       = B_0\,dx\,(\hat{\imath}\times\hat{k}) + B_0\,dy\,(\hat{\jmath}\times\hat{k}) \\[4pt]
      &= -B_0\,dx\,\hat{\jmath} + B_0\,dy\,\hat{\imath}
\end{align*}

Integrando ($x:-L\to0$, $\;y:R\to-R$):
\begin{align*}
    \vec{F}_1 &= I_0 B_0\left[\hat{\imath}\int_{R}^{-R} dy
                 - \hat{\jmath}\int_{-L}^{0} dx\right] \\[4pt]
              &= I_0 B_0\left[\hat{\imath}\,(-R-R) - \hat{\jmath}\,(0-(-L))\right]
               = I_0 B_0\left[-2R\,\hat{\imath} - L\,\hat{\jmath}\right]
\end{align*}

\[
    \boxed{\vec{F}_1 = -\,I_0 B_0\,(2R\,\hat{\imath} + L\,\hat{\jmath})\ [\text{N}]}
\]

\bigskip

%--- (b) segmento curvo ---
\subsection*{(b) Fuerza sobre el segmento curvo $\vec{F}_2$}

Con $\vec{d\ell}=R\sin\theta\,d\theta(-\hat{\imath})+R\cos\theta\,d\theta\,\hat{\jmath}$
y $\vec{B}=-B_0\hat{k}$, el integrando es:
\begin{align*}
    \vec{d\ell}\times\vec{B}
      &= \bigl[R\sin\theta(-\hat{\imath})+R\cos\theta\,\hat{\jmath}\bigr]\,d\theta
         \times(-B_0\hat{k}) \\[4pt]
      &= -B_0 R\,\cos\theta\,d\theta\,\hat{\imath}
         - B_0 R\,\sin\theta\,d\theta\,\hat{\jmath}
\end{align*}

Integrando entre $\theta=-\tfrac{\pi}{2}$ y $\theta=\tfrac{\pi}{3}$:
\begin{align*}
    \vec{F}_2 &= -I_0 B_0 R\int_{-\pi/2}^{\pi/3}
                 \bigl(\cos\theta\,\hat{\imath}+\sin\theta\,\hat{\jmath}\bigr)d\theta \\[4pt]
              &= -I_0 B_0 R\Bigl[\,\hat{\imath}\,(\sin\theta)
                 -\hat{\jmath}\,(\cos\theta)\,\Bigr]_{-\pi/2}^{\pi/3} \\[4pt]
              &= -I_0 B_0 R\Bigl[\hat{\imath}\Bigl(\tfrac{\sqrt{3}}{2}+1\Bigr)
                 -\hat{\jmath}\Bigl(\tfrac{1}{2}-0\Bigr)\Bigr]
\end{align*}

\[
    \boxed{\vec{F}_2 = I_0 B_0 R\left[-\Bigl(1+\tfrac{\sqrt{3}}{2}\Bigr)\hat{\imath}
        + \tfrac{1}{2}\,\hat{\jmath}\right]\ [\text{N}]}
\]

\bigskip

%--- (c) fuerza total ---
\subsection*{(c) Fuerza magn\'etica total $\vec{F}$}

\[
    \vec{F} = \vec{F}_1 + \vec{F}_2
            = -I_0 B_0\,(2R\,\hat{\imath} + L\,\hat{\jmath})
              + I_0 B_0 R\left[-\Bigl(1+\tfrac{\sqrt{3}}{2}\Bigr)\hat{\imath}
              + \tfrac{1}{2}\,\hat{\jmath}\right]
\]

\[
    \boxed{\vec{F} = I_0 B_0\left[-\Bigl(3+\tfrac{\sqrt{3}}{2}\Bigr)R\,\hat{\imath}
        + \Bigl(\tfrac{1}{2}R - L\Bigr)\hat{\jmath}\right]\ [\text{N}]}
\]
