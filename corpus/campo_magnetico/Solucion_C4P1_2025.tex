\section*{Enunciado}

\noindent Dos filamentos conductores circulares, centrados en el origen, transportan una
corriente constante $I$. El \textbf{filamento 1} es un arco de radio $R_1$ en el plano $xy$,
desde $\theta = \pi/8$ hasta $\theta = 3\pi/8$ (sentido \textbf{antihorario}). El
\textbf{filamento 2} es un arco de radio $R_2$ en el plano $yz$, que abarca un \'angulo total de
$\pi/2$ (sentido \textbf{horario}). Usando la ley de Biot--Savart, determine el campo magn\'etico
en el origen debido a (a)~$R_1$, (b)~$R_2$ y (c)~ambos.

\textbf{Ley de Biot--Savart:}
\[
    d\vec{B} = \frac{\mu_0 I}{4\pi}\,\frac{d\vec{\ell}\times(\vec{r}-\vec{r}\,')}{|\vec{r}-\vec{r}\,'|^3}
\]
Se trabaja con el \textbf{m\'odulo} del producto cruz:
\[
    dB = \frac{\mu_0 I}{4\pi}\,\frac{d\ell\,|\vec{r}-\vec{r}\,'|\,\operatorname{sen}90^\circ}{|\vec{r}-\vec{r}\,'|^3}
\]
ya que $d\vec{\ell}$ (tangente al arco) es \textbf{perpendicular} a $(\vec{r}-\vec{r}\,')$ (radial),
de modo que $\operatorname{sen}90^\circ = 1$. La \textbf{direcci\'on} de $\vec{B}$ se asigna luego
con la \textbf{regla de la mano derecha}.
\end{minipage}

\vspace{0.3cm}

\noindent\textbf{Simplificaci\'on (campo en el origen).} El punto de observaci\'on es el origen,
$\vec{r}=\vec{0}$, y un elemento del arco est\'a en $\vec{r}\,' = R\cos\theta\,\hat{\imath} + R\operatorname{sen}\theta\,\hat{\jmath}$,
con $|\vec{r}-\vec{r}\,'| = R$ y $d\ell = R\,d\theta$. Por lo tanto:
\[
    dB = \frac{\mu_0 I}{4\pi}\,\frac{(R\,d\theta)(R)(1)}{R^3}
       = \frac{\mu_0 I}{4\pi R}\,d\theta
\]

%========================================================
% (a)
%========================================================
\subsection*{(a) Campo del filamento de radio $R_1$ (arco en el plano $xy$)}
\[
    B_1 = \frac{\mu_0 I}{4\pi R_1}\int_{\pi/8}^{3\pi/8} d\theta
        = \frac{\mu_0 I}{4\pi R_1}\left(\frac{3\pi}{8}-\frac{\pi}{8}\right)
        = \frac{\mu_0 I}{4\pi R_1}\cdot\frac{\pi}{4}
\]
\textbf{Regla de la mano derecha:} corriente \emph{antihoraria} en el plano $xy$ $\Rightarrow$
$\vec{B}_1$ sale del plano $xy$, en direcci\'on $+\hat{k}$ (eje $z$):
\[
    \boxed{\vec{B}_1 = \frac{\mu_0 I}{16\,R_1}\,\hat{k}}
\]

%========================================================
% (b)
%========================================================
\subsection*{(b) Campo del filamento de radio $R_2$ (arco en el plano $yz$)}
\[
    B_2 = \frac{\mu_0 I}{4\pi R_2}\int_{0}^{\pi/2} d\theta
        = \frac{\mu_0 I}{4\pi R_2}\cdot\frac{\pi}{2}
\]
\textbf{Regla de la mano derecha:} corriente \emph{horaria} en el plano $yz$ $\Rightarrow$
$\vec{B}_2$ apunta en direcci\'on $-\hat{\imath}$ (eje $-x$):
\[
    \boxed{\vec{B}_2 = \frac{\mu_0 I}{8\,R_2}\,(-\hat{\imath})}
\]

%========================================================
% (c)
%========================================================
\subsection*{(c) Campo magn\'etico total en el origen}
Por superposici\'on:
\[
    \vec{B} = \vec{B}_1 + \vec{B}_2
    \;\Rightarrow\;
    \boxed{\vec{B} = \mu_0 I\left(\frac{1}{16\,R_1}\,\hat{k} - \frac{1}{8\,R_2}\,\hat{\imath}\right)}
\]

\vspace{0.3cm}

\noindent\small\textit{Nota (caso general):} el campo en el centro de un arco de radio $R$ y
apertura angular $\Delta\theta$ es $B = \dfrac{\mu_0 I\,\Delta\theta}{4\pi R}$. Para una
circunferencia completa ($\Delta\theta = 2\pi$) se recupera $B = \dfrac{\mu_0 I}{2R}$.\normalsize
