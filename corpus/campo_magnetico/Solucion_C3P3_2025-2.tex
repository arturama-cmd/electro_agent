%========================================
% PROBLEMA 1: ESPIRA (momento dipolar y torque)
%========================================
\section*{Problema 1: Momento dipolar magn\'etico y torque sobre una espira}

\noindent Una espira por la que circula una corriente $I = 5\ [\text{A}]$ est\'a orientada de
modo que su vector de \'area $\vec{A} = 10\,(\hat{\imath} - \hat{\jmath})\ [\text{m}^2]$ es
perpendicular al plano de la espira. La espira se encuentra inmersa en un campo magn\'etico
uniforme $\vec{B} = 5\,(\hat{\jmath} - \hat{k})\ [\text{T}]$. Determine:

\medskip
\noindent\textbf{Datos:}
\begin{itemize}
    \item $I = 5\ [\text{A}]$ \quad (corriente de la espira)
    \item $\vec{A} = 10\,\hat{\imath} - 10\,\hat{\jmath}\ [\text{m}^2]$ \quad (vector de \'area)
    \item $\vec{B} = 5\,\hat{\jmath} - 5\,\hat{k}\ [\text{T}]$ \quad (campo magn\'etico uniforme)
\end{itemize}

\bigskip

%--- (a) momento dipolar ---
\subsection*{(a) Momento dipolar magn\'etico $\vec{\mu}$}

El momento dipolar magn\'etico de una espira se define como el producto de la corriente
por su vector de \'area:
\[
    \vec{\mu} = I\,\vec{A}
\]
\[
    \vec{\mu} = (5)\,\bigl[10\,(\hat{\imath} - \hat{\jmath})\bigr]
\]
\[
    \boxed{\vec{\mu} = 50\,(\hat{\imath} - \hat{\jmath})\ [\text{A}\cdot\text{m}^2]}
\]

\bigskip

%--- (b) torque ---
\subsection*{(b) Momento de torsi\'on (torque) sobre la espira $\vec{\tau}$}

El torque sobre un dipolo magn\'etico en un campo uniforme es:
\[
    \vec{\tau} = \vec{\mu} \times \vec{B}
    = \begin{vmatrix}
        \hat{\imath} & \hat{\jmath} & \hat{k} \\[2pt]
        50 & -50 & 0 \\[2pt]
        0 & 5 & -5
      \end{vmatrix}
\]

Desarrollando el determinante por la primera fila:
\begin{align*}
    \vec{\tau} &= \hat{\imath}\bigl[(-50)(-5) - (0)(5)\bigr]
                - \hat{\jmath}\bigl[(50)(-5) - (0)(0)\bigr]
                + \hat{k}\bigl[(50)(5) - (-50)(0)\bigr] \\[4pt]
               &= \hat{\imath}\,(250 - 0) - \hat{\jmath}\,(-250 - 0) + \hat{k}\,(250 - 0) \\[4pt]
               &= 250\,\hat{\imath} + 250\,\hat{\jmath} + 250\,\hat{k}
\end{align*}

\[
    \boxed{\vec{\tau} = 250\,(\hat{\imath} + \hat{\jmath} + \hat{k})\ [\text{N}\cdot\text{m}]}
\]

\section*{Problema 2: Fuerza magn\'etica sobre un conductor de geometr\'ia mixta}

\textbf{segmento curvo} de radio $R$ transporta una corriente $I = I_0$ en una regi\'on de
campos magn\'eticos uniformes de diferentes direcciones, seg\'un la figura. Determine la
fuerza magn\'etica sobre cada tramo y la fuerza total.

\medskip

\hfill
\begin{minipage}[t]{0.50\textwidth}
    \vspace{0pt}
    \textbf{Ley de fuerza magn\'etica sobre una corriente:}
    \[
        \vec{F} = I\!\int \vec{d\ell}\times\vec{B}
    \]

    \smallskip
    \textbf{Productos vectoriales \'utiles:}
    \[
        \hat{\imath}\times\hat{k} = -\hat{\jmath},\qquad
        \hat{\jmath}\times\hat{k} = \hat{\imath}
    \]

    \smallskip
    \textbf{Tramo recto:} horizontal, corriente hacia $+\hat{\imath}$,
    inmerso en $\vec{B}=B_0\hat{k}$, con $\vec{d\ell}=dx\,\hat{\imath}$
    y $x:\,-L\to 0$.

    \smallskip
    \textbf{Tramo curvo:} arco de radio $R$ recorrido en sentido
    antihorario, inmerso en $\vec{B}=-B_0\hat{k}$, parametrizado por
    \[
        \vec{d\ell}=R\sin\theta\,d\theta(-\hat{\imath})+R\cos\theta\,d\theta\,\hat{\jmath},
    \]
    con $\theta:\,-\tfrac{\pi}{2}\to \tfrac{\pi}{3}$.
\end{minipage}

\bigskip

%--- (a) segmento recto ---
\subsection*{(a) Fuerza sobre el segmento recto $\vec{F}_1$}

Con $\vec{d\ell}=dx\,\hat{\imath}$ y $\vec{B}=B_0\hat{k}$:
\begin{align*}
    \vec{F}_1 &= I_0\int_{-L}^{0} (dx\,\hat{\imath})\times(B_0\hat{k})
             = I_0 B_0 \int_{-L}^{0} dx\,(\hat{\imath}\times\hat{k}) \\[4pt]
            &= I_0 B_0 \int_{-L}^{0} dx\,(-\hat{\jmath})
             = -I_0 B_0\,\hat{\jmath}\,\bigl[x\bigr]_{-L}^{0}
             = -I_0 B_0\,\hat{\jmath}\,\bigl(0-(-L)\bigr)
\end{align*}

\[
    \boxed{\vec{F}_1 = -\,I_0 B_0\,L\,\hat{\jmath}\ [\text{N}]}
\]

\bigskip

%--- (b) segmento curvo ---
\subsection*{(b) Fuerza sobre el segmento curvo $\vec{F}_2$}

Con $\vec{d\ell}=R\sin\theta\,d\theta(-\hat{\imath})+R\cos\theta\,d\theta\,\hat{\jmath}$
y $\vec{B}=-B_0\hat{k}$, el integrando es:
\begin{align*}
    \vec{d\ell}\times\vec{B}
      &= \bigl[R\sin\theta(-\hat{\imath})+R\cos\theta\,\hat{\jmath}\bigr]\,d\theta
         \times(-B_0\hat{k}) \\[4pt]
      &= -B_0 R\,\sin\theta\,d\theta\,(\hat{\imath}\times\hat{k})
         - B_0 R\,\cos\theta\,d\theta\,(\hat{\jmath}\times\hat{k}) \\[4pt]
      &= -B_0 R\,\cos\theta\,d\theta\,\hat{\imath}
         - B_0 R\,\sin\theta\,d\theta\,\hat{\jmath}
\end{align*}

Integrando entre $\theta=-\tfrac{\pi}{2}$ y $\theta=\tfrac{\pi}{3}$:
\begin{align*}
    \vec{F}_2 &= -I_0 B_0 R\int_{-\pi/2}^{\pi/3}
                 \bigl(\cos\theta\,\hat{\imath}+\sin\theta\,\hat{\jmath}\bigr)d\theta \\[4pt]
              &= -I_0 B_0 R\Bigl[\,\hat{\imath}\,(\sin\theta)
                 -\hat{\jmath}\,(\cos\theta)\,\Bigr]_{-\pi/2}^{\pi/3} \\[4pt]
              &= -I_0 B_0 R\Bigl[\hat{\imath}\bigl(\sin\tfrac{\pi}{3}-\sin(-\tfrac{\pi}{2})\bigr)
                 -\hat{\jmath}\bigl(\cos\tfrac{\pi}{3}-\cos(-\tfrac{\pi}{2})\bigr)\Bigr] \\[4pt]
              &= -I_0 B_0 R\Bigl[\hat{\imath}\Bigl(\tfrac{\sqrt{3}}{2}+1\Bigr)
                 -\hat{\jmath}\Bigl(\tfrac{1}{2}-0\Bigr)\Bigr]
\end{align*}

\[
    \boxed{\vec{F}_2 = I_0 B_0 R\left[-\Bigl(1+\tfrac{\sqrt{3}}{2}\Bigr)\hat{\imath}
        + \tfrac{1}{2}\,\hat{\jmath}\right]\ [\text{N}]}
\]

\bigskip

%--- (c) fuerza total ---
\subsection*{(c) Fuerza magn\'etica total $\vec{F}$}

\[
    \vec{F} = \vec{F}_1 + \vec{F}_2
            = -I_0 B_0 L\,\hat{\jmath}
              + I_0 B_0 R\left[-\Bigl(1+\tfrac{\sqrt{3}}{2}\Bigr)\hat{\imath}
              + \tfrac{1}{2}\,\hat{\jmath}\right]
\]

\[
    \boxed{\vec{F} = I_0 B_0\left[-\Bigl(1+\tfrac{\sqrt{3}}{2}\Bigr)R\,\hat{\imath}
        + \Bigl(\tfrac{1}{2}R - L\Bigr)\hat{\jmath}\right]\ [\text{N}]}
\]
