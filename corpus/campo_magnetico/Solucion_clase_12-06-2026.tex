\subsection*{An\'alisis (potenciales de nodo)}
Los nodos son $a$, $b$ y el nodo central $m$ (entre $\varepsilon_3$ y $\varepsilon_2$). Las
fuentes $\varepsilon_3$ y $\varepsilon_2$ son \textbf{ideales} (sin resistencia, con su $+$
hacia la izquierda), por lo que fijan directamente la ca\'ida de potencial:
\[
    V_a - V_m = \varepsilon_3 = 4\ [\text{V}], \qquad
    V_m - V_b = \varepsilon_2 = 5\ [\text{V}]
\]
\[
    \Rightarrow\quad
    \boxed{V_{ab} = V_a - V_b = \varepsilon_3 + \varepsilon_2 = 9\ [\text{V}]}
\]
Tomando $V_b = 0$: $\;V_a = 9\ \text{V}$, $\;V_m = 5\ \text{V}$.

\subsection*{C\'alculo de las corrientes}
\textbf{Rama superior} ($\varepsilon_1$ y $R_2$ en serie entre $a$ y $b$):
\[
    V_a - V_b = \varepsilon_1 + I_1 R_2 \;\Rightarrow\; 9 = 6 + 50\,I_1
    \;\Rightarrow\; \boxed{I_1 = 0{,}06\ [\text{A}]\ (a\to b)}
\]
\textbf{Rama de $R_1$} (entre $a$ y $m$, con $V_a-V_m = 4\ \text{V}$):
\[
    I_{R_1} = \frac{V_a - V_m}{R_1} = \frac{4}{100}
    \;\Rightarrow\; \boxed{I_{R_1} = 0{,}04\ [\text{A}]\ (a\to m)}
\]
\textbf{Rama de $\varepsilon_3$} (KCL en el nodo $a$):
\[
    I_3 = I_1 + I_{R_1} = 0{,}06 + 0{,}04
    \;\Rightarrow\; \boxed{I_3 = 0{,}10\ [\text{A}]\ (m\to a)}
\]
\textbf{Rama de $\varepsilon_2$} (KCL en el nodo $b$):
\[
    I_2 = I_1 = \boxed{0{,}06\ [\text{A}]\ (b\to m)}
\]

\subsection*{Corrientes de rama (KCL)}
Las corrientes dadas circulan \textbf{hacia abajo} por el riel izquierdo: $1{,}0$ A en el
segmento superior y $2{,}0$ A en el inferior. Aplicando KCL en el nodo $a$, las corrientes de
rama resultan:
\[
    I_{\text{sup}} = 1{,}0\ \text{A}\ (b\to a),\qquad
    I_{\text{med}} = 1{,}0\ \text{A}\ (b\to a),\qquad
    I_{\text{inf}} = 2{,}0\ \text{A}\ (a\to b)
\]

\subsection*{Diferencia de potencial (rama superior, totalmente conocida)}
Recorriendo la rama superior de $a$ a $b$ ($1\,\Omega$, fuente de $20$ V con su $-$ hacia $b$,
$6\,\Omega$), con su corriente de $1$ A en sentido $b\to a$ (las resistencias \emph{suben} el
potencial y la fuente lo \emph{baja}):
\[
    V_b - V_a = (1)(1) - 20 + (1)(6) = -13
    \;\Rightarrow\; \boxed{V_{ab} = V_a - V_b = 13\ [\text{V}]}
\]

\subsection*{Determinaci\'on de $\varepsilon_1$ y $\varepsilon_2$}
\textbf{Rama media} ($4\,\Omega$, $1\,\Omega$, $\varepsilon_1$; corriente $1$ A en sentido $b\to a$):
\[
    V_b - V_a = (1)(4) + (1)(1) - \varepsilon_1 = 5 - \varepsilon_1 = -13
    \;\Rightarrow\; \boxed{\varepsilon_1 = 18\ [\text{V}]}
\]
\textbf{Rama inferior} ($1\,\Omega$, $\varepsilon_2$, $2\,\Omega$; corriente $2$ A en sentido $a\to b$):
\[
    V_b - V_a = -(2)(1) - \varepsilon_2 - (2)(2) = -6 - \varepsilon_2 = -13
    \;\Rightarrow\; \boxed{\varepsilon_2 = 7\ [\text{V}]}
\]

\section*{Problema 3: Fuerza y torque sobre una espira triangular}

\noindent Por una espira triangular (rect\'angulo en $a$) circula una corriente
$I_0 = 1\ [\text{A}]$, inmersa en un campo magn\'etico uniforme $B = 0{,}5\ [\text{T}]$
orientado a $30^\circ$ del eje $x$, seg\'un la figura. Los lados miden
$l_{ab} = 3\ [\text{cm}]$, $l_{ac} = 4\ [\text{cm}]$ y $l_{bc} = 5\ [\text{cm}]$ (tri\'angulo
$3$-$4$-$5$). Determine la fuerza magn\'etica sobre cada segmento, el momento magn\'etico de la
espira y el torque que ejerce $\vec{B}$.

\medskip

\noindent
\begin{minipage}[t]{0.42\textwidth}
\centering
\vspace{0pt}
\begin{tikzpicture}[scale=1.0, >=Stealth, line width=1pt]
    % ---- ejes ----

\hfill
\begin{minipage}[t]{0.55\textwidth}
\vspace{0pt}
\textbf{Datos:}
\begin{itemize}\setlength\itemsep{1pt}
    \item $I_0 = 1\ [\text{A}]$,\quad $B = 0{,}5\ [\text{T}]$ a $30^\circ$ de $\hat{\imath}$
    \item $l_{ab}=0{,}03\ \text{m}$, $\;l_{ac}=0{,}04\ \text{m}$, $\;l_{bc}=0{,}05\ \text{m}$
    \item $a=(0,0)$, $\;b=(-0{,}03,\,0)$, $\;c=(0,\,-0{,}04)$
    \item Sentido de la corriente: $a\to b\to c\to a$
\end{itemize}

\smallskip
\textbf{Productos cruz de unitarios:}
\[
    \hat{\imath}\times\hat{\jmath}=\hat{k},\quad
    \hat{\jmath}\times\hat{\imath}=-\hat{k},\quad
    \hat{k}\times\hat{\imath}=\hat{\jmath},\quad
    \hat{k}\times\hat{\jmath}=-\hat{\imath}
\]
\end{minipage}

\medskip

%--------------------------------------------------
\subsection*{(a) Campo magn\'etico $\vec{B}$}
El campo est\'a en el plano de la espira, a $30^\circ$ del eje $x$ (apuntando hacia
abajo-izquierda):
\[
    \vec{B} = -B\cos30^\circ\,\hat{\imath} - B\sin30^\circ\,\hat{\jmath}
            = -0{,}5\cos30^\circ\,\hat{\imath} - 0{,}5\sin30^\circ\,\hat{\jmath}
\]
\[
    \boxed{\vec{B} = -0{,}433\,\hat{\imath} - 0{,}250\,\hat{\jmath}\ [\text{T}]}
\]

%--------------------------------------------------
\subsection*{(b) Fuerza magn\'etica sobre cada segmento}
$\vec{F} = I_0\,\vec{L}\times\vec{B}$, con $\vec{L}$ en el sentido de la corriente:
\[
    \vec{L}_{ab} = -0{,}03\,\hat{\imath}\ (a\to b),\quad
    \vec{L}_{bc} = 0{,}03\,\hat{\imath}-0{,}04\,\hat{\jmath}\ (b\to c),\quad
    \vec{L}_{ca} = 0{,}04\,\hat{\jmath}\ (c\to a)
\]

\noindent\textbf{Segmento $ab$:}
\begin{align*}
    \vec{F}_{ab} &= I_0\,(-0{,}03\,\hat{\imath})\times(-0{,}433\,\hat{\imath}-0{,}250\,\hat{\jmath}) \\
                 &= I_0\big[(-0{,}03)(-0{,}250)(\hat{\imath}\times\hat{\jmath})\big]
                  = (1)(0{,}0075)\,\hat{k}
\end{align*}
\[
    \boxed{\vec{F}_{ab} = +7{,}5\times10^{-3}\,\hat{k}\ [\text{N}]}
\]

\noindent\textbf{Segmento $bc$ (hipotenusa):}
\begin{align*}
    \vec{F}_{bc} &= I_0\,(0{,}03\,\hat{\imath}-0{,}04\,\hat{\jmath})\times(-0{,}433\,\hat{\imath}-0{,}250\,\hat{\jmath}) \\
                 &= I_0\big[(0{,}03)(-0{,}250)(\hat{\imath}\times\hat{\jmath})
                    + (-0{,}04)(-0{,}433)(\hat{\jmath}\times\hat{\imath})\big] \\
                 &= I_0\big[-0{,}0075\,\hat{k} - 0{,}0173\,\hat{k}\big]
\end{align*}
\[
    \boxed{\vec{F}_{bc} = -2{,}48\times10^{-2}\,\hat{k}\ [\text{N}]}
\]

\noindent\textbf{Segmento $ca$ (cateto vertical):}
\begin{align*}
    \vec{F}_{ca} &= I_0\,(0{,}04\,\hat{\jmath})\times(-0{,}433\,\hat{\imath}-0{,}250\,\hat{\jmath}) \\
                 &= I_0\big[(0{,}04)(-0{,}433)(\hat{\jmath}\times\hat{\imath})\big]
                  = (1)(-0{,}0173)(-\hat{k})
\end{align*}
\[
    \boxed{\vec{F}_{ca} = +1{,}73\times10^{-2}\,\hat{k}\ [\text{N}]}
\]

\noindent\textit{Verificaci\'on:} $\vec{F}_{ab}+\vec{F}_{bc}+\vec{F}_{ca}
= (7{,}5-24{,}8+17{,}3)\times10^{-3}\,\hat{k} \approx \vec{0}$ \;(lazo cerrado en campo uniforme).

%--------------------------------------------------
\subsection*{(c) Momento magn\'etico $\vec{\mu}$}
\'Area del tri\'angulo:
\[
    A = \tfrac{1}{2}\,l_{ab}\,l_{ac} = \tfrac{1}{2}(0{,}03)(0{,}04) = 6{,}0\times10^{-4}\ [\text{m}^2]
\]

\noindent\textbf{La direcci\'on del momento magn\'etico se determina con la regla de la mano
derecha:} al curvar los dedos de la mano derecha en el sentido de circulaci\'on de la corriente
($a\to b\to c\to a$, antihorario), el \textbf{pulgar} apunta en la direcci\'on de $\vec{\mu}$, es
decir, \textbf{saliendo del plano} de la espira ($+\hat{k}$). Por lo tanto $\hat{n}=+\hat{k}$ y:
\[
    \vec{\mu} = I_0\,A\,\hat{n} = (1)(6{,}0\times10^{-4})(+\hat{k})
\]
\[
    \boxed{\vec{\mu} = +6{,}0\times10^{-4}\,\hat{k}\ [\text{A}\cdot\text{m}^2]}
\]

%--------------------------------------------------
\subsection*{(d) Torque magn\'etico $\vec{\tau}$}
\begin{align*}
    \vec{\tau} = \vec{\mu}\times\vec{B}
       &= (6{,}0\times10^{-4}\,\hat{k})\times(-0{,}433\,\hat{\imath}-0{,}250\,\hat{\jmath}) \\
       &= 6{,}0\times10^{-4}\big[-0{,}433\,(\hat{k}\times\hat{\imath}) - 0{,}250\,(\hat{k}\times\hat{\jmath})\big] \\
       &= 6{,}0\times10^{-4}\big[-0{,}433\,\hat{\jmath} - 0{,}250\,(-\hat{\imath})\big]
\end{align*}
\[
    \boxed{\vec{\tau} = 1{,}50\times10^{-4}\,\hat{\imath} - 2{,}60\times10^{-4}\,\hat{\jmath}\ [\text{N}\cdot\text{m}]}
    \qquad |\vec{\tau}| = 3{,}0\times10^{-4}\ [\text{N}\cdot\text{m}]
\]

\subsection*{Interpretaci\'on f\'isica del movimiento de torque}
El torque hace \textbf{girar} la espira tendiendo a alinear su momento magn\'etico $\vec{\mu}$
con el campo $\vec{B}$. La \textbf{regla de la mano derecha} indica el sentido del giro:

\begin{itemize}
    \item El \textbf{dedo pulgar} de la mano derecha apunta en la direcci\'on del vector torque
          $\vec{\tau}$ (el \emph{eje de rotaci\'on}).
    \item Los \textbf{dedos} de la mano (al curvarse) indican el \textbf{sentido de circulaci\'on
          del giro} de la espira alrededor de ese eje.
\end{itemize}

\noindent As\'i, la espira rota en torno al eje
$\vec{\tau} = (1{,}50\,\hat{\imath} - 2{,}60\,\hat{\jmath})\times10^{-4}\ [\text{N}\cdot\text{m}]$
hasta que $\vec{\mu}$ quede paralelo a $\vec{B}$, posici\'on en la que el torque se anula
($\vec{\mu}\parallel\vec{B}\Rightarrow\vec{\tau}=\vec{0}$): la configuraci\'on de equilibrio
estable de la espira.
