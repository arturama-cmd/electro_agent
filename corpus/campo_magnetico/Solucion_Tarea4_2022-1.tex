\section*{Problema 1: Tres alambres y una carga puntual}

\noindent Tres conductores rectos muy extensos conducen $I_1 = 2{,}6$, $I_2 = 5{,}1$ e
$I_3 = 3{,}2\ [\text{A}]$ (direcciones de la figura). Una carga $q = 5{,}8\ [\text{mC}]$ se mueve
con rapidez $v = 50\ [\text{m/s}]$. Con $a = 5{,}4$, $b = 2{,}8$ y $c = 7{,}3\ [\text{cm}]$,
determine $\vec{B}$ y $\vec{F}_m$ sobre $q$, y la fuerza sobre $2\ [\text{m}]$ de $I_2$.

\medskip

\textbf{(a) Campo magn\'etico en la posici\'on de $q$:}
\[
    \vec{B} = \frac{\mu_0 I_1}{2\pi a}\hat{k}
            + \frac{\mu_0 I_2}{2\pi (a+b)}(-\hat{k})
            + \frac{\mu_0 I_3}{2\pi c}\hat{k}
\]
\[
    = 2\times10^{-7}\!\left[\frac{2{,}6}{0{,}054} - \frac{5{,}1}{0{,}082} + \frac{3{,}2}{0{,}073}\right]\hat{k}
\]
\[
    \boxed{\vec{B} = 5{,}958\times10^{-6}\,\hat{k}\ [\text{T}]}
\]
\end{minipage}

\medskip

\noindent\textbf{(b) Fuerza magn\'etica sobre la carga} $\vec{F}_m = q\,\vec{v}\times\vec{B}$, con
$\vec{v} = 50\cos60^\circ(-\hat{\imath}) + 50\sin60^\circ\,\hat{\jmath}$:
\begin{align*}
    \vec{F}_m &= (5{,}8\times10^{-3})\big[-25\,\hat{\imath} + 43{,}3\,\hat{\jmath}\big]\times(5{,}958\times10^{-6}\,\hat{k}) \\
              &= (5{,}8\times10^{-3})(5{,}958\times10^{-6})\big[-25(\hat{\imath}\times\hat{k}) + 43{,}3(\hat{\jmath}\times\hat{k})\big] \\
              &= (3{,}456\times10^{-8})\big[25\,\hat{\jmath} + 43{,}3\,\hat{\imath}\big]
\end{align*}
\[
    \boxed{\vec{F}_m = 1{,}496\times10^{-6}\,\hat{\imath} + 8{,}639\times10^{-7}\,\hat{\jmath}\ [\text{N}]}
\]

\medskip

\noindent\textbf{(c) Fuerza sobre $2\ [\text{m}]$ de $I_2$ debida a $I_1$:} el campo de $I_1$ en la
posici\'on de $I_2$ (a distancia $b$) es $\vec{B}_1 = \dfrac{\mu_0 I_1}{2\pi b}(-\hat{k})$. Con
$\vec{L} = 2(-\hat{\jmath})$ (sentido de $I_2$):
\[
    \vec{F} = I_2\,\vec{L}\times\vec{B}_1
            = (5{,}1)(2)(-\hat{\jmath})\times\frac{\mu_0(2{,}6)}{2\pi(0{,}028)}(-\hat{k})
\]
\[
    \boxed{\vec{F} = 1{,}894\times10^{-4}\,\hat{\imath}\ [\text{N}]}
\]

\section*{Problema 2: Espira en el centro de un solenoide variable}

\noindent Una espira de $N = 20$ vueltas, radio $R_1 = 7{,}0\ [\text{cm}]$ y resistencia
$R = 45\ [\Omega]$ rodea (en su centro) a un solenoide muy largo de $n = 200\ [\text{vueltas/m}]$
y radio $R_2 = 5{,}0\ [\text{cm}]$. Por el solenoide circula $I(t) = 3\sqrt{t}\ [\text{A}]$.
Determine el flujo, la fem inducida y la corriente inducida en la espira.

\medskip

\noindent\textbf{(a) Flujo magn\'etico} (el campo solo existe dentro del solenoide,
$A = \pi R_2^2$):
\begin{align*}
    \Phi = \int \vec{B}\cdot d\vec{S} = \mu_0\,n\,I(t)\,(\pi R_2^2)
       &= (4\pi\times10^{-7})(200)(3\sqrt{t})\,\pi(0{,}05)^2
\end{align*}
\[
    \boxed{\Phi(t) = 5{,}922\times10^{-6}\,\sqrt{t}\ [\text{Wb}]}
\]

\noindent\textbf{(b) Fem inducida} (ley de Faraday, $N$ vueltas):
\[
    \varepsilon = -N\frac{d\Phi}{dt}
                = -20\cdot 5{,}922\times10^{-6}\cdot\frac{1}{2\sqrt{t}}
\]
\[
    \boxed{\varepsilon(t) = -\frac{5{,}922\times10^{-5}}{\sqrt{t}}\ [\text{V}]}
\]

\noindent\textbf{(c) Corriente inducida:}
\[
    I(t) = \frac{|\varepsilon|}{R} = \frac{5{,}922\times10^{-5}}{45\sqrt{t}}
    \;\Rightarrow\; \boxed{I(t) = \frac{1{,}316\times10^{-6}}{\sqrt{t}}\ [\text{A}]}
\]
Como el flujo \textbf{aumenta} con $t$, por la ley de Lenz la corriente inducida circula en
\textbf{sentido opuesto} a la corriente del solenoide (para oponerse al aumento del flujo).

\section*{Problema 3: Problema 7 de la Gu\'ia 9}

\noindent Una espira circular de radio $R = 10\ [\text{cm}]$ con $I_2 = 1\ [\text{A}]$ (horaria),
cuyo centro est\'a a $D = 20\ [\text{cm}]$ de un alambre recto. Hallar $I_1$ para que el campo en
el centro sea cero.

\medskip

\noindent El campo de la espira en su centro y el del alambre deben anularse:
\[
    \frac{\mu_0 I_1}{2\pi D} = \frac{\mu_0 I_2}{2R}
    \;\Rightarrow\;
    I_1 = \frac{\pi D}{R}\,I_2 = \frac{\pi(0{,}20)}{0{,}10}(1)
    \;\Rightarrow\; \boxed{I_1 = 6{,}28\ [\text{A}]\ \text{(de izquierda a derecha)}}
\]
\noindent\small\textit{(Desarrollo completo en la soluci\'on de la Gu\'ia 9, Problema 7.)}\normalsize
