\section*{Problema 3: Fuerza sobre una espira rectangular junto a un alambre}

\noindent Un alambre recto y muy largo $AB$ conduce una corriente $I_1 = 14{,}0\ [\text{A}]$
(hacia la derecha). Una espira rectangular, cuyos lados largos son paralelos al alambre,
conduce una corriente $I_2 = 5{,}00\ [\text{A}]$ en sentido horario, como muestra la Fig.\ 3.
El lado superior de la espira est\'a a $2{,}6\ [\text{cm}]$ del alambre y el lado inferior a
$10{,}0\ [\text{cm}]$; los lados largos miden $20{,}0\ [\text{cm}]$. Determine el
\textbf{momento magn\'etico} de la espira y la \textbf{fuerza} que el campo del alambre ejerce
sobre cada lado.

\medskip

\medskip

\noindent\textbf{Datos:}
\[
    I_1 = 14{,}0\ [\text{A}],\quad I_2 = 5{,}00\ [\text{A}],\quad
    a = 0{,}026\ [\text{m}],\quad b = 0{,}10\ [\text{m}],\quad L = 0{,}20\ [\text{m}]
\]
donde $a$ y $b$ son las distancias del alambre a los lados superior e inferior, y $L$ la
longitud de los lados largos. La altura de la espira (lados cortos) es
$h = b - a = 0{,}074\ [\text{m}]$.

\medskip

\noindent\textbf{Sistema de referencia.} Adoptamos ejes cartesianos con
$\hat{\imath}$ hacia la \emph{derecha} ($+x$), $\hat{\jmath}$ hacia \emph{arriba} ($+y$) y
$\hat{k}$ \emph{saliendo de la p\'agina} ($+z$). Recordamos los productos cruz de la base:
\[
    \hat{\imath}\times\hat{\jmath}=\hat{k},\qquad
    \hat{\jmath}\times\hat{k}=\hat{\imath},\qquad
    \hat{k}\times\hat{\imath}=\hat{\jmath},\qquad
    \hat{\imath}\times\hat{k}=-\hat{\jmath}.
\]

\medskip

\noindent\textbf{Campo del alambre recto.} El alambre lleva $I_1$ en $+\hat{\imath}$. En un punto
a distancia $r$ \emph{por debajo} del alambre, el vector que va del alambre al punto es
$\hat{r}=-\hat{\jmath}$, y por la regla de la mano derecha
$\hat{B}\propto\hat{\imath}\times(-\hat{\jmath})=-\hat{k}$. As\'i, en toda la regi\'on de la espira
el campo \textbf{entra a la p\'agina}:
\[
    \vec{B}(r) = -\,\frac{\mu_0 I_1}{2\pi r}\,\hat{k}.
\]
La fuerza sobre un tramo recto con corriente es $\vec{F} = I_2\,\vec{L}\times\vec{B}$, con
$\vec{L}$ apuntando en el sentido de la corriente.

%--------------------------------------------------
\subsection*{Momento magn\'etico de la espira}

El \'area encerrada es $A = L\,h = (0{,}20)(0{,}074) = 1{,}48\times10^{-2}\ [\text{m}^2]$.
La corriente circula en sentido \emph{horario} en el plano $xy$; por la regla de la mano derecha
la normal asociada apunta \textbf{entrando a la p\'agina}, es decir $\hat{n}=-\hat{k}$. Por lo tanto
\[
    \vec{\mu} = I_2\,A\,\hat{n} = (5{,}00)(1{,}48\times10^{-2})\,(-\hat{k})
\]
\[
    \boxed{\vec{\mu} = -7{,}4\times10^{-2}\,\hat{k}\ [\text{A}\cdot\text{m}^2]\quad\text{(entrando a la p\'agina)}}
\]

%--------------------------------------------------
\subsection*{Lado superior (a $a=2{,}6$ cm, corriente hacia la derecha)}

La corriente va hacia la derecha, $\vec{L}_{\text{sup}}=L\,\hat{\imath}$, y el campo a distancia
$a$ es $\vec{B}(a)=-\dfrac{\mu_0 I_1}{2\pi a}\,\hat{k}$. El producto cruz da
\begin{align*}
    \vec{F}_{\text{sup}} &= I_2\,\vec{L}_{\text{sup}}\times\vec{B}(a)
    = I_2\,(L\,\hat{\imath})\times\!\left(-\frac{\mu_0 I_1}{2\pi a}\,\hat{k}\right)
    = -\frac{\mu_0 I_1 I_2 L}{2\pi a}\,(\hat{\imath}\times\hat{k}) \\[2pt]
    &= -\frac{\mu_0 I_1 I_2 L}{2\pi a}\,(-\hat{\jmath})
    = +\frac{\mu_0 I_1 I_2 L}{2\pi a}\,\hat{\jmath}
    = \frac{(2\times10^{-7})(14)(5)(0{,}20)}{0{,}026}\,\hat{\jmath}
\end{align*}
\[
    \boxed{\vec{F}_{\text{sup}} = +1{,}08\times10^{-4}\,\hat{\jmath}\ [\text{N}]\quad\text{(hacia el alambre, atracci\'on)}}
\]
El signo $+\hat{\jmath}$ (hacia arriba, hacia el alambre) confirma la \textbf{atracci\'on} entre
corrientes paralelas.

%--------------------------------------------------
\subsection*{Lado inferior (a $b=10{,}0$ cm, corriente hacia la izquierda)}

La corriente va hacia la izquierda, $\vec{L}_{\text{inf}}=-L\,\hat{\imath}$, y el campo a distancia
$b$ es $\vec{B}(b)=-\dfrac{\mu_0 I_1}{2\pi b}\,\hat{k}$. Entonces
\begin{align*}
    \vec{F}_{\text{inf}} &= I_2\,\vec{L}_{\text{inf}}\times\vec{B}(b)
    = I_2\,(-L\,\hat{\imath})\times\!\left(-\frac{\mu_0 I_1}{2\pi b}\,\hat{k}\right)
    = +\frac{\mu_0 I_1 I_2 L}{2\pi b}\,(\hat{\imath}\times\hat{k}) \\[2pt]
    &= +\frac{\mu_0 I_1 I_2 L}{2\pi b}\,(-\hat{\jmath})
    = -\frac{\mu_0 I_1 I_2 L}{2\pi b}\,\hat{\jmath}
    = -\frac{(2\times10^{-7})(14)(5)(0{,}20)}{0{,}10}\,\hat{\jmath}
\end{align*}
\[
    \boxed{\vec{F}_{\text{inf}} = -2{,}8\times10^{-5}\,\hat{\jmath}\ [\text{N}]\quad\text{(alej\'andose del alambre, repulsi\'on)}}
\]
El signo $-\hat{\jmath}$ (hacia abajo, alej\'andose del alambre) confirma la \textbf{repulsi\'on}
entre corrientes antiparalelas.

%--------------------------------------------------
\subsection*{Lados laterales (verticales, de $a$ a $b$)}

Sobre estos lados verticales el campo $\vec{B}=-\frac{\mu_0 I_1}{2\pi r}\hat{k}$ \emph{var\'ia}
con la distancia $r$, pero su \emph{direcci\'on} ($-\hat{k}$) es constante; lo mismo la
direcci\'on de la corriente. Por ello la \textbf{direcci\'on} de la fuerza sale del producto cruz
de los versores y la \textbf{magnitud} de integrar $1/r$ entre $a$ y $b$.

\medskip

\noindent\emph{Lado izquierdo} (corriente hacia arriba, $\hat{L}=\hat{\jmath}$):
\[
    \hat{L}\times\hat{B} = \hat{\jmath}\times(-\hat{k}) = -(\hat{\jmath}\times\hat{k}) = -\hat{\imath}
    \quad\Rightarrow\quad \vec{F}_{\text{izq}}\parallel -\hat{\imath}\ \text{(hacia la izquierda).}
\]
\noindent\emph{Lado derecho} (corriente hacia abajo, $\hat{L}=-\hat{\jmath}$):
\[
    \hat{L}\times\hat{B} = (-\hat{\jmath})\times(-\hat{k}) = \hat{\jmath}\times\hat{k} = \hat{\imath}
    \quad\Rightarrow\quad \vec{F}_{\text{der}}\parallel +\hat{\imath}\ \text{(hacia la derecha).}
\]
La magnitud com\'un es
\begin{align*}
    F_{\text{lat}} &= I_2\!\int_a^b \frac{\mu_0 I_1}{2\pi r}\,dr
    = \frac{\mu_0 I_1 I_2}{2\pi}\int_a^b \frac{dr}{r}
    = \frac{\mu_0 I_1 I_2}{2\pi}\,\ln\!\left(\frac{b}{a}\right) \\[2pt]
    &= (2\times10^{-7})(14)(5)\,\ln\!\left(\frac{0{,}10}{0{,}026}\right)
    = (1{,}4\times10^{-5})\,\ln(3{,}85)
\end{align*}
\[
    \boxed{\vec{F}_{\text{izq}} = -1{,}89\times10^{-5}\,\hat{\imath}\ [\text{N}],\qquad
           \vec{F}_{\text{der}} = +1{,}89\times10^{-5}\,\hat{\imath}\ [\text{N}]}
\]
Son iguales y opuestas ($\pm\hat{\imath}$): solo tienden a \emph{estirar} la espira y su suma es nula.

%--------------------------------------------------
\subsection*{Resumen}

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|c|c|l|}
\hline
\textbf{Lado} & \textbf{$\vec{F}$ [N]} & \textbf{Direcci\'on} \\
\hline
Superior  & $+1{,}08\times10^{-4}\,\hat{\jmath}$ & hacia el alambre (atracci\'on) \\
Inferior  & $-2{,}8\times10^{-5}\,\hat{\jmath}$  & alej\'andose del alambre (repulsi\'on) \\
Izquierdo & $-1{,}89\times10^{-5}\,\hat{\imath}$ & hacia la izquierda \\
Derecho   & $+1{,}89\times10^{-5}\,\hat{\imath}$ & hacia la derecha \\
\hline
\end{tabular}
\end{center}

\medskip
\noindent\textbf{Fuerza neta.} Sumando las cuatro contribuciones vectoriales, las laterales se
cancelan ($-1{,}89\times10^{-5}\,\hat{\imath}+1{,}89\times10^{-5}\,\hat{\imath}=\vec{0}$) y quedan
solo las verticales:
\begin{align*}
    \vec{F}_{\text{neta}} &= \vec{F}_{\text{sup}}+\vec{F}_{\text{inf}}+\vec{F}_{\text{izq}}+\vec{F}_{\text{der}} \\[2pt]
    &= \left(1{,}08\times10^{-4} - 2{,}8\times10^{-5}\right)\hat{\jmath}
    = +8{,}0\times10^{-5}\,\hat{\jmath}\ [\text{N}]
\end{align*}
\[
    \boxed{\vec{F}_{\text{neta}} = +8{,}0\times10^{-5}\,\hat{\jmath}\ [\text{N}]\quad\text{(hacia el alambre)}}
\]
El signo $+\hat{\jmath}$ indica que la espira es atra\'ida netamente \textbf{hacia el conductor}:
la atracci\'on del lado superior (m\'as cercano) supera a la repulsi\'on del inferior.
