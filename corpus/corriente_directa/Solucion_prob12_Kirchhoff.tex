\section*{Enunciado del Problema}

En el circuito de la Fig.\ 3, $R_1=100\ [\Omega]$, $R_2=50\ [\Omega]$, $\varepsilon_1=6{,}0\ [\text{V}]$,
$\varepsilon_2=5{,}0\ [\text{V}]$ y $\varepsilon_3=4{,}0\ [\text{V}]$. Determine las corrientes que
circulan en el circuito y la diferencia de potencial $V_{ab}$ entre los puntos $a$ y $b$.

\section*{Datos del Problema}
\[
    \varepsilon_1 = 6{,}0\ [\text{V}],\quad
    \varepsilon_2 = 5{,}0\ [\text{V}],\quad
    \varepsilon_3 = 4{,}0\ [\text{V}],\quad
    R_1 = 100\ [\Omega],\quad
    R_2 = 50\ [\Omega]
\]

%========================================
% ANÁLISIS
%========================================
\section*{An\'alisis del Circuito}

El circuito tiene s\'olo \textbf{dos nodos} ($a$ y el nodo central $m$) y \textbf{tres ramas en
paralelo} entre ellos. El punto $b$ \emph{no es un nodo}: s\'olo une $R_2$ con $\varepsilon_2$ en
serie, por lo que la corriente de la rama superior y la de $\varepsilon_2$ son \textbf{la misma}
($I_1$). Todas las fuentes tienen su polo $+$ hacia la izquierda. Las tres ramas son:

\begin{itemize}\setlength\itemsep{1pt}
    \item \textbf{Rama 1} ($a\!\to\! m$, v\'ia $b$): $\varepsilon_1$, $R_2$ y $\varepsilon_2$ en
          \emph{serie}. Corriente $I_1$.
    \item \textbf{Rama 2} ($a\!-\!m$): fuente $\varepsilon_3$. Corriente $I_3$ ($m\!\to\! a$).
    \item \textbf{Rama 3} ($a\!-\!m$): resistencia $R_1$. Corriente $I_{R_1}$ ($a\!\to\! m$).
\end{itemize}

\noindent\small\textit{Nota sobre sentidos:} adoptamos sentidos compatibles con KCL. Coinciden con
las flechas de la figura para $\varepsilon_2$ y $\varepsilon_3$; el resultado confirmar\'a que $I_1$
circula de $a$ hacia $m$ (es decir $a\to b\to m$).\normalsize

%========================================
% DESARROLLO
%========================================
\section*{Ecuaciones de Kirchhoff}

\textbf{KCL --- Nodo $a$ (o $m$):} la corriente $I_3$ entra al nodo $a$ y se reparte en $I_1$
e $I_{R_1}$:
\[
    \boxed{I_3 = I_1 + I_{R_1}} \tag{N}
\]

\medskip
\textbf{KVL --- Malla $M_1$} (rama 2 $\varepsilon_3$ + rama 3 $R_1$). Recorriendo
$a\to\varepsilon_3\to m\to R_1\to a$ (la fuente $\varepsilon_3$ \emph{baja} el potencial y $R_1$
lo \emph{sube} al ir contra $I_{R_1}$):
\[
    -\varepsilon_3 + I_{R_1}R_1 = 0
    \;\Rightarrow\;
    \boxed{I_{R_1}R_1 = \varepsilon_3} \tag{M1}
\]

\medskip
\textbf{KVL --- Malla $M_2$} (rama 1 + rama 2). Recorriendo
$a\to\varepsilon_1\to R_2\to b\to\varepsilon_2\to m\to\varepsilon_3\to a$:
\[
    -\varepsilon_1 - I_1 R_2 + \varepsilon_2 + \varepsilon_3 = 0
    \;\Rightarrow\;
    \boxed{I_1 R_2 = \varepsilon_2 + \varepsilon_3 - \varepsilon_1} \tag{M2}
\]

\section*{Resoluci\'on del sistema}

\noindent De \textbf{(M1)}, la corriente por $R_1$ queda despejada de inmediato:
\[
    I_{R_1} = \frac{\varepsilon_3}{R_1} = \frac{4}{100}
    \;\Rightarrow\; \boxed{I_{R_1} = 0{,}04\ [\text{A}]\ (a\to m)}
\]

\noindent De \textbf{(M2)}, la corriente de la rama superior (la misma que pasa por $\varepsilon_2$):
\[
    I_1 = \frac{\varepsilon_2+\varepsilon_3-\varepsilon_1}{R_2}
        = \frac{5+4-6}{50} = \frac{3}{50}
    \;\Rightarrow\; \boxed{I_1 = 0{,}06\ [\text{A}]\ (a\to b\to m)}
\]

\noindent De la ecuaci\'on de nodo \textbf{(N)}:
\[
    I_3 = I_1 + I_{R_1} = 0{,}06 + 0{,}04
    \;\Rightarrow\; \boxed{I_3 = 0{,}10\ [\text{A}]\ (m\to a)}
\]

\section*{Diferencia de potencial $V_{ab}$}

Recorriendo la \textbf{rama superior} de $a$ a $b$ (cae $\varepsilon_1$ en la fuente y $I_1R_2$ en
la resistencia):
\[
    V_{ab} = V_a - V_b = \varepsilon_1 + I_1 R_2 = 6 + (0{,}06)(50)
\]
\[
    \boxed{V_{ab} = 9{,}0\ [\text{V}]}
\]

\noindent\textbf{Verificaci\'on (por la rama media, fuentes ideales):}
$\;V_{ab} = \varepsilon_3 + \varepsilon_2 = 4 + 5 = 9{,}0$ [V] \checkmark

\medskip
\noindent\small\textit{Balance de potencia:} las fuentes $\varepsilon_2$ y $\varepsilon_3$ entregan
$\varepsilon_2 I_1 + \varepsilon_3 I_3 = 0{,}30 + 0{,}40 = 0{,}70$ [W], que se reparten en las
resistencias ($I_1^2R_2 + I_{R_1}^2R_1 = 0{,}18+0{,}16 = 0{,}34$ [W]) y en la fuente
$\varepsilon_1$, que se \emph{carga} ($\varepsilon_1 I_1 = 0{,}36$ [W]): $0{,}34+0{,}36 = 0{,}70$
[W] \checkmark\normalsize
