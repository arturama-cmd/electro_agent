%========================================
% ANÁLISIS
%========================================
\section*{An\'alisis del Circuito}

El circuito tiene \textbf{dos nodos} ($c$ arriba y $d$ abajo) y \textbf{tres ramas}
en paralelo. En cada rama las resistencias en serie se agrupan:

\begin{itemize}
    \item \textbf{Rama izquierda} ($I_1$): \textcolor{colR1}{$8\ \Omega$}.
    \item \textbf{Rama central} ($I_2$): \textcolor{colR2}{$5\ \Omega$} $+$
          \textcolor{colR3}{$1\ \Omega$} $= 6\ \Omega$, con la fuente
          \textcolor{colE1}{$\varepsilon_1 = 4\ \text{V}$}.
    \item \textbf{Rama derecha} ($I_3$): \textcolor{colR4}{$3\ \Omega$} $+$
          \textcolor{colR5}{$1\ \Omega$} $= 4\ \Omega$, con la fuente
          \textcolor{colE2}{$\varepsilon_2 = 12\ \text{V}$}.
\end{itemize}

Se definen dos mallas: $M_1$ (ramas izquierda + central) y $M_2$ (ramas central + derecha).

\textbf{KCL -- Nodo $c$:}

Las corrientes $I_2$ e $I_3$ llegan al nodo $c$ y salen como $I_1$:
\[
    \boxed{I_2 + I_3 = I_1} \tag{N}
\]

\medskip
\textbf{KVL -- Malla $M_1$} (izquierda + central):

Suma de ca\'idas en $8\,\Omega$, fuente de $4\,$V y resistencias $5\,\Omega$, $1\,\Omega$:
\[
    -8\,I_1 + 4 - I_2 - 5\,I_2 = 0
\]
\[
    \boxed{8\,I_1 + 6\,I_2 = 4} \tag{M1}
\]

\medskip
\textbf{KVL -- Malla $M_2$} (central + derecha):

Resistencias $5\,\Omega$, $1\,\Omega$, fuentes $4\,$V y $12\,$V, y $1\,\Omega$, $3\,\Omega$:
\[
    5\,I_2 + I_2 - 4 + 12 - I_3 - 3\,I_3 = 0
\]
\[
    \boxed{6\,I_2 - 4\,I_3 = -8} \tag{M2}
\]

\subsection*{b) Resoluci\'on del sistema}

Sustituyendo la ecuaci\'on de nodos $I_1 = I_2 + I_3$ en \textbf{(M1)}:

\begin{align*}
    8\,(I_2 + I_3) + 6\,I_2 &= 4 \\
    14\,I_2 + 8\,I_3 &= 4 \\
    7\,I_2 + 4\,I_3 &= 2 \tag{M1$'$}
\end{align*}

Sumando \textbf{(M1$'$)} y \textbf{(M2)} para eliminar $I_3$:

\begin{align*}
    (7\,I_2 + 4\,I_3) + (6\,I_2 - 4\,I_3) &= 2 + (-8) \\
    13\,I_2 &= -6
\end{align*}

\[
    \boxed{I_2 = -\frac{6}{13} \approx -0{,}46\ [\text{A}]}
\]

De \textbf{(M2)}, $6\,I_2 - 4\,I_3 = -8$, se despeja $I_3$:

\begin{align*}
    4\,I_3 &= 6\,I_2 + 8 = 6\left(-\frac{6}{13}\right) + 8 = -\frac{36}{13} + \frac{104}{13} = \frac{68}{13}
\end{align*}

\[
    \boxed{I_3 = \frac{17}{13} \approx 1{,}31\ [\text{A}]}
\]

Finalmente, de la ecuaci\'on de nodos \textbf{(N)}:

\[
    I_1 = I_2 + I_3 = -\frac{6}{13} + \frac{17}{13} = \frac{11}{13}
\]

\[
    \boxed{I_1 = \frac{11}{13} \approx 0{,}85\ [\text{A}]}
\]

\noindent\textbf{Verificaci\'on (M1):}
$\;8\,I_1 + 6\,I_2 = 8\cdot\dfrac{11}{13} + 6\cdot\left(-\dfrac{6}{13}\right)
= \dfrac{88 - 36}{13} = \dfrac{52}{13} = 4$ \checkmark
