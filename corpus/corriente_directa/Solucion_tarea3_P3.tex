\section*{Enunciado del Problema}

La Fig.\ 3 muestra un circuito que combina resistencias y bater\'ias en distintas
configuraciones. Considere los sentidos de las corrientes $I_1$, $I_2$ e $I_3$, as\'i como
el recorrido de cada malla, indicados como $M_1$ y $M_2$.
\begin{enumerate}[(a)]
    \item Simplifique el circuito todo lo posible obteniendo las resistencias equivalentes
          y dibuje el esquema reducido resultante.
    \item Para el circuito reducido, escriba las ecuaciones de Kirchhoff (corriente y voltaje)
          de cada malla y del nodo relevante.
    \item Resuelva el sistema para determinar $I_1$, $I_2$ e $I_3$.
    \item Calcule la potencia disipada en cada resistencia del circuito reducido.
\end{enumerate}

\section*{(a) Simplificaci\'on del circuito}

Las resistencias de cada esquina inferior unen los mismos dos nodos, por lo que cada
terna est\'a \textbf{en paralelo}.

\medskip
\noindent\textbf{Grupo izquierdo} $\rightarrow R_{eq1}$:
\[
    \frac{1}{R_{eq1}} = \frac{1}{18}+\frac{1}{9}+\frac{1}{3}
        = \frac{1+2+6}{18} = \frac{9}{18}
    \qquad\Rightarrow\qquad
    \boxed{R_{eq1} = 2\,\Omega}
\]

\noindent\textbf{Grupo derecho} $\rightarrow R_{eq2}$:
\[
    \frac{1}{R_{eq2}} = \frac{1}{2}+\frac{1}{3}+\frac{1}{6}
        = \frac{3+2+1}{6} = \frac{6}{6}
    \qquad\Rightarrow\qquad
    \boxed{R_{eq2} = 1\,\Omega}
\]

\noindent El $4\,\Omega$ (riel superior) y el $6\,\Omega$ (rama central) se conservan. El
circuito queda con cuatro resistencias: $4\,\Omega$, $R_{eq1}=2\,\Omega$, $6\,\Omega$ y
$R_{eq2}=1\,\Omega$, seg\'un el esquema reducido:

%========================================
% (b) ECUACIONES DE KIRCHHOFF
%========================================
\section*{(b) Ecuaciones de Kirchhoff}

\noindent\textbf{KCL --- Nodo $B$:} $I_1$ e $I_2$ entran, $I_3$ sale:
\[
    \boxed{I_1 + I_2 = I_3} \tag{1}
\]

\medskip
\noindent\textbf{KVL --- Malla $M_1$} (izquierda + central). Recorri\'endola se atraviesan
$4\,\Omega$ y $R_{eq1}=2\,\Omega$ (por $I_1$), las fuentes $12\,$V y $6\,$V (como ca\'idas seg\'un
sus polaridades) y el $6\,\Omega$ central (por $I_2$):
\[
    \boxed{4I_1 - 12 + 2I_1 - 6 - 6I_2 = 0} \tag{2}
\]

\medskip
\noindent\textbf{KVL --- Malla $M_2$} (central + derecha). Se atraviesan $R_{eq2}=1\,\Omega$
(por $I_3$), la fuente $3\,$V, el $6\,\Omega$ central (por $I_2$) y la fuente $14\,$V:
\[
    \boxed{1\cdot I_3 + 3 + 6I_2 - 14 = 0} \tag{3}
\]

%========================================
% (c) RESOLUCIÓN
%========================================
\section*{(c) Resoluci\'on del sistema}

\noindent Trabajando la ecuaci\'on \textbf{(2)}:
\begin{align*}
    4I_1 - 12 + 2I_1 - 6 - 6I_2 &= 0 \\
    6I_1 - 18 - 6I_2 &= 0 \qquad\Big/\ \tfrac{1}{6} \\
    I_1 - 3 - I_2 &= 0 \;\;\Rightarrow\;\; I_1 = 3 + I_2 \tag{4}
\end{align*}

\noindent Reemplazando \textbf{(4)} en \textbf{(1)}:
\begin{align*}
    (3 + I_2) + I_2 &= I_3 \;\;\Rightarrow\;\; I_3 = 3 + 2I_2 \tag{5}
\end{align*}

\noindent Reemplazando \textbf{(5)} en \textbf{(3)}, $I_3 + 6I_2 - 11 = 0$:
\begin{align*}
    (3 + 2I_2) + 6I_2 - 11 &= 0 \\
    8I_2 - 8 &= 0
\end{align*}
\[
    \boxed{I_2 = 1\,[\text{A}]}
\]

\noindent Con $I_2 = 1$ en \textbf{(4)} y \textbf{(5)}:
\[
    I_1 = 3 + 1 \;\Rightarrow\; \boxed{I_1 = 4\,[\text{A}]}
    \qquad\qquad
    I_3 = 3 + 2(1) \;\Rightarrow\; \boxed{I_3 = 5\,[\text{A}]}
\]

\noindent\textbf{Verificaci\'on (nodo):} $\;I_1 + I_2 = 4 + 1 = 5 = I_3$ \checkmark

%========================================
% (d) POTENCIA DISIPADA
%========================================
\section*{(d) Potencia disipada en cada resistencia}

Con $P = I^2 R$ y las corrientes de cada rama:

\begin{align*}
    P_{R_{eq1}} &= I_1^2\,R_{eq1} = (4)^2 (2) = 32\,[\text{W}] \\
    P_{4\,\Omega} &= I_1^2\,(4) = (4)^2 (4) = 64\,[\text{W}] \\
    P_{6\,\Omega} &= I_2^2\,(6) = (1)^2 (6) = 6\,[\text{W}] \\
    P_{R_{eq2}} &= I_3^2\,R_{eq2} = (5)^2 (1) = 25\,[\text{W}]
\end{align*}

\[
    \boxed{P_{\text{total}} = 32 + 64 + 6 + 25 = 127\,[\text{W}]}
\]

\noindent\small\textit{Comprobaci\'on (balance de potencia):} la potencia neta entregada por las
fuentes es $\varepsilon_{\text{izq}}I_1 + \varepsilon_{\text{der}}I_3 = (12{+}6)(4) + (14{-}3)(5)
= 72 + 55 = 127\,[\text{W}]$, igual a la disipada. \checkmark
\normalsize
