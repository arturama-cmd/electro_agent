\vspace{0.5cm}

\noindent\textbf{Efecto de $C_7$:} al reemplazar el cable $b$--$d$ por el condensador
$\textcolor{newcol}{C_7}$, el nodo $b$ queda \textbf{separado} del nodo central
$M=\{c,d,e\}$ (unidos a\'un por los cables $d$--$e$ y $c$--$e$). As\'i \textbf{ya no hay
cortocircuito}: $C_2$ y $C_3$ vuelven a ser condensadores activos.

\medskip

\noindent\textbf{Reducci\'on del circuito} (todos los nodos centrales $c,d,e$ forman un solo
nodo $M$):
\[
    \underbrace{C_2 \parallel C_3 \parallel C_7}_{\text{entre } b \text{ y } M}
    \;-\;
    \underbrace{C_4 \parallel C_5}_{\text{entre } M \text{ y } g},
    \quad\text{en serie con } C_1 \text{ y } C_6.
\]
\[
    \frac{1}{C_{eq}} = \frac{1}{C_1}
    + \frac{1}{C_2 + C_3 + C_7}
    + \frac{1}{C_4 + C_5}
    + \frac{1}{C_6}
\]
