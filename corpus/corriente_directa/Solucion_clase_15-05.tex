\section*{Problema 1 -- Circuito Mixto con C\textsubscript{3} y C\textsubscript{4} Geom\'etricos}

\noindent\textbf{Enunciado:}
En el circuito de la figura, $C_1 = 6\ [\text{pF}]$, $C_5 = 4\ [\text{pF}]$, $C_2 = 8\ [\text{pF}]$.
$C_3$ es un condensador de placas paralelas de \'area $A = 18\ [\text{cm}^2]$, separaci\'on $d = 3\ [\text{mm}]$,
relleno con PVC ($\kappa = 3$). $C_4$ es un condensador cil\'indrico de largo $\ell = 5\ [\text{cm}]$,
radio interior $a = 3\ [\text{mm}]$ y radio exterior $b = 5\ [\text{mm}]$, relleno con mica ($\kappa = 4$).

\begin{enumerate}[(a)]
    \item Determine la capacitancia equivalente $C_{eq}$ del circuito.
    \item Si $V_{ab} = 24\ [\text{V}]$, determine la carga almacenada en cada condensador.
\end{enumerate}

%--- PARTE (a) ---
\section*{Parte (a) -- Capacitancia equivalente}

\noindent\textbf{Paso 1: Calcular $C_3$ y $C_4$ desde su geometr\'ia.}

\[
    C_3 = \kappa\,\varepsilon_0\,\frac{A}{d}
        = 3\cdot\varepsilon_0\cdot\frac{18\times10^{-4}\ [\text{m}^2]}{3\times10^{-3}\ [\text{m}]}
        = 3\cdot\varepsilon_0\cdot 0{,}6\ [\text{m}]
        = \mathbf{15{,}94\ [\text{pF}]}
\]

\[
    C_4 = \kappa\,\frac{2\pi\varepsilon_0\,\ell}{\ln(b/a)}
        = 4\cdot\frac{2\pi\varepsilon_0\times 0{,}05}{\ln(5/3)}
        = \mathbf{21{,}78\ [\text{pF}]}
\]

\noindent\textbf{Paso 2: Reducir $C_3$ y $C_4$ en serie $\rightarrow C_a$.}

\[
    \frac{1}{C_a} = \frac{1}{C_3} + \frac{1}{C_4}
                  = \frac{1}{15{,}94} + \frac{1}{21{,}78}
    \implies \boxed{C_a = 9{,}20\ [\text{pF}]}
\]

\noindent\textbf{Paso 3: $C_2$ en paralelo con $C_a$ $\rightarrow C_b$.}

\[
    C_b = C_2 + C_a = 8{,}00 + 9{,}20 = \boxed{17{,}20\ [\text{pF}]}
\]

\noindent\textbf{Paso 4: $C_1$, $C_b$ y $C_5$ en serie $\rightarrow C_{eq}$.}

\begin{align*}
    \frac{1}{C_{eq}} &= \frac{1}{C_1} + \frac{1}{C_b} + \frac{1}{C_5}\\[6pt]
                     &= \frac{1}{6{,}00} + \frac{1}{17{,}20} + \frac{1}{4{,}00}\ \ [\text{pF}^{-1}]
\end{align*}

\[
    \boxed{C_{eq} = 2{,}106\ [\text{pF}]}
\]

%--- PARTE (b) ---
\section*{Parte (b) -- Carga en cada condensador ($V_{ab} = 24\ [\text{V}]$)}

\noindent\textbf{Carga total:} $C_1$, $C_b$ y $C_5$ est\'an en serie $\Rightarrow$ acumulan la misma carga.

\[
    Q = C_{eq}\cdot V_{ab} = 2{,}106\times10^{-12}\times 24
\]
\[
    \boxed{q_1 = q_5 = q_b = Q = 50{,}55\ [\text{pC}]}
\]

\noindent\textbf{Voltaje sobre el bloque $C_b$:}

\[
    V_b = \frac{q_b}{C_b} = \frac{50{,}55\ [\text{pC}]}{17{,}20\ [\text{pF}]} = 2{,}939\ [\text{V}]
\]

\noindent$C_2$ y $C_a$ comparten ese voltaje (paralelo) $\Rightarrow V_2 = V_a = 2{,}939\ [\text{V}]$:

\[
    q_2 = V_b\cdot C_2 = 2{,}939 \times 8{,}00\times10^{-12}
    \implies \boxed{q_2 = 23{,}51\ [\text{pC}]}
\]

\[
    q_a = V_b\cdot C_a = 2{,}939 \times 9{,}20\times10^{-12} = 27{,}04\ [\text{pC}]
\]

$C_3$ y $C_4$ est\'an en serie dentro de $C_a$ $\Rightarrow$ comparten la misma carga:

\[
    \boxed{q_3 = q_4 = 27{,}04\ [\text{pC}]}
\]

\noindent\textbf{Verificaci\'on:} $q_2 + q_a = 23{,}51 + 27{,}04 = 50{,}55\ [\text{pC}] = q_b$ \checkmark

\noindent\textbf{Resumen:}

\begin{center}
\begin{tabular}{ccc}
\hline
Condensador & Carga $[\text{pC}]$ & Voltaje $[\text{V}]$ \\
\hline
$C_1$ & $50{,}55$ & $50{,}55/6{,}00 = 8{,}43$ \\
$C_5$ & $50{,}55$ & $50{,}55/4{,}00 = 12{,}64$ \\
$C_2$ & $23{,}51$ & $2{,}939$ \\
$C_3$ & $27{,}04$ & $27{,}04/15{,}94 = 1{,}696$ \\
$C_4$ & $27{,}04$ & $27{,}04/21{,}78 = 1{,}242$ \\
\hline
\end{tabular}
\end{center}

\noindent\small\textit{Comprobaci\'on de voltajes en serie:} $V_{C_1} + V_{C_b} + V_{C_5} = 8{,}43 + 2{,}939 + 12{,}64 \approx 24{,}0\ [\text{V}]$ \checkmark

\noindent\textit{Comprobaci\'on en $C_a$:} $V_{C_3} + V_{C_4} = 1{,}696 + 1{,}242 = 2{,}938 \approx 2{,}939\ [\text{V}]$ \checkmark
