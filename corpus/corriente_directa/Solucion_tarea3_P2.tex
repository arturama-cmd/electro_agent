\section*{Datos del Problema}

\[
    \varepsilon_1 = 10\ [\text{V}],\qquad

%========================================
% ANÁLISIS
%========================================
\section*{An\'alisis del Circuito}

El circuito posee \textbf{dos nodos} ($c$ arriba y $d$ abajo) y \textbf{tres ramas en paralelo}:

\begin{itemize}
    \item \textbf{Rama izquierda} ($I_1\uparrow$): \textcolor{colR1}{$R_1 = 3\,[\Omega]$}
          en serie con \textcolor{colE1}{$\varepsilon_1 = 10\,[\text{V}]$} (polo $+$ arriba).
    \item \textbf{Rama central} ($I_2\downarrow$): \textcolor{colR2}{$R_2 = 2\,[\Omega]$}
          en serie con \textcolor{colE2}{$\varepsilon_2 = 5\,[\text{V}]$} (polo $+$ arriba).
    \item \textbf{Rama derecha} ($I_3\downarrow$): \'unicamente
          \textcolor{colR3}{$R_3 = 7\,[\Omega]$}.
\end{itemize}

Se identifican \textbf{dos mallas independientes}: $M_1$ (ramas izquierda + central)
y $M_2$ (ramas central + derecha).

\textbf{KCL --- Nodo $c$:}

La corriente $I_1$ llega al nodo $c$ y se divide en $I_2$ e $I_3$:
\[
    \boxed{I_1 = I_2 + I_3} \tag{N}
\]

\medskip
\textbf{KVL --- Malla $M_1$} (sube izquierda, baja central):
\[
    -R_1 I_1 + \varepsilon_1 - \varepsilon_2 - R_2 I_2 = 0
\]
\[
    \boxed{3\,I_1 + 2\,I_2 = 5} \tag{M1}
\]

\medskip
\textbf{KVL --- Malla $M_2$} (sube central, baja derecha):
\[
    R_2 I_2 + \varepsilon_2 - R_3 I_3 = 0
\]
\[
    \boxed{7\,I_3 - 2\,I_2 = 5} \tag{M2}
\]
\end{minipage}

%----------------------------------------
\subsection*{Resoluci\'on del sistema}

De \textbf{(M1)} se despeja $I_1$ en funci\'on de $I_2$:
\[
    I_1 = \frac{5 - 2\,I_2}{3} \tag{I}
\]

Sustituyendo (I) en la ecuaci\'on de nodo \textbf{(N)}, $I_1 = I_2 + I_3$:
\begin{align*}
    \frac{5 - 2\,I_2}{3} &= I_2 + I_3 \\[4pt]
    5 - 2\,I_2 &= 3\,I_2 + 3\,I_3 \\[4pt]
    5 &= 5\,I_2 + 3\,I_3 \tag{II}
\end{align*}

De \textbf{(M2)}: $I_3 = \dfrac{5 + 2\,I_2}{7}$.\ \ Sustituyendo en \textbf{(II)}:
\begin{align*}
    5 &= 5\,I_2 + \frac{3\,(5 + 2\,I_2)}{7} \\[4pt]
    35 &= 35\,I_2 + 15 + 6\,I_2 \\[4pt]
    41\,I_2 &= 20
\end{align*}

\[
    \boxed{I_2 = \frac{20}{41} \approx 0{,}488\ [\text{A}]}
\]

Con (M2) y (I) se obtienen las corrientes restantes:
\[
    I_3 = \frac{5 + \tfrac{40}{41}}{7} = \frac{245/41}{7}
    \qquad\Rightarrow\qquad
    \boxed{I_3 = \frac{35}{41} \approx 0{,}854\ [\text{A}]}
\]
\[
    I_1 = \frac{5 - \tfrac{40}{41}}{3} = \frac{165/41}{3}
    \qquad\Rightarrow\qquad
    \boxed{I_1 = \frac{55}{41} \approx 1{,}341\ [\text{A}]}
\]

\noindent\textbf{Verificaci\'on (N):}
$\;I_2 + I_3 = \dfrac{20}{41} + \dfrac{35}{41} = \dfrac{55}{41} = I_1$ \checkmark

%----------------------------------------
\subsection*{Voltajes y potencias en cada resistencia}

\[
    V_{R_i} = I_i\,R_i \qquad\qquad P_{R_i} = I_i^2\,R_i
\]

\begin{align*}
    V_1 &= \frac{55}{41}\times 3 = \frac{165}{41} \approx 4{,}02\ [\text{V}]
    &\quad
    P_1 &= \left(\frac{55}{41}\right)^{\!2}\times 3 \approx 5{,}40\ [\text{W}]\\[6pt]
    V_2 &= \frac{20}{41}\times 2 = \frac{40}{41} \approx 0{,}98\ [\text{V}]
    &\quad
    P_2 &= \left(\frac{20}{41}\right)^{\!2}\times 2 \approx 0{,}48\ [\text{W}]\\[6pt]
    V_3 &= \frac{35}{41}\times 7 = \frac{245}{41} \approx 5{,}98\ [\text{V}]
    &\quad
    P_3 &= \left(\frac{35}{41}\right)^{\!2}\times 7 \approx 5{,}10\ [\text{W}]
\end{align*}

\noindent\small\textit{Comprobaci\'on (balance de potencia):}
$\;\varepsilon_1 I_1 - \varepsilon_2 I_2
  = 10\times\tfrac{55}{41} - 5\times\tfrac{20}{41}
  = \tfrac{450}{41} \approx 10{,}98\ [\text{W}]
  = P_1 + P_2 + P_3$\ \checkmark
\normalsize

\medskip
\noindent\textbf{Nota f\'isica:} Todos los valores son positivos: los sentidos supuestos
son correctos. La fuente \textcolor{colE2}{$\varepsilon_2 = 5\ [\text{V}]$} est\'a siendo
\emph{cargada} por \textcolor{colE1}{$\varepsilon_1 = 10\ [\text{V}]$}: la corriente $I_2$
entra por su polo positivo, de modo que absorbe potencia
$P_{\varepsilon_2} = \varepsilon_2\,I_2 \approx 2{,}44\ [\text{W}]$.
La fuente mayor entrega $P_{\varepsilon_1} = \varepsilon_1\,I_1 \approx 13{,}42\ [\text{W}]$;
la diferencia ($\approx 10{,}98\ [\text{W}]$) se disipa en las tres resistencias.
