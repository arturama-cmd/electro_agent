\section*{Enunciado del Problema}

Considere el circuito de corriente continua de la figura, con dos fuentes id\'enticas
$\varepsilon_1 = \varepsilon_2 = \varepsilon = 24\ [\text{V}]$ y tres resistencias
$R_1 = 3\ [\Omega]$, $R_2 = 2\ [\Omega]$, $R_3 = 7\ [\Omega]$.
Los sentidos supuestos de las corrientes se indican en la figura.
Determine:

\begin{enumerate}[(a)]
    \item Las corrientes $I_1$, $I_2$ e $I_3$ que circulan por cada rama.
    \item La potencia total disipada en las resistencias.
    \item La diferencia de potencial $V_{ab} = V_a - V_b$ entre los puntos $a$ y $b$.
\end{enumerate}

\section*{An\'alisis del Circuito}

El circuito posee \textbf{dos nodos} ($c$ arriba y $d$ abajo; n\'otese que $c$ y $a$ est\'an
unidos por cable, igual que $d$ y $b$) y \textbf{tres ramas} en paralelo:

\begin{itemize}
    \item \textbf{Rama izquierda} ($I_1 \uparrow$): fuente \textcolor{colE1}{$\varepsilon_1$}
          y resistencia \textcolor{colR1}{$R_1$}.
    \item \textbf{Rama central} ($I_2 \downarrow$): fuente \textcolor{colE2}{$\varepsilon_2$}
          y resistencia \textcolor{colR2}{$R_2$}.
    \item \textbf{Rama derecha} ($I_3 \downarrow$): resistencia \textcolor{colR3}{$R_3$}.
\end{itemize}

Se identifican \textbf{dos mallas independientes}: la malla izquierda $M_1$ (ramas izquierda +
central) y la malla derecha $M_2$ (ramas central + derecha).

\textbf{KCL -- Nodo $c$:}

La corriente $I_1$ llega al nodo $c$ y se divide en $I_2$ e $I_3$:
\[
    \boxed{I_1 = I_2 + I_3} \tag{N}
\]

\medskip
\textbf{KVL -- Malla $M_1$} (izquierda + central):

Recorriendo: subiendo por $R_1$ y $\varepsilon_1$, bajando por $\varepsilon_2$ y $R_2$:
\[
    \varepsilon_1 - R_1 I_1 - \varepsilon_2 - R_2 I_2 = 0
\]
Como $\varepsilon_1 = \varepsilon_2$, las fuentes se cancelan:
\[
    \boxed{-3\,I_1 - 2\,I_2 = 0 \;\Rightarrow\; 3\,I_1 + 2\,I_2 = 0} \tag{M1}
\]

\medskip
\textbf{KVL -- Malla $M_2$} (central + derecha):

Recorriendo: subiendo por $R_2$ y $\varepsilon_2$, bajando por $R_3$:
\[
    R_2 I_2 + \varepsilon_2 - R_3 I_3 = 0
\]
\[
    \boxed{2\,I_2 + 24 - 7\,I_3 = 0 \;\Rightarrow\; 2\,I_2 - 7\,I_3 = -24} \tag{M2}
\]
\end{minipage}

%----------------------------------------
\subsection*{b) Resoluci\'on del sistema}

De \textbf{(M1)} se despeja $I_1$ en funci\'on de $I_2$:
\[
    3\,I_1 = -2\,I_2 \qquad\Rightarrow\qquad I_1 = -\frac{2}{3}\,I_2 \tag{I}
\]

Sustituyendo (I) en la ecuaci\'on de nodos \textbf{(N)}, $I_1 = I_2 + I_3$:
\begin{align*}
    -\frac{2}{3}\,I_2 &= I_2 + I_3 \\[4pt]
    I_3 &= -\frac{2}{3}\,I_2 - I_2 = -\frac{5}{3}\,I_2 \tag{II}
\end{align*}

Sustituyendo (II) en \textbf{(M2)}, $2\,I_2 - 7\,I_3 = -24$:
\begin{align*}
    2\,I_2 - 7\left(-\frac{5}{3}\,I_2\right) &= -24 \\[4pt]
    2\,I_2 + \frac{35}{3}\,I_2 &= -24 \\[4pt]
    \frac{6\,I_2 + 35\,I_2}{3} &= -24 \\[4pt]
    \frac{41}{3}\,I_2 &= -24
\end{align*}

\[
    \boxed{I_2 = -\frac{72}{41} \approx -1{,}76\ [\text{A}]}
\]

Con (I) y (II) se obtienen las otras corrientes:
\[
    I_1 = -\frac{2}{3}\left(-\frac{72}{41}\right) = \frac{48}{41}
    \qquad\Rightarrow\qquad
    \boxed{I_1 = \frac{48}{41} \approx 1{,}17\ [\text{A}]}
\]
\[
    I_3 = -\frac{5}{3}\left(-\frac{72}{41}\right) = \frac{120}{41}
    \qquad\Rightarrow\qquad
    \boxed{I_3 = \frac{120}{41} \approx 2{,}93\ [\text{A}]}
\]

\noindent\textbf{Verificaci\'on (N):}
$\;I_2 + I_3 = -\dfrac{72}{41} + \dfrac{120}{41} = \dfrac{48}{41} = I_1$ \checkmark

%----------------------------------------
\subsection*{b) Potencia total disipada}

La potencia disipada en cada resistencia es $P = I^2 R$:

\begin{align*}
    P_1 &= I_1^2 R_1 = (1{,}17)^2 \times 3 \approx 4{,}11\ [\text{W}] \\
    P_2 &= I_2^2 R_2 = (1{,}76)^2 \times 2 \approx 6{,}17\ [\text{W}] \\
    P_3 &= I_3^2 R_3 = (2{,}93)^2 \times 7 \approx 59{,}96\ [\text{W}]
\end{align*}

\[
    \boxed{P_{\text{total}} = P_1 + P_2 + P_3 \approx 70{,}24\ [\text{W}]}
\]

\noindent\small\textit{Comprobaci\'on (balance de potencia):} la potencia entregada por las fuentes
es $\varepsilon_1 I_1 + \varepsilon_2 |I_2| = 24(1{,}17) + 24(1{,}76) \approx 70{,}24\ [\text{W}]$,
igual a la disipada. \checkmark
\normalsize

%----------------------------------------
\subsection*{c) Diferencia de potencial $V_{ab}$}

La corriente $I_3$ circula de $a$ hacia $b$ a trav\'es de $R_3$, por lo que:
\[
    V_{ab} = V_a - V_b = I_3 \, R_3 = \frac{120}{41}\times 7 = \frac{840}{41}
\]
\[
    \boxed{V_{ab} \approx 20{,}49\ [\text{V}]}
\]

\noindent Tambi\'en puede verificarse por la rama izquierda:
$V_{ab} = \varepsilon_1 - I_1 R_1 = 24 - (1{,}17)(3) \approx 20{,}49\ [\text{V}]$ \checkmark
