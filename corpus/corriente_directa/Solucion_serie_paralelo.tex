\section*{Datos del Problema}

\[
    \varepsilon = 12\ [\text{V}],\qquad
    R_1 = 20\ [\Omega],\qquad
    R_2 = 20\ [\Omega],\qquad
    R_3 = 30\ [\Omega],\qquad
    R_4 = 8{,}0\ [\Omega]
\]

%========================================
% ANÁLISIS
%========================================
\section*{An\'alisis del Circuito}

El circuito \textbf{no requiere mallas}: se resuelve por \textbf{reducci\'on serie-paralelo}.
Partiendo de la fuente, la corriente recorre:

\begin{itemize}
    \item \textbf{\textcolor{colR1}{$R_1$}} en serie, desde la fuente hasta el nodo $b$.
    \item En el nodo $b$ la corriente \textbf{se divide} hacia el nodo $c$ por dos caminos:
          \begin{itemize}
              \item la \textbf{\textcolor{colR2}{$R_2$}} (rama central, $I_2 \downarrow$),
              \item la \textbf{\textcolor{colR3}{$R_3$}} ($b \to$ lado derecho $\to c$, $I_3$).
          \end{itemize}
          Como ambas conectan los mismos nodos $b$ y $c$, est\'an \textbf{en paralelo}:
          $R_2 \parallel R_3$.
    \item En $c$ las corrientes se recombinan y atraviesan \textbf{\textcolor{colR4}{$R_4$}}
          en serie, de regreso a la fuente.
\end{itemize}

\noindent En consecuencia, la resistencia equivalente es
$R_{\text{eq}} = R_1 + (R_2 \parallel R_3) + R_4$.

\vspace{6pt}

\textbf{Paralelo $R_2 \parallel R_3$:}
\[
    R_{23} = \frac{R_2\,R_3}{R_2 + R_3}
           = \frac{(20)(30)}{20 + 30}
           = \frac{600}{50}
\]
\[
    \boxed{R_{23} = 12\ [\Omega]}
\]

\medskip
\textbf{Resistencia equivalente (serie):}
\[
    R_{\text{eq}} = R_1 + R_{23} + R_4 = 20 + 12 + 8 = \boxed{40\ [\Omega]}
\]

\medskip
\textbf{Corriente total} (ley de Ohm):
\[
    I_1 = \frac{\varepsilon}{R_{\text{eq}}} = \frac{12}{40}
        \;\Rightarrow\; \boxed{I_1 = 0{,}30\ [\text{A}]}
\]
Esta corriente circula por $R_1$ y por $R_4$ (est\'an en serie con la fuente).
\end{minipage}

%----------------------------------------
\subsection*{b) Corrientes en las ramas paralelas}

La ca\'ida de tensi\'on entre los nodos $b$ y $c$ es la misma para $R_2$ y $R_3$, e igual
a la que cae en la resistencia equivalente $R_{23}$:
\[
    V_{bc} = I_1\,R_{23} = (0{,}30)(12) = 3{,}6\ [\text{V}]
\]

Aplicando la ley de Ohm en cada rama:
\[
    I_2 = \frac{V_{bc}}{R_2} = \frac{3{,}6}{20}
        \;\Rightarrow\; \boxed{I_2 = 0{,}18\ [\text{A}]}
    \qquad\qquad
    I_3 = \frac{V_{bc}}{R_3} = \frac{3{,}6}{30}
        \;\Rightarrow\; \boxed{I_3 = 0{,}12\ [\text{A}]}
\]

\noindent\textbf{Verificaci\'on (nodo $b$):}
$\;I_2 + I_3 = 0{,}18 + 0{,}12 = 0{,}30 = I_1$ \checkmark

%----------------------------------------
\subsection*{c) Potencia disipada en cada resistencia}

La potencia disipada en cada resistencia es $P = I^2 R$:

\begin{align*}
    P_1 &= I_1^2\,R_1 = (0{,}30)^2 (20) = 1{,}80\ [\text{W}] \\
    P_2 &= I_2^2\,R_2 = (0{,}18)^2 (20) = 0{,}648\ [\text{W}] \\
    P_3 &= I_3^2\,R_3 = (0{,}12)^2 (30) = 0{,}432\ [\text{W}] \\
    P_4 &= I_1^2\,R_4 = (0{,}30)^2 (8{,}0) = 0{,}72\ [\text{W}]
\end{align*}

\[
    \boxed{P_{\text{total}} = P_1 + P_2 + P_3 + P_4 = 3{,}6\ [\text{W}]}
\]

\noindent\small\textit{Comprobaci\'on (balance de potencia):} la potencia entregada por la
fuente es $P = \varepsilon\,I_1 = (12)(0{,}30) = 3{,}6\ [\text{W}]$, igual a la disipada
por las resistencias. \checkmark
\normalsize
